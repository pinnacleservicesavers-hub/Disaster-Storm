## 🎯 Usage Example

```typescript
import { 
  apiFetch, 
  getImpact, 
  getLegendUrl,
  defaultBaseUrl 
} from '@disaster-direct/sdk';

// API calls with retries
const data = await apiFetch('/api/claims');

// Get impact score
const impact = await getImpact(defaultBaseUrl, 33.749, -84.388, true);

// Generate legend URL
const legendUrl = await getLegendUrl(defaultBaseUrl, {
  scheme: 'viridis',
  width: 300,
  height: 60,
  bg: 'transparent',
  fmt: 'svg'
}, false);
```
