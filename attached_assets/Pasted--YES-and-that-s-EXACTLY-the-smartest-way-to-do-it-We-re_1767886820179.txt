import fetch from "node-fetch";

const TREE_SERVICE_TARGETS = [
  // Contract / recurring targets
  "property management company",
  "apartment complex",
  "homeowners association",
  "shopping center",
  "commercial property",
  "real estate agency",
  "construction company",
  "general contractor",
  // Emergency / storm cleanup targets
  "restoration company",
  "insurance agency",
  // Municipal work targets
  "city hall",
  "public works department",
];

async function geocodeLocation(location) {
  const url = `https://maps.googleapis.com/maps/api/geocode/json?address=${encodeURIComponent(
    location
  )}&key=${process.env.GOOGLE_API_KEY}`;

  const res = await fetch(url);
  const data = await res.json();

  if (!data.results?.length) throw new Error("Location not found.");
  return data.results[0].geometry.location; // {lat, lng}
}

async function placesTextSearch(query, lat, lng, radiusMeters) {
  const url = `https://maps.googleapis.com/maps/api/place/textsearch/json?query=${encodeURIComponent(
    query
  )}&location=${lat},${lng}&radius=${radiusMeters}&key=${process.env.GOOGLE_API_KEY}`;

  const res = await fetch(url);
  const data = await res.json();
  return data.results || [];
}

export async function leadHunter({ location, radius }) {
  // Convert miles → meters
  const radiusMeters = Math.min(Math.floor(radius * 1609.34), 50000); // cap at 50km
  const { lat, lng } = await geocodeLocation(location);

  let allLeads = [];

  for (const keyword of TREE_SERVICE_TARGETS) {
    const query = `${keyword} near ${location}`;
    const results = await placesTextSearch(query, lat, lng, radiusMeters);

    const leads = results.map((r) => ({
      external_id: r.place_id,
      company_name: r.name,
      category: r.types?.[0] || "business",
      phone: null, // requires Place Details call (optional next)
      website: null, // requires Place Details call (optional next)
      address: r.formatted_address || r.vicinity || "",
      rating: r.rating || null,
      reviewCount: r.user_ratings_total || 0,
      lat: r.geometry?.location?.lat,
      lng: r.geometry?.location?.lng,
      source: "google_places",
    }));

    allLeads.push(...leads);
  }

  // Remove duplicates by external_id
  const seen = new Set();
  allLeads = allLeads.filter((l) => {
    if (seen.has(l.external_id)) return false;
    seen.add(l.external_id);
    return true;
  });

  return allLeads.slice(0, 40); // cap to 40 results per search
}

import fetch from "node-fetch";

export async function getPlaceDetails(placeId) {
  const fields = [
    "formatted_phone_number",
    "website",
    "opening_hours",
    "name",
  ].join(",");

  const url = `https://maps.googleapis.com/maps/api/place/details/json?place_id=${placeId}&fields=${fields}&key=${process.env.GOOGLE_API_KEY}`;

  const res = await fetch(url);
  const data = await res.json();

  const d = data.result || {};
  return {
    phone: d.formatted_phone_number || null,
    website: d.website || null,
    hours: d.opening_hours?.weekday_text || null,
  };
}

import express from "express";
import { authKey } from "../middleware/authKey.js";
import { leadHunter } from "../services/leadHunter.js";
import { leadScorer } from "../services/leadScorer.js";
import { competitionDetector } from "../services/competitionDetector.js";
import { getPlaceDetails } from "../services/placeDetails.js";

const router = express.Router();

router.post("/search", authKey, async (req, res) => {
  const { location, radius } = req.body;

  const rawLeads = await leadHunter({ location, radius });

  const leads = await Promise.all(rawLeads.map(async (lead, i) => {
    // Enrich only first 12 leads to control API cost
    let details = {};
    if (i < 12) {
      details = await getPlaceDetails(lead.external_id);
    }

    const scoreData = leadScorer({ ...lead, ...details });
    const comp = await competitionDetector(lead, location, radius);

    return { ...lead, ...details, ...scoreData, ...comp };
  }));

  res.json({ leads });
});

export default router;
