export function buildFollowUpTasks({ user, leadId, outreachMessageId, outreachPack }) {
  const now = new Date();

  // Helper to schedule date by days from now
  const daysFromNow = (days) => {
    const d = new Date(now);
    d.setDate(d.getDate() + days);
    return d;
  };

  // Use the followups already generated by the AI
  const followups = outreachPack.followups || [];

  // Ensure followups array has 4 items (fallback safety)
  const safeFollowups = [
    followups[0] || "Just checking in — we can handle hazard trees, trimming, and storm cleanup. Want a quick quote this week?",
    followups[1] || "We’re booking maintenance contracts for property managers & HOAs this month. Want pricing options?",
    followups[2] || "We have an opening this week for inspections and quotes. Want me to stop by and assess any hazard trees?",
    followups[3] || "If you ever need emergency storm response, we can respond quickly. Want me to send our contact info?",
  ];

  return [
    {
      user_id: user.id,
      lead_id: leadId,
      outreach_message_id: outreachMessageId,
      channel: "sms",
      task_title: "Follow-up #1 (Day 2)",
      message_text: safeFollowups[0],
      due_at: daysFromNow(2),
      status: "pending",
    },
    {
      user_id: user.id,
      lead_id: leadId,
      outreach_message_id: outreachMessageId,
      channel: "sms",
      task_title: "Follow-up #2 (Day 5)",
      message_text: safeFollowups[1],
      due_at: daysFromNow(5),
      status: "pending",
    },
    {
      user_id: user.id,
      lead_id: leadId,
      outreach_message_id: outreachMessageId,
      channel: "sms",
      task_title: "Follow-up #3 (Day 10)",
      message_text: safeFollowups[2],
      due_at: daysFromNow(10),
      status: "pending",
    },
    {
      user_id: user.id,
      lead_id: leadId,
      outreach_message_id: outreachMessageId,
      channel: "sms",
      task_title: "Re-engage (Day 20)",
      message_text: safeFollowups[3],
      due_at: daysFromNow(20),
      status: "pending",
    },
  ];
}

    contractorOnly(user);

"use client";
import { useEffect, useState } from "react";

export default function FollowUpWidget() {
  const [tasks, setTasks] = useState([]);

  useEffect(() => {
    async function load() {
      const res = await fetch("/api/leadvault/followups/today");
      const data = await res.json();
      setTasks(data.tasks || []);
    }
    load();
  }, []);

  if (!tasks.length) {
    return (
      <div className="rounded-2xl border p-4 shadow-sm">
        <h2 className="text-lg font-semibold">Follow-ups Due Today</h2>
        <p className="text-sm text-gray-500 mt-2">No follow-ups due today. Keep pushing hot leads!</p>
      </div>
    );
  }

  return (
    <div className="rounded-2xl border p-4 shadow-sm space-y-3">
      <h2 className="text-lg font-semibold">Follow-ups Due Today</h2>
      <div className="space-y-3">
        {tasks.map((t) => (
          <div key={t.id} className="rounded-xl bg-gray-50 p-3">
            <p className="text-sm font-semibold">{t.task_title}</p>
            <p className="text-xs text-gray-500">Channel: {t.channel.toUpperCase()}</p>
            <p className="text-sm text-gray-700 mt-2">{t.message_text}</p>

            <div className="flex gap-3 mt-3">
              <button className="text-sm underline">Send</button>
              <button className="text-sm underline">Mark Complete</button>
              <button className="text-sm underline">Skip</button>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}
