💎 PRODUCTION-GRADE BACKEND CODE (FastAPI)

Here’s your high-end backend starter:

# main.py
from fastapi import FastAPI, UploadFile, File
from pydantic import BaseModel
import openai
import moviepy.editor as mp
import os

app = FastAPI()

class StoryInput(BaseModel):
    story_text: str
    industry: str  # roofing, tree removal, mechanic, etc.

@app.post("/generate-movie/")
async def generate_movie(story: StoryInput, image: UploadFile = File(...)):

    # 1. Enhance Story to Cinematic Script
    cinematic_script = openai.ChatCompletion.create(
        model="gpt-4o",
        messages=[
            {"role": "system", "content": "Turn this into a Hollywood movie trailer script."},
            {"role": "user", "content": story.story_text}
        ]
    )

    script_text = cinematic_script["choices"][0]["message"]["content"]

    # 2. Save Uploaded Image
    image_path = f"uploads/{image.filename}"
    with open(image_path, "wb") as f:
        f.write(await image.read())

    # 3. Generate Cinematic Voiceover (ElevenLabs API example placeholder)
    os.system(f"curl -X POST https://api.elevenlabs.io/v1/text-to-speech ...")

    # 4. Combine into Video
    clip = mp.ImageClip(image_path).set_duration(10)
    audio = mp.AudioFileClip("voiceover.mp3")
    final = clip.set_audio(audio)

    final.write_videofile("final_movie.mp4", fps=24)

    return {"status": "Movie Generated", "script": script_text}
