// index.js - ATTOM-first proxy with county fallback (Muscogee GA & Lee AL) + Regrid fallback
require('dotenv').config();
const express = require('express');
const fetch = require('node-fetch');
const NodeCache = require('node-cache');
const helmet = require('helmet');
const morgan = require('morgan');
const { JSDOM } = require('jsdom'); // small DOM parser for HTML fallback parsing

const app = express();
app.use(express.json());
app.use(helmet());
app.use(morgan('combined'));

const cache = new NodeCache({ stdTTL: 60 * 60, checkperiod: 120 }); // 1 hour default

const PROVIDER = (process.env.PROVIDER || 'attom').toLowerCase();
const ATTOM_API_KEY = process.env.ATTOM_API_KEY || '';
const REGRID_API_KEY = process.env.REGRID_API_KEY || ''; // optional, paid fallback (recommended)
const WHITEPAGES_API_KEY = process.env.WHITEPAGES_API_KEY || '';
const PORT = process.env.PORT || 3000;

// tiny helper
async function cached(key, ttlSeconds, fn) {
  const v = cache.get(key);
  if (v) return v;
  const val = await fn();
  cache.set(key, val, ttlSeconds);
  return val;
}
function safeGet(o, path, fallback=null) {
  try { return path.split('.').reduce((a,p)=>a && a[p]!==undefined ? a[p] : undefined, o) ?? fallback; } catch(e) { return fallback; }
}

/* ----------------------------
   ATTOM: property detail by address
   ---------------------------- */
async function fetchAttom(address) {
  if (!ATTOM_API_KEY) throw new Error('ATTOM_API_KEY not set');
  const url = `https://api.gateway.attomdata.com/propertyapi/v1.0.0/property/detail?address=${encodeURIComponent(address)}`;
  const r = await fetch(url, { headers: { 'accept': 'application/json', 'apikey': ATTOM_API_KEY }});
  if (!r.ok) {
    const txt = await r.text().catch(()=>String(r.status));
    throw new Error(`ATTOM error ${r.status}: ${txt}`);
  }
  return r.json();
}
function normalizeAttom(raw) {
  const out = { provider:'attom', raw, address:null, ownerNames:[], ownerMailing:null, parcelId:null, lastSale:null, assessedValue:null, lat:null, lon:null };
  const prop = safeGet(raw,'property.0') || safeGet(raw,'property') || raw;
  out.address = safeGet(prop,'address.address1') || safeGet(prop,'address.line') || null;
  out.parcelId = safeGet(prop,'identifiers.parcelNumber') || safeGet(prop,'parcel_number') || null;
  // owners
  const owners = safeGet(raw,'owners') || safeGet(prop,'owners') || [];
  if (Array.isArray(owners) && owners.length) {
    owners.forEach(o => {
      const n = o.name || o.ownerName || `${o.firstName||''} ${o.lastName||''}`.trim();
      if (n) out.ownerNames.push(n);
      if (!out.ownerMailing) out.ownerMailing = o.mailingAddress || o.mailing_address || out.ownerMailing;
    });
  } else {
    const fallbackOwner = safeGet(prop,'owner') || safeGet(raw,'owner');
    if (fallbackOwner) {
      const n = fallbackOwner.name || fallbackOwner.ownerName || null;
      if (n) out.ownerNames.push(n);
      out.ownerMailing = fallbackOwner.mailingAddress || fallbackOwner.mailing_address || out.ownerMailing;
    }
  }
  out.lastSale = safeGet(prop,'lastSale') || safeGet(prop,'sales.0') || null;
  out.assessedValue = safeGet(prop,'assessment.assessedValue') || null;
  out.lat = safeGet(prop,'location.latitude') || safeGet(prop,'address.latitude') || null;
  out.lon = safeGet(prop,'location.longitude') || safeGet(prop,'address.longitude') || null;
  return out;
}

/* ----------------------------
   Regrid fallback (recommended paid fallback)
   Regrid has a simple parcel lookup API (requires account)
   Docs: https://www.regrid.com/docs (signup required)
   ---------------------------- */
async function fetchRegridByAddress(address) {
  if (!REGRID_API_KEY) throw new Error('REGRID_API_KEY not set');
  // Regrid search endpoint — their "property" lookup supports address param
  const url = `https://api.regrid.com/v1/parcels/search?address=${encodeURIComponent(address)}&api_key=${encodeURIComponent(REGRID_API_KEY)}`;
  const r = await fetch(url);
  if (!r.ok) {
    const t = await r.text().catch(()=>String(r.status));
    throw new Error(`Regrid error ${r.status}: ${t}`);
  }
  return r.json();
}
function normalizeRegrid(raw) {
  const out = { provider:'regrid', raw, address:null, ownerNames:[], parcelId:null, lat:null, lon:null };
  const first = Array.isArray(raw.items) && raw.items.length ? raw.items[0] : raw;
  out.address = safeGet(first,'address.address_line') || safeGet(first,'address') || null;
  out.parcelId = safeGet(first,'parcel_number') || safeGet(first,'id') || null;
  out.lat = safeGet(first,'geometry.coordinates.1') || safeGet(first,'lat') || null;
  out.lon = safeGet(first,'geometry.coordinates.0') || safeGet(first,'lon') || null;
  // Regrid may include owner in properties
  if (first && first.owner) {
    out.ownerNames = Array.isArray(first.owner) ? first.owner : [first.owner];
  }
  return out;
}

/* ----------------------------
   County fallback: Muscogee County (Columbus, GA)
   Public site: https://publicaccess.columbusga.gov/
   NOTE: county pages are HTML; we use a best-effort HTML parse.
   This is fragile: if the site uses heavy JS or blocks automated calls, this may not work.
   If you need guaranteed data, request county data access or use Regrid/ATTOM.
   ---------------------------- */
async function fetchMuscogeeCountyByAddress(address) {
  // best-effort: Muscogee publicaccess supports address search via query param 'searchstring' and 'searchtype=ADDRESS'
  // Example UI path (observed): /search/commonsearch.aspx?mode=address
  // We'll attempt a GET to the search results page and parse owner/parcel from the HTML.
  const q = encodeURIComponent(address);
  const url = `https://publicaccess.columbusga.gov/search/commonsearch.aspx?mode=search&searchtype=ADDRESS&searchstring=${q}`;
  const r = await fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0 (compatible; StrategicLandBot/1.0)' }});
  if (!r.ok) {
    const t = await r.text().catch(()=>String(r.status));
    throw new Error(`Muscogee publicaccess error ${r.status}: ${t}`);
  }
  const html = await r.text();
  // parse with JSDOM - look for common table rows that contain "Owner" or "Parcel"
  const dom = new JSDOM(html);
  const doc = dom.window.document;
  // Attempt common QPublic/iASWorld results table parsing
  // 1) Look for an element containing "Owner" or "Parcel" label
  const possible = [];
  const text = doc.body.textContent || '';
  // quick regex for owner and parcel
  const ownerMatch = text.match(/Owner(?:s)?:\s*([\s\S]{1,120}?)\s{2,}/i);
  const parcelMatch = text.match(/Parcel(?:\s|#|Id|Number)[:\s]*([A-Za-z0-9\-\.]+)/i);
  const addressMatch = text.match(/Address[:\s]*([\s\S]{1,120}?)\s{2,}/i);
  const owner = ownerMatch ? ownerMatch[1].trim() : null;
  const parcel = parcelMatch ? parcelMatch[1].trim() : null;
  const foundAddress = addressMatch ? addressMatch[1].trim() : null;

  return { provider:'muscogee_county', rawHtmlSnippet: html.slice(0,2000), ownerNames: owner ? [owner] : [], parcelId: parcel || null, address: foundAddress || address };
}

/* ----------------------------
   County fallback: Lee County, AL
   Several public portals exist (AlabamaGIS / vendor websites). We'll attempt Lee's 'alabamagis' GIS frameset or the Deltacomputer Systems search engine.
   This is best-effort: many county portals require JS or form posting.
   ---------------------------- */
async function fetchLeeCountyByAddress(address) {
  // Try Lee County Deltacomputersystems public engine (often used for Alabama counties)
  // This is a defensive attempt and may need updates if site uses JS.
  const q = encodeURIComponent(address);
  const url = `https://www.deltacomputersystems.com/al/al43/PropertySearch.aspx?SearchType=Address&Address=${q}`; // best-effort guess
  const r = await fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0 (compatible; StrategicLandBot/1.0)' }});
  if (!r.ok) {
    // Fallback: try Regrid if available (recommended)
    if (REGRID_API_KEY) {
      const re = await fetchRegridByAddress(address);
      return normalizeRegrid(re);
    }
    const t = await r.text().catch(()=>String(r.status));
    throw new Error(`Lee County search error ${r.status}: ${t}`);
  }
  const html = await r.text();
  const dom = new JSDOM(html);
  const text = dom.window.document.body.textContent || '';
  const ownerMatch = text.match(/Owner(?:s)?:\s*([\s\S]{1,120}?)\s{2,}/i);
  const parcelMatch = text.match(/Parcel(?:\s|#|Id|Number)[:\s]*([A-Za-z0-9\-\.]+)/i);
  const owner = ownerMatch ? ownerMatch[1].trim() : null;
  const parcel = parcelMatch ? parcelMatch[1].trim() : null;
  return { provider:'lee_county', rawHtmlSnippet: html.slice(0,2000), ownerNames: owner ? [owner] : [], parcelId: parcel || null, address: address };
}

/* ----------------------------
   Master property endpoint: try ATTOM -> Regrid -> County fallbacks
   Returns a single normalized object with providerUsed field and helpful notes.
   ---------------------------- */
app.get('/api/property', async (req, res) => {
  try {
    const address = (req.query.address || '').trim();
    if (!address) return res.status(400).json({ error: 'address query required (e.g. ?address=5385 Westwood Drive Columbus GA)' });

    const cacheKey = `prop:${address}`;
    const result = await cached(cacheKey, 60 * 60 * 24, async () => {
      // 1) ATTOM primary
      try {
        const attomRaw = await fetchAttom(address);
        const normalized = normalizeAttom(attomRaw);
        // if we found owner or parcel, return it
        if ((normalized.ownerNames && normalized.ownerNames.length) || normalized.parcelId) {
          return { provider: 'attom', data: normalized };
        }
      } catch (e) {
        // log and continue to fallback
        console.warn('ATTOM lookup failed (continuing to fallback)', e.message);
      }

      // 2) Regrid fallback (if configured) - more reliable than HTML scraping
      if (REGRID_API_KEY) {
        try {
          const r = await fetchRegridByAddress(address);
          const norm = normalizeRegrid(r);
          if ((norm.ownerNames && norm.ownerNames.length) || norm.parcelId) {
            return { provider: 'regrid', data: norm };
          }
        } catch (e) {
          console.warn('Regrid lookup failed', e.message);
        }
      }

      // 3) County-specific: try Muscogee if address is in Columbus GA (best-effort)
      if (/columbus/i.test(address) || /ga\b/i.test(address) || /muscogee/i.test(address)) {
        try {
          const mus = await fetchMuscogeeCountyByAddress(address);
          if ((mus.ownerNames && mus.ownerNames.length) || mus.parcelId) {
            return { provider: 'muscogee_county', data: mus };
          }
        } catch (e) {
          console.warn('Muscogee county lookup failed', e.message);
        }
      }

      // 4) Try Lee County if address contains AL or Auburn/Opelika area
      if (/\b(AL|Ala|Auburn|Opelika|Lee)\b/i.test(address)) {
        try {
          const lee = await fetchLeeCountyByAddress(address);
          if ((lee.ownerNames && lee.ownerNames.length) || lee.parcelId) {
            return { provider: 'lee_county', data: lee };
          }
        } catch (e) {
          console.warn('Lee county lookup failed', e.message);
        }
      }

      // 5) Last resort: return failure and explain what to do
      return { provider: 'none', error: 'No owner/parcel found from ATTOM, Regrid, or county fallbacks. For production reliability get a Regrid account or contact county for direct API access.' };
    });

    return res.json({ success: true, q: address, result });
  } catch (err) {
    console.error('property endpoint error', err);
    res.status(500).json({ error: err.message });
  }
});

/* ----------------------------
   Enrichment endpoint placeholder (Whitepages)
   ---------------------------- */
app.post('/api/enrich', async (req, res) => {
  try {
    const { name, address } = req.body;
    if (!name && !address) return res.status(400).json({ error: 'name or address required' });
    // simple cache key
    const cacheKey = `enrich:${name||''}:${address||''}`;
    const data = await cached(cacheKey, 60 * 60 * 24, async () => {
      if (!WHITEPAGES_API_KEY) return { note: 'WHITEPAGES_API_KEY not set; enrichment disabled' };
      const url = `https://proapi.whitepages.com/3.0/identity_check.json?api_key=${WHITEPAGES_API_KEY}&name=${encodeURIComponent(name||'')}&address=${encodeURIComponent(address||'')}`;
      const r = await fetch(url);
      if (!r.ok) {
        const t = await r.text().catch(()=>String(r.status));
        throw new Error(`Whitepages error ${r.status}: ${t}`);
      }
      return r.json();
    });
    return res.json({ success: true, data });
  } catch (err) {
    console.error('enrich error', err);
    res.status(500).json({ error: err.message });
  }
});

6) Documentation & next production steps (I already recommend)

Prefer Regrid or ATTOM for production — scraping county pages is fragile. Regrid is great for nationwide parcels, ATTOM is great for a full property dataset.

If you need bulk downloads for a county, call the county assessor office (Muscogee: publicaccess.columbusga.gov; Lee: alabamagis.com/lee or leepa.org) and request data licensing or FTP/CSV exports. Many counties will provide bulk records for a fee or under terms.

Logging & audit trail — store who requested what and why for TCPA/CAN-SPAM & privacy audits.

Rate limiting — add a per-IP or per-user throttle if your contractors will make many lookups quickly.
