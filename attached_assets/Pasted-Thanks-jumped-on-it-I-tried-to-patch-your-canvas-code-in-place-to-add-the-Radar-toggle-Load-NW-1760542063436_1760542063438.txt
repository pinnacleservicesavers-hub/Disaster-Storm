// --- Radar + NWS controls ---
let radarLayer = null; let nwsLayer = null;

function toggleRadar(cb){
  try{
    if(!radarLayer){
      radarLayer = L.tileLayer(
        'https://mesonet.agron.iastate.edu/cache/tile.py/1.0.0/nexrad-n0q-900913/{z}/{x}/{y}.png',
        {opacity:0.6}
      );
    }
    if(cb && cb.checked){ radarLayer.addTo(map); }
    else { if(radarLayer){ map.removeLayer(radarLayer); } }
  }catch(e){ console.log('radar toggle failed', e); }
}
function ensureRadar(){
  const cb = document.getElementById('radarToggle');
  if(cb){ toggleRadar(cb); }
}
async function loadNWS(){
  try{
    const gj = await (await fetch('/nws.geojson')).json();
    if(!nwsLayer){
      nwsLayer = L.geoJSON(null, {
        style:{color:'#ff0000', weight:2},
        onEachFeature:(f,l)=>l.bindPopup(f.properties && (f.properties.event||'NWS Alert'))
      });
    }
    nwsLayer.clearLayers(); nwsLayer.addData(gj); nwsLayer.addTo(map);
  }catch(e){ console.log('NWS load failed', e); }
}
setTimeout(ensureRadar, 600);

async function addSchedule(){
  const name = document.getElementById('sch_name').value || 'Daily Job';
  let lat = document.getElementById('sch_lat').value;
  let lon = document.getElementById('sch_lon').value;
  const addr = document.getElementById('sch_addr').value;
  const zone = document.getElementById('sch_zone') ? document.getElementById('sch_zone').value : '';
  if((!lat || !lon) && addr){
    const g = await (await fetch('/geocode?q='+encodeURIComponent(addr))).json();
    if(g && g.lat && g.lon){ lat = g.lat; lon = g.lon; }
  }
  const rad = document.getElementById('sch_rad').value || 25;
  const hrs = document.getElementById('sch_hrs').value || 72;
  const th  = document.getElementById('sch_thresh').value || 60;
  const wh  = document.getElementById('sch_wh')?.value || '';
  const sms = document.getElementById('sch_sms')?.value || '';
  const parcelsSel = document.getElementById('sch_parcels');
  const parcel_ids = Array.from(parcelsSel?.selectedOptions || []).map(o=>parseInt(o.value)).filter(Boolean);

  const r = await fetch(`/schedules`, {
    method:'POST',
    headers:{'Content-Type':'application/json'},
    body: JSON.stringify({
      name,
      lat: lat?parseFloat(lat):null,
      lon: lon?parseFloat(lon):null,
      radius_km: rad?parseFloat(rad):null,
      hours_back: parseInt(hrs),
      threshold: parseFloat(th),
      zone_id: zone?parseInt(zone):null,
      webhook_url: wh||null,
      sms_csv: sms||null,
      parcel_ids
    })
  });
  const j = await r.json();
  alert('Added schedule #'+j.id);
  renderSchedules();
}

async function renderSchedules(){
  const r = await fetch('/schedules');
  const j = await r.json();
  const el = document.getElementById('schedules');
  el.innerHTML = '';
  const ul = document.createElement('ul');
  (j||[]).forEach(it=>{
    const li = document.createElement('li');
    const z = it.zone_id ? `zone#${it.zone_id}` : `lat:${it.lat}, lon:${it.lon}, r:${it.radius_km}km`;
    const ptxt = it.parcel_ids ? `, parcels:[${it.parcel_ids}]` : '';
    li.innerHTML = `#${it.id} ${it.name} — ${z}, last ${it.hours_back}h, thr ${it.threshold}${ptxt} `+
      (it.webhook_url?`(wh override) `:'') + (it.sms_csv?`(sms override) `:'') +
      `[ <a href=\"#\" onclick=\"testSchedule(${it.id},0,0);return false;\">Test</a> | `+
      `<a href=\"#\" onclick=\"testSchedule(${it.id},1,0);return false;\">Test+Notify</a> | `+
      `<a href=\"#\" onclick=\"testSchedule(${it.id},0,1);return false;\">Test+CSV</a> | `+
      `<a href=\"#\" onclick=\"testSchedule(${it.id},1,1);return false;\">Test+Notify+CSV</a> | `+
      `<a href=\"#\" onclick=\"cloneSchedule(${it.id});return false;\">clone</a> | `+
      `<a href=\"#\" onclick=\"deleteSchedule(${it.id});return false;\">delete</a> ]`;
    ul.appendChild(li);
  });
  el.appendChild(ul);
}


(Leave your existing async function deleteSchedule(id){...} as-is; it will now call the global renderSchedules().)
