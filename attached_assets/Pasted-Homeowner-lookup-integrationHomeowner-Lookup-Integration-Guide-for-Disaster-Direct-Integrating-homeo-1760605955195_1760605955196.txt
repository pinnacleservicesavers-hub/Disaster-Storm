Public records APIs are ideal for US addresses, but phone data often requires paid, compliant providers like Whitepages or Cole Information. Free tiers exist for geocoding, but scale to paid for production.

Building on your previous drone agent, here's an updated Python script. It adds:

reverse_geocode: Converts GPS to address using BigDataCloud (free tier).
homeowner_lookup: Fetches name/phone using Whitepages API (replace with your key).

This runs autonomously: After fetching drone data, it looks up the homeowner and notifies contractors with full details.
pythonfrom langchain.agents import initialize_agent, AgentType
from langchain.tools import tool
from langchain_openai import ChatOpenAI  # Use xAI-compatible if available
from langchain.memory import ConversationBufferMemory
import os
from dotenv import load_dotenv
from twilio.rest import Client
import requests
import json

# Load environment variables
load_dotenv()
xai_api_key = os.getenv("XAI_API_KEY")  # For LLM
twilio_sid = os.getenv("TWILIO_SID")
twilio_auth_token = os.getenv("TWILIO_AUTH_TOKEN")
twilio_phone = os.getenv("TWILIO_PHONE_NUMBER")
bigdatacloud_key = os.getenv("BIGDATACLOUD_KEY")  # Free signup
whitepages_key = os.getenv("WHITEPAGES_KEY")  # Paid, for phone lookup

# Initialize LLM (Grok via xAI/OpenAI compat)
llm = ChatOpenAI(api_key=xai_api_key, model="grok", temperature=0.7)

# Memory for context
memory = ConversationBufferMemory(memory_key="chat_history")

# Existing Tools (from previous code)
@tool
def fetch_drone_data():
    """Fetch latest drone photos/videos with GPS data from external API."""
    # Mock or real API call
    headers = {"Authorization": f"Bearer {xai_api_key}"}
    response = requests.get(os.getenv("DRONE_API_URL"), headers=headers)
    if response.status_code == 200:
        data = response.json()
        return [{"id": item["id"], "gps": item["gps"], "url": item["url"], "timestamp": item["timestamp"]} for item in data["data"]]
    return "Error fetching drone data"

@tool
def prioritize_damage(photo_data):
    """Prioritize damage based on photo/video content."""
    description = llm.invoke(f"Analyze damage from drone data: {photo_data}. Prioritize as high (tree on home), medium (tree on car), or low (minor debris). Return JSON.")
    return description

@tool
def notify_contractors(priority, address, homeowner_phone, contractor_phone):
    """Send SMS notification to contractors with priority, address, and homeowner contact."""
    client = Client(twilio_sid, twilio_auth_token)
    message = f"Urgent: {priority} damage at {address}. Homeowner: {homeowner_phone}. Deploy via Starlink."
    client.messages.create(body=message, from_=twilio_phone, to=contractor_phone)
    return f"Notification sent to {contractor_phone}"

@tool
def generate_report(photo_data, priority, address, homeowner_info):
    """Generate a damage report with photo URL, priority, address, and homeowner details."""
    report = llm.invoke(f"Create a report: Drone data {photo_data}, priority {priority}, address {address}, homeowner {homeowner_info}. Format as markdown for insurance.")
    return report

# New Tool 1: Reverse Geocoding (GPS → Address)
@tool
def reverse_geocode(gps):
    """Convert GPS coordinates to address using BigDataCloud API."""
    lat, lng = gps.split(",")  # Assume format "lat,lng"
    url = f"https://api.bigdatacloud.net/data/reverse-geocode-client?latitude={lat}&longitude={lng}&localityLanguage=en"
    response = requests.get(url)
    if response.status_code == 200:
        data = response.json()
        address = f"{data.get('streetName', '')} {data.get('streetNumber', '')}, {data.get('city', '')}, {data.get('countryName', '')}"
        return address if address.strip() else "Address not found"
    return "Geocoding error"

# New Tool 2: Homeowner Lookup (Address → Name/Phone)
@tool
def homeowner_lookup(address):
    """Lookup homeowner name and phone from address using Whitepages API (privacy-compliant)."""
    # Whitepages endpoint example (adapt to actual API docs)
    url = "https://proapi.whitepages.com/lookup/address"
    params = {"address": address, "api_key": whitepages_key}
    response = requests.get(url, params=params)
    if response.status_code == 200:
        data = response.json()
        # Assume response structure: e.g., {"name": "John Doe", "phone": "+1-555-1234"}
        return {"name": data.get("name", "Unknown"), "phone": data.get("phone", "N/A")}
    return {"name": "Lookup failed", "phone": "N/A"}

# Initialize Agent
tools = [fetch_drone_data, prioritize_damage, reverse_geocode, homeowner_lookup, notify_contractors, generate_report]
agent = initialize_agent(tools, llm, agent_type=AgentType.ZERO_SHOT_REACT_DESCRIPTION, memory=memory)

# Example Execution
def run_drone_agent_with_lookup():
    # Step 1: Fetch data
    drone_data = fetch_drone_data()
    if isinstance(drone_data, str):  # Error handling
        return drone_data

    # Step 2: Process each item
    for item in drone_data:
        priority = prioritize_damage([item])
        address = reverse_geocode(item["gps"])
        homeowner_info = homeowner_lookup(address)
        contractor_phone = "+1987654321"  # Dynamic from app DB
        notify_contractors(priority, address, homeowner_info["phone"], contractor_phone)
        report = generate_report([item], priority, address, homeowner_info)
        print(f"Report for {address}: {report}")

if __name__ == "__main__":
    run_drone_agent_with_lookup()
