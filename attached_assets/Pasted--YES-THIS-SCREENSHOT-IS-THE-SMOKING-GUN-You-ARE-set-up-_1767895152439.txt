import { sql } from "drizzle-orm";

export function registerDbTest(app: any) {
  app.get("/api/db-test", async (req: any, res: any) => {
    try {
      const result = await db.execute(sql`SELECT NOW() as now`);
      res.json({ connected: true, now: result.rows?.[0]?.now || null });
    } catch (e) {
      res.status(500).json({ connected: false, error: String(e) });
    }
  });
}
