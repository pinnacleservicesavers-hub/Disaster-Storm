async function listRuns(){
  const r = await fetch('/runs');
  const j = await r.json();
  const el = document.getElementById('runs');
  el.innerHTML = '';
  const ul = document.createElement('ul');
  (j||[]).forEach(it=>{
    const li = document.createElement('li');
    li.innerHTML = `<strong>#${it.id}</strong> ${new Date(it.created_at*1000).toLocaleString()} — ${it.name} — risk ${it.risk_score} `+
      `[ <a target="_blank" href="/runs/${it.id}">JSON</a> | <a target="_blank" href="/runs/${it.id}/events.csv">events.csv</a> | <a target="_blank" href="/runs/${it.id}/impacted.csv">impacted.csv</a> ]`;
    ul.appendChild(li);
  });
  el.appendChild(ul);
}


async function doGeocode(){
  const q = document.getElementById('addr').value;
  if(!q) return;
  const r = await fetch('/geocode?q='+encodeURIComponent(q));
  const j = await r.json();
  if(j && j.lat && j.lon){
    document.getElementById('lat').value = j.lat;
    document.getElementById('lon').value = j.lon;
    map.setView([parseFloat(j.lat), parseFloat(j.lon)], 10);
  } else {
    alert('No result');
  }
}
function openParcelReport(id){
  const lat = document.getElementById('lat').value;
  const lon = document.getElementById('lon').value;
  const rad = document.getElementById('rad').value;
  const hrs = document.getElementById('hrs').value;
  window.open(`/report/parcel/${id}?lat=${lat}&lon=${lon}&radius_km=${rad}&hours_back=${hrs}`,'_blank');
}


function exportCSV(type){
  const lat = document.getElementById('lat').value;
  const lon = document.getElementById('lon').value;
  const rad = document.getElementById('rad').value;
  const hrs = document.getElementById('hrs').value;
  const url = type==='events'
    ? `/export/events.csv?lat=${lat}&lon=${lon}&radius_km=${rad}&hours_back=${hrs}`
    : `/export/impacted.csv?lat=${lat}&lon=${lon}&radius_km=${rad}&hours_back=${hrs}`;
  window.open(url, '_blank');
}
