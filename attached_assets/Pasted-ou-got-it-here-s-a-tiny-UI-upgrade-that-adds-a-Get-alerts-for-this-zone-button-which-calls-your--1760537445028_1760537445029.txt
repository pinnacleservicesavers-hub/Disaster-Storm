2) Add these helper functions inside your <script> tag

Place them near your other helpers (e.g., after buildIdText):

function extractZoneIdFromUrl(url) {
  if (!url) return null;
  const parts = String(url).split("/");
  return parts[parts.length - 1] || null;
}

// Pick the best zone id to use for alerts: prefer forecast zone, fallback to county zone
function getPreferredZoneId(payload) {
  const z = payload?.zones || {};
  const fz = extractZoneIdFromUrl(z.forecastZone);
  const cz = extractZoneIdFromUrl(z.county);
  return fz || cz || null;
}

function renderZoneAlertsList(items = []) {
  const wrap = document.getElementById("zone-alerts");
  if (!items.length) {
    wrap.innerHTML = `<div class="muted">No active alerts for this zone.</div>`;
    return;
  }
  wrap.innerHTML = items.map(it => `
    <div class="card">
      <div class="headline">${it.headline || it.event || "Alert"}</div>
      <div class="small muted">
        ${it.severity || ""}${it.severity ? " • " : ""}${it.urgency || ""}
        ${it.expires ? ` • Expires: ${new Date(it.expires).toLocaleString()}` : ""}
      </div>
      <div class="small">${(it.areas || "").slice(0, 200)}</div>
    </div>
  `).join("");
}

async function fetchZoneAlertsForCurrent() {
  const zoneWrap = document.getElementById("zone-alerts");
  if (!lastPayload) {
    zoneWrap.innerHTML = `<div class="card">No location loaded yet. Pick a city or use GPS first.</div>`;
    return;
  }
  const zoneId = getPreferredZoneId(lastPayload);
  if (!zoneId) {
    zoneWrap.innerHTML = `<div class="card">Couldn’t determine a zone ID for this location.</div>`;
    return;
  }
  zoneWrap.innerHTML = `<div class="card skeleton" style="height:80px"></div>`;
  try {
    const r = await fetch(`/api/alerts?zone=${encodeURIComponent(zoneId)}`);
    if (!r.ok) throw new Error(`HTTP ${r.status}`);
    const data = await r.json();
    renderZoneAlertsList(data.items || []);
    showToast(`Loaded alerts for ${zoneId}`);
  } catch (e) {
    zoneWrap.innerHTML = `
      <div class="error">
        <div class="card">
          <div class="headline">Zone alerts failed</div>
          <div class="small">${e.message}</div>
          <div class="error-actions">
            <button class="retry-btn" data-target="${lastMode}">Retry</button>
          </div>
        </div>
      </div>`;
  }
}

document.addEventListener("click", async (e) => {
  const btn = e.target.closest(".retry-btn");
  if (btn) {
    const target = btn.dataset.target || lastMode;
    if (target === "point" && lastPoint) { fetchPoint(lastPoint.lat, lastPoint.lon); }
    else { fetchCity(lastCity); }
    return;
  }

  if (e.target.id === "copyIdsBtn") {
    const text = buildIdText(lastPayload);
    if (!text) { showToast("No IDs to copy yet"); return; }
    try { await navigator.clipboard.writeText(text); showToast("Copied zone & office IDs"); }
    catch { window.prompt("Copy IDs:", text); }
    return;
  }

  // NEW: Zone alerts button

    fetchZoneAlertsForCurrent();
    return;
  }
});
