1) Show NWS alerts as polygons (Leaflet overlay)

src/components/AlertsOverlay.jsx

import { useEffect, useRef } from "react";
import L from "leaflet";

function styleFor(event, severity="Moderate"){
  const color = /Tornado/.test(event) ? "#d0021b"
      : /Hurricane|Tropical/.test(event) ? "#f5a623"
      : /Severe Thunderstorm/.test(event) ? "#f8e71c"
      : /Flash Flood|Storm Surge/.test(event) ? "#50e3c2"
      : "#4a90e2";
  const weight = (/Tornado/.test(event) ? 3 : 2);
  const dashArray = (/Watch/.test(event) ? "6,4" : null);
  const fillOpacity = (/Warning/.test(event) ? 0.20 : 0.10);
  return { color, weight, dashArray, fillOpacity, fillColor: color };
}

/**
 * alerts: [{ id, event, severity, polygon (GeoJSON geometry) }, ...]
 */
export default function AlertsOverlay({ map, alerts=[] }) {
  const layerRef = useRef(null);

  useEffect(() => {
    if (!map) return;
    if (layerRef.current) {
      map.removeLayer(layerRef.current);
      layerRef.current = null;
    }
    const fc = {
      type: "FeatureCollection",
      features: alerts.filter(a => a.polygon).map(a => ({
        type: "Feature",
        geometry: a.polygon,
        properties: { id: a.id, event: a.event, severity: a.severity }
      }))
    };
    const layer = L.geoJSON(fc, {
      style: f => styleFor(f.properties.event, f.properties.severity),
      onEachFeature: (f, lyr) => {
        const p = f.properties;
        lyr.bindPopup(`<b>${p.event}</b><br/>Severity: ${p.severity || "n/a"}`);
      }
    });
    layer.addTo(map);
    layerRef.current = layer;
    return () => { if (layerRef.current) map.removeLayer(layerRef.current); };
  }, [map, alerts]);

  return null;
}

function insideGeo(lead, geometry){
  const pt = turf.point([lead.lon, lead.lat]);
  const feat = turf.feature(geometry);
  return turf.booleanPointInPolygon(pt, feat);
}

async function maybePromoteLead(lead, alert){
  if (!alert?.polygon) return;
  if (!insideGeo(lead, alert.polygon)) return;

  // Promote and notify
  await fetch("/events", { method: "POST", body: JSON.stringify({
    type: "lead_promoted",
    leadId: lead.id,
    newPriority: "very_urgent",
    reason: `Inside ${alert.event}`,
    alertId: alert.id
  })});
}

3) Batch route planning (multi-stop) with OSRM /trip

src/lib/routeTrip.js

let OSRM_BASE = "https://router.project-osrm.org";
(async () => {
  try {
    const j = await fetch("/config").then(r => r.json());
    if (j?.osrm_url) OSRM_BASE = j.osrm_url.replace(/\/$/, "");
  } catch {}
})();

/**
 * Plan best order for N leads from crew position.
 * @param crew {lon,lat}
 * @param leads [{id,lon,lat,priority}], limit 12 recommended
 * @returns { order:[leadIds], seconds:number, distance:number }
 */
export async function planTrip({ crew, leads }) {
  if (!leads?.length) return null;

  // Take top by priority, then nearest (quick heuristic)
  const sorted = leads.slice().sort((a,b) => (b.priority||0)-(a.priority||0)).slice(0, 12);

  const coords = [
    `${crew.lon},${crew.lat}`,
    ...sorted.map(p => `${p.lon},${p.lat}`)
  ].join(";");

  const url = `${OSRM_BASE}/trip/v1/driving/${coords}?source=first&roundtrip=false&overview=false`;
  const r = await fetch(url);
  if (!r.ok) return null;
  const j = await r.json();
  const trip = j?.trips?.[0];
  if (!trip) return null;

  // waypoints[0] is the crew start; others map to input points by "waypoint_index"
  const order = j.waypoints
    .slice(1)
    .sort((a,b) => a.waypoint_index - b.waypoint_index)
    .map(wp => {
      const idx = wp.waypoint_index - 1; // offset because we sliced(1)
      return sorted[idx]?.id;
    });

  return { order, seconds: Math.round(trip.duration), distance: Math.round(trip.distance) };
}
