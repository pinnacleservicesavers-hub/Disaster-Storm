✅ lib/vaultApi.js (calls Node + Python)
export async function fetchLeadsFromEngines({ location, radius, tradeType }) {
  const headers = {
    "Content-Type": "application/json",
    "X-VAULT-KEY": process.env.VAULT_SERVICE_KEY,
  };

  const nodeRes = await fetch(`${process.env.NODE_LEAD_ENGINE_URL}/leads/search`, {
    method: "POST",
    headers,
    body: JSON.stringify({ location, radius, tradeType }),
  });

  const pyRes = await fetch(`${process.env.PY_SIGNAL_ENGINE_URL}/signals/enrich`, {
    method: "POST",
    headers,
    body: JSON.stringify({ location, radius, tradeType }),
  });

  const nodeData = await nodeRes.json();
  const pyData = await pyRes.json();

  // Merge results (node provides base leads, python adds signal enrichment)
  const merged = nodeData.leads.map((lead) => {
    const signals = pyData.signalsByLeadId?.[lead.external_id] || [];
    return { ...lead, signal_tags: [...(lead.signal_tags || []), ...signals] };
  });

  return merged;
}

import { fetchLeadsFromEngines } from "@/lib/vaultApi";
import { contractorOnly } from "@/lib/contractorOnly";
import { getUser } from "@/lib/auth";

export async function POST(req) {
  const user = await getUser();
  try {
    contractorOnly(user);
  } catch {
    return Response.json({ error: "Contractor-only module" }, { status: 403 });
  }

  const body = await req.json();
  const { location, radius } = body;

  const leads = await fetchLeadsFromEngines({
    location,
    radius,
    tradeType: user.trade_type,
  });

  return Response.json({ leads });
}

import express from "express";
import { authKey } from "../middleware/authKey.js";
import { leadHunter } from "../services/leadHunter.js";
import { leadScorer } from "../services/leadScorer.js";
import { competitionDetector } from "../services/competitionDetector.js";

const router = express.Router();

router.post("/search", authKey, async (req, res) => {
  const { location, radius, tradeType } = req.body;

  const rawLeads = await leadHunter({ location, radius, tradeType });

  const leads = await Promise.all(rawLeads.map(async (lead) => {
    const scoreData = leadScorer(lead, tradeType);
    const comp = await competitionDetector(lead, location, radius);

    return {
      ...lead,
      ...scoreData,
      ...comp,
    };
  }));

  res.json({ leads });
});

export default router;

✅ services/leadScorer.js
export function leadScorer(lead, tradeType) {
  let score = 0;
  const tags = [];

  if (!lead.website) { score += 15; tags.push("no_website"); }
  if (lead.rating && lead.rating < 4.2) { score += 12; tags.push("low_rating"); }
  if (lead.reviewCount && lead.reviewCount > 50) { score += 8; tags.push("high_reviews"); }

  // Trade-specific rules
  if (tradeType === "pressure_washing" && /restaurant|gas|retail/i.test(lead.category)) {
    score += 10;
    tags.push("trade_match");
  }
  if (tradeType === "tree_service" && /hoa|property management|apartment/i.test(lead.category)) {
    score += 12;
    tags.push("property_match");
  }

  let score_label = "cold";
  if (score >= 38) score_label = "hot";
  else if (score >= 22) score_label = "warm";

  return { score, score_label, signal_tags: tags };
}

from middleware.auth_key import auth_key
from services.permit_signals import get_permit_signals
from services.violation_signals import get_violation_signals
from services.predictive_signals import predictive_signals

router = APIRouter()

@router.post("/enrich")
def enrich(payload: dict, authorized=Depends(auth_key)):
    location = payload["location"]
    radius = payload["radius"]
    trade_type = payload["tradeType"]

    permits = get_permit_signals(location, radius, trade_type)
    violations = get_violation_signals(location, radius, trade_type)
    prediction = predictive_signals(location, trade_type)

    return {
        "signalsByLeadId": {},  # later: map to external lead IDs
        "areaSignals": {
            "permits": permits,
            "violations": violations,
            "prediction": prediction
        }
    }
