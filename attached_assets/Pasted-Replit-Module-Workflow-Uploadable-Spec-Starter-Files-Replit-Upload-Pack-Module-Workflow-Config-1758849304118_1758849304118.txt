5) /src/eventBus.ts — tiny pub/sub for inter‑module comms
import { AppEvent } from "./types";


type Handler = (evt: AppEvent) => void;


class EventBus {
  private map = new Map<string, Set<Handler>>();


  on(type: string, handler: Handler) {
    if (!this.map.has(type)) this.map.set(type, new Set());
    this.map.get(type)!.add(handler);
    return () => this.off(type, handler);
  }


  off(type: string, handler: Handler) {
    this.map.get(type)?.delete(handler);
  }


  emit(evt: AppEvent) {
    const set = this.map.get(evt.type);
    set?.forEach((h) => h(evt));
  }
}


export const bus = new EventBus();


// Example helpers
export const publish = (type: string, payload?: unknown, source?: string) =>
  bus.emit({ type, payload, meta: { source, ts: Date.now() } });
6) /src/useModules.ts — load/save order locally
import { useEffect, useState } from "react";
import type { ModuleDef } from "./types";


const KEY = "module-order-v1";


export function useModules() {
  const [modules, setModules] = useState<ModuleDef[]>([]);


  useEffect(() => {
    async function load() {
      const res = await fetch("/src/modules.json");
      const defs: ModuleDef[] = await res.json();


      const saved = localStorage.getItem(KEY);
      if (saved) {
        const order: string[] = JSON.parse(saved);
        // sort by saved order (fallback to def.order)
        defs.sort((a, b) => {
          const ia = order.indexOf(a.id);
          const ib = order.indexOf(b.id);
          if (ia === -1 && ib === -1) return a.order - b.order;
          if (ia === -1) return 1;
          if (ib === -1) return -1;
          return ia - ib;
        });
      } else {
        defs.sort((a, b) => a.order - b.order);
      }


      setModules(defs);
    }
    load();
  }, []);


  function saveOrder(next: ModuleDef[]) {
    setModules(next);
    localStorage.setItem(KEY, JSON.stringify(next.map((m) => m.id)));
  }


  function reset(defs: ModuleDef[]) {
    localStorage.removeItem(KEY);
    setModules([...defs].sort((a, b) => a.order - b.order));
  }


  return { modules, setModules: saveOrder };
}

function move(mods: ModuleDef[], from: number, to: number) {
  const next = [...mods];
  const [item] = next.splice(from, 1);
  next.splice(to, 0, item);
  return next;
}


export default function Dashboard() {

14) Lead Prioritization & Auto‑Routing (Replit Implementation Spec)

This section adds lead scoring & routing so crews get the most urgent jobs first while minimizing drive time.

A. Priority Levels & Definitions

Use a clear, safety‑first severity scale. Keep labels short on mobile:

P0 – Life/Safety Critical (Very Urgent)

Examples: Tree on home with people possibly trapped, tree blocking primary egress, live utility hazards impacting occupants, car on house, structure collapse risk.

P1 – Property Critical (Urgent)

Examples: Tree on building (non‑residential) or any other structure (garage, fence onto pool equipment, shed on power lines), tree causing active water intrusion but no entrapment.

P2 – Significant (High)

Examples: Large tree on vehicle not affecting egress, tree leaning with high likelihood of failure near home, partial roof damage with temp protection.

P3 – Standard (Normal)

Examples: Yard debris, fence sections down away from utilities, cosmetic damage.

Tie‑breakers: If two leads share the same P‑level, favor: (1) closer distance to assigned crew, then (2) earlier submission time.

B. Lead Schema (add to your backend or JSON store)

export interface Lead {

  createdAt: number; // Date.now()
  lat: number;
  lng: number;

C. Priority Rules (deterministic fallback + AI assist)

Rules first (fast, reliable):

If tags contains treeOnHome or blockedExit or carOnHouse ⇒ P0.

Else if tags contains treeOnBuilding or treeOnStructure ⇒ P1.

Else if tags contains treeOnCar or imminentLeanTowardHome ⇒ P2.

Else ⇒ P3.

AI overlay (optional): Use text+image model to confirm/upgrade/downgrade one level based on media (e.g., detect entrapment cues, active water ingress, live wires). Guardrail: AI cannot downgrade from P0 without human confirmation.

D. Scoring Function (for sort order)

Turn the P‑levels into numeric weights and add distance/time & recency:

const PRIORITY_WEIGHT: Record<Priority, number> = { P0: 1000, P1: 700, P2: 400, P3: 100 };


// distance in km -> smaller is better; clamp to avoid huge swings
const clamp = (v: number, min: number, max: number) => Math.max(min, Math.min(max, v));


export function leadScore(p: Priority, kmFromCrew: number, minutesOld: number) {
  const wP = PRIORITY_WEIGHT[p];
  const wD = 200 - clamp(kmFromCrew, 0, 200); // closer adds up to +200
  const wT = Math.min(minutesOld, 180); // up to +180 for aging in queue
  return wP + wD + wT; // higher = sooner
}
E. Distance Calculation (Haversine)
export function km(a: {lat: number; lng: number}, b: {lat: number; lng: number}) {
  const R = 6371; // km
  const dLat = (b.lat - a.lat) * Math.PI / 180;
  const dLng = (b.lng - a.lng) * Math.PI / 180;
  const la1 = a.lat * Math.PI / 180;
  const la2 = b.lat * Math.PI / 180;
  const s = Math.sin(dLat/2)**2 + Math.cos(la1)*Math.cos(la2)*Math.sin(dLng/2)**2;
  return 2 * R * Math.asin(Math.sqrt(s));
}
F. Real‑Time Re‑prioritization Flow

Events In: VICTIM.INTAKE.NEW, LENS.MEDIA.NEW, manual dispatcher edits, contractor check‑ins with GPS (CREW.CHECKIN).

Process:

Compute/confirm priority via Rules → AI overlay.

Compute score = leadScore(priority, distance, age) per contractor region.

Update queue order and emit JOB.QUEUE.UPDATE with the new sorted list.

Events Out: JOB.DISPATCH when a lead is assigned to a crew.

import { publish } from "./eventBus";


export function updateLeadQueue(leads: Lead[], contractorLoc: {lat:number; lng:number}) {
  const now = Date.now();
  const scored = leads.map(l => {
    const p: Priority = l.priority ?? inferPriorityFromTags(l.tags ?? []);
    const distanceKm = km(contractorLoc, l);
    const minutesOld = (now - l.createdAt) / 60000;
    return { ...l, priority: p, score: leadScore(p, distanceKm, minutesOld) };
  });
  const sorted = scored.sort((a, b) => b.score! - a.score!);
  publish("JOB.QUEUE.UPDATE", { items: sorted });
  return sorted;
}


function inferPriorityFromTags(tags: string[]): Priority {
  const set = new Set(tags);
  if (set.has("treeOnHome") || set.has("blockedExit") || set.has("carOnHouse")) return "P0";
  if (set.has("treeOnBuilding") || set.has("treeOnStructure")) return "P1";
  if (set.has("treeOnCar") || set.has("imminentLeanTowardHome")) return "P2";
  return "P3";
}
G. Route Builder (minimize zig‑zag)

Start with a Nearest‑Neighbor pass from crew location, then a quick 2‑opt improvement. Good enough for city‑scale routing without heavy libs.

export function buildRoute(leads: Lead[], start: {lat:number; lng:number}) {
  const pool = [...leads];
  const route: Lead[] = [];
  let curr = start;
  while (pool.length) {
    pool.sort((a,b) => km(curr, a) - km(curr, b));
    const next = pool.shift()!;
    route.push(next);
    curr = next;
  }
  return twoOpt(route, start);
}


function twoOpt(route: Lead[], start: {lat:number; lng:number}) {
  // simple 2-opt pass
  let improved = true;
  while (improved) {
    improved = false;
    for (let i=1; i<route.length-2; i++) {
      for (let k=i+1; k<route.length-1; k++) {
        const a1 = i-1<0 ? start : route[i-1];
        const a2 = route[i];
        const b1 = route[k];
        const b2 = route[k+1];
        const dNow = km(a1, a2) + km(b1, b2);
        const dNew = km(a1, b1) + km(a2, b2);
        if (dNew + 0.01 < dNow) { // small epsilon to avoid churn
          const slice = route.slice(i, k+1).reverse();
          route.splice(i, k-i+1, ...slice);
          improved = true;
        }
      }
    }
  }
  return route;
}
