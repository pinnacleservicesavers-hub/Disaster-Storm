✅ 4) The Campaign Runner (Core Logic)

This is the heart of everything.

What the runner does:

Calls Node Lead Engine to find leads

Scores & filters

Saves leads to DB & prevents duplicates

Generates Outreach Pack for selected leads

Creates Follow-Up tasks

Logs everything into campaign_runs history

✅ Create this file:
app-next/lib/campaigns/runCampaign.js
import { fetchLeadsFromEngines } from "@/lib/vaultApi";
import { buildFollowUpTasks } from "@/lib/followups/createFollowUpTasks";

// TODO: connect real DB inserts here
export async function runCampaign({ user, campaign }) {
  const runLog = {
    campaign_id: campaign.id,
    user_id: user.id,
    status: "running",
    leads_found: 0,
    hot_leads: 0,
    warm_leads: 0
  };

  // 1) Get leads
  const leads = await fetchLeadsFromEngines({
    location: campaign.target_location,
    radius: campaign.radius,
    tradeType: "tree_service"
  });

  // 2) Filter by score threshold
  const filtered = leads
    .filter(l => (l.score || 0) >= campaign.min_score)
    .slice(0, campaign.max_leads_per_run);

  runLog.leads_found = filtered.length;
  runLog.hot_leads = filtered.filter(l => l.score_label === "hot").length;
  runLog.warm_leads = filtered.filter(l => l.score_label === "warm").length;

  // 3) Decide which leads get outreach
  let outreachLeads = filtered;
  if (campaign.generate_outreach_for === "hot") {
    outreachLeads = filtered.filter(l => l.score_label === "hot");
  }
  if (campaign.generate_outreach_for === "warm") {
    outreachLeads = filtered.filter(l => l.score_label === "warm");
  }

  // 4) Generate outreach + follow-ups (call your existing API route locally)
  const outreachResults = [];
  for (const lead of outreachLeads.slice(0, 10)) { // hard cap to control costs
    const outreachPack = await generateOutreachPack(user, lead, campaign);
    const followUpTasks = buildFollowUpTasks({
      user,
      leadId: lead.external_id,
      outreachMessageId: `outreach-${lead.external_id}`,
      outreachPack
    });

    outreachResults.push({ lead, outreachPack, followUpTasks });
  }

  // 5) Return what you need to store + display
  return {
    runLog,
    filteredLeads: filtered,
    outreachResults
  };
}

// Calls the Next.js outreach endpoint internally
async function generateOutreachPack(user, lead, campaign) {
  const res = await fetch(`${process.env.NEXT_PUBLIC_BASE_URL || ""}/api/leadvault/outreach`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      lead,
      context: {
        location: campaign.target_location,
        radius: campaign.radius,
        why: (lead.signal_tags || []).slice(0, 4).join(", ") || "Campaign lead"
      }
    })
  });

  const data = await res.json();
  return data.outreachPack;
}

  }
}

import cron from "node-cron";
import fetch from "node-fetch";

export function startCampaignScheduler() {
  // Run every day at 8:05 AM server time (adjust as needed)
  cron.schedule("5 8 * * *", async () => {
    console.log("[Scheduler] Running daily campaigns...");
    await runDueCampaigns();
  });

  // Run weekly on Monday at 8:10 AM
  cron.schedule("10 8 * * 1", async () => {
    console.log("[Scheduler] Running weekly campaigns...");
    await runDueCampaigns("weekly");
  });
}

async function runDueCampaigns(scheduleType = "daily") {
  try {
    // Call Next.js route that fetches enabled campaigns
    // You’ll create this endpoint (below)
    const res = await fetch(`${process.env.NEXT_APP_URL}/api/leadvault/campaigns/scheduled/run`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "X-VAULT-KEY": process.env.VAULT_SERVICE_KEY
      },
      body: JSON.stringify({ scheduleType })
    });

    const data = await res.json();
    console.log("[Scheduler] Completed:", data);
  } catch (e) {
    console.error("[Scheduler] Failed:", e);
  }
}
