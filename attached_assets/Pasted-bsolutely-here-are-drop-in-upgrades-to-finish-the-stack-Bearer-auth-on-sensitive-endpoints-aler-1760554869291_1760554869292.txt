2) Optional read-only token for dashboard GETs

If you want dashboards to call JSON APIs only when a token is present, add this small guard and apply it to GET endpoints (tiles remain protected via HMAC already).

Add backend/middleware/readonlyAuth.js

export function requireReadOnlyIfEnabled(req, res, next) {
  const token = process.env.READONLY_TOKEN; // optional
  if (!token) return next();               // no gating if unset
  const hdr = req.headers.authorization || "";
  const got = hdr.startsWith("Bearer ") ? hdr.slice(7) : "";
  if (got === token) return next();
  return res.status(401).json({ error: "Unauthorized (read-only)" });
}
