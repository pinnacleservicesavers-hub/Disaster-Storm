export function leadScorer(lead, tradeType) {
  let score = 0;
  const tags = [];

  if (!lead.website) { score += 15; tags.push("no_website"); }
  if (lead.rating && lead.rating < 4.0) { score += 10; tags.push("low_rating"); }
  if (lead.reviewCount && lead.reviewCount > 50) { score += 7; tags.push("high_volume"); }

  // Trade-specific boosts
  if (tradeType === "pressure_washing" && lead.category.includes("Restaurant")) score += 8;
  if (tradeType === "roofing" && lead.category.includes("Property Management")) score += 10;

  let label = "cold";
  if (score >= 35) label = "hot";
  else if (score >= 20) label = "warm";

  return { score, score_label: label, signal_tags: tags };
}

export async function fetchLeadsFromEngines({ location, radius, tradeType }) {
  const headers = {
    "Content-Type": "application/json",
    "X-VAULT-KEY": process.env.VAULT_SERVICE_KEY,
  };

  const nodeRes = await fetch(`${process.env.NODE_LEAD_ENGINE_URL}/leads/search`, {
    method: "POST",
    headers,
    body: JSON.stringify({ location, radius, tradeType }),
  });

  const pyRes = await fetch(`${process.env.PY_SIGNAL_ENGINE_URL}/signals/enrich`, {
    method: "POST",
    headers,
    body: JSON.stringify({ location, radius, tradeType }),
  });

  const nodeData = await nodeRes.json();
  const pyData = await pyRes.json();

  // Merge results (node provides base leads, python adds signal enrichment)
  const merged = nodeData.leads.map((lead) => {
    const signals = pyData.signalsByLeadId?.[lead.external_id] || [];
    return { ...lead, signal_tags: [...(lead.signal_tags || []), ...signals] };
  });

  return merged;
}

  let score = 0;
  const tags = [];

  if (!lead.website) { score += 15; tags.push("no_website"); }
  if (lead.rating && lead.rating < 4.2) { score += 12; tags.push("low_rating"); }

export async function fetchLeadsFromEngines({ location, radius, tradeType }) {
  const headers = {
    "Content-Type": "application/json",
    "X-VAULT-KEY": process.env.VAULT_SERVICE_KEY,
  };

  const nodeRes = await fetch(`${process.env.NODE_LEAD_ENGINE_URL}/leads/search`, {
    method: "POST",
    headers,
    body: JSON.stringify({ location, radius, tradeType }),
  });

  const pyRes = await fetch(`${process.env.PY_SIGNAL_ENGINE_URL}/signals/enrich`, {
    method: "POST",
    headers,
    body: JSON.stringify({ location, radius, tradeType }),
  });

  const nodeData = await nodeRes.json();
  const pyData = await pyRes.json();

  const merged = (nodeData.leads || []).map((lead) => {
    const areaSignals = pyData.areaSignals || {};
    const leadSignals = pyData.signalsByLeadId?.[lead.external_id] || [];
    return {
      ...lead,
      signal_tags: [...(lead.signal_tags || []), ...leadSignals],
      area_alerts: areaSignals.alerts || [],
      storm_risk: areaSignals.stormRisk || null,
    };
  });

  return merged;
}

✅ lead-engine-node/services/leadScorer.js (TREE SERVICE SCORING)
export function leadScorer(lead) {
  let score = 0;
  const tags = [];

  // Universal signals
  if (!lead.website) { score += 18; tags.push("no_website"); }
  if (lead.rating && lead.rating < 4.2) { score += 10; tags.push("low_rating_opportunity"); }
  if (lead.reviewCount && lead.reviewCount > 60) { score += 8; tags.push("high_traffic_location"); }

  // Tree Service trade blueprint (where the money is)
  const cat = (lead.category || "").toLowerCase();

  if (cat.includes("property")) { score += 18; tags.push("property_manager_contracts"); }
  if (cat.includes("apartment")) { score += 16; tags.push("apartment_complex_maintenance"); }
  if (cat.includes("hoa")) { score += 16; tags.push("hoa_vendor_contract"); }
  if (cat.includes("construction")) { score += 12; tags.push("land_clearing_opportunity"); }
  if (cat.includes("government") || cat.includes("municipal")) { score += 15; tags.push("city_vendor_work"); }

  // Quick win: no site + poor rating = needs vendor help + easy pitch
  if (!lead.website && lead.rating && lead.rating < 4.0) {
    score += 10;
    tags.push("quick_win");
  }

  let score_label = "cold";
  if (score >= 45) score_label = "hot";
  else if (score >= 25) score_label = "warm";

  return { score, score_label, signal_tags: tags };
}
