Got you—here are tiny, practical wrappers for tiles and legend that work in two modes:

Unsigned mode (default): use a plain template string.

HMAC mode (recommended): add a one-endpoint signer on your API; the client helper auto-signs each tile (with caching) and the legend.

1) Backend: optional signing endpoints (only if HMAC is enabled)

Add these two routes to your server (they just call the signTileUrl you already export):

// server.js
import { signTileUrl } from "disaster-direct-proxy";

// Signs a single tile (expects exact z/x/y)
app.get("/api/sign/tile", (req, res) => {
  const { z, x, y, pollen = "1", grid = "3", scheme = "viridis", ttl = "180", fmt = "png" } = req.query;
  if (!z || !x || !y) return res.status(400).json({ error: "z,x,y required" });
  const path = `/api/impact/tiles/${z}/${x}/${y}.${fmt}?pollen=${pollen}&grid=${grid}&scheme=${scheme}&ttl=${ttl}`;
  const signed = signTileUrl(path, { secret: process.env.DD_TILE_HMAC_SECRET, expiresInSec: 300 });
  res.json({ url: signed, exp: Math.floor(Date.now()/1000)+300 });
});

app.get("/api/sign/legend", (req, res) => {
  const { scheme = "traffic", width = "256", height = "48", bg = "solid", fmt = "png" } = req.query;
  const path = `/api/legend.${fmt}?scheme=${scheme}&width=${width}&height=${height}&bg=${bg}`;
  const signed = signTileUrl(path, { secret: process.env.DD_TILE_HMAC_SECRET, expiresInSec: 300 });
  res.json({ url: signed, exp: Math.floor(Date.now()/1000)+300 });
});

2) Frontend helper: tiles + legend (TypeScript)
// ddTiles.ts
type SignTileFn = (path: string) => Promise<{ url: string; exp?: number }>;
type LegendOpts = { scheme?: string; width?: number; height?: number; bg?: "solid"|"transparent"; fmt?: "png"|"svg" };

export function makeUnsignedTileTemplate(base = "/api/impact/tiles/{z}/{x}/{y}.png", qp: Record<string,string|number> = {}) {
  const qs = new URLSearchParams(qp as Record<string,string>).toString();
  return qs ? `${base}?${qs}` : base;
}

/** Minimal LRU-ish cache by path */
function memo<T>(max = 200) {
  const m = new Map<string, { v:T; exp?: number }>();
  return {
    async getOrSet(key: string, exp: number|undefined, make: () => Promise<T>): Promise<T> {
      const hit = m.get(key);
      const now = Math.floor(Date.now()/1000);
      if (hit && (!hit.exp || hit.exp - 10 > now)) return hit.v;
      const v = await make();
      if (m.size > max) m.delete(m.keys().next().value);
      m.set(key, { v, exp });
      return v;
    }
  };
}

/**
 * Mapbox GL JS: transformRequest that auto-signs /api/impact/tiles requests.
 * Pass baseUrl for your API (same-origin relative "" is fine).
 */
export function makeMapboxTransformRequest(baseUrl = "", signEndpoint = "/api/sign/tile") {
  const cache = memo<string>(500);

  return async function transformRequest(url: string, resourceType: string) {
    if (resourceType !== "Tile" && resourceType !== "Image") return { url };

    // Only intercept your tiles
    if (!url.includes("/api/impact/tiles/")) return { url };

    try {
      const u = new URL(url, window.location.origin);
      const m = u.pathname.match(/\/api\/impact\/tiles\/(\d+)\/(\d+)\/(\d+)\.(png|svg)/);
      if (!m) return { url }; // not our pattern

      const [ , z, x, y, fmt ] = m;
      const params = u.searchParams;
      const pollen = params.get("pollen") ?? "1";
      const grid   = params.get("grid")   ?? "3";
      const scheme = params.get("scheme") ?? "viridis";
      const ttl    = params.get("ttl")    ?? "180";

      const key = `${z}/${x}/${y}.${fmt}?p=${pollen}&g=${grid}&s=${scheme}&t=${ttl}`;

      const signedUrl = await cache.getOrSet(key, undefined, async () => {
        const sigURL = new URL(signEndpoint, baseUrl);
        sigURL.searchParams.set("z", z);
        sigURL.searchParams.set("x", x);
        sigURL.searchParams.set("y", y);
        sigURL.searchParams.set("fmt", fmt);
        sigURL.searchParams.set("pollen", pollen);
        sigURL.searchParams.set("grid", grid);
        sigURL.searchParams.set("scheme", scheme);
        sigURL.searchParams.set("ttl", ttl);
        const r = await fetch(sigURL.toString(), { credentials: "include" });
        if (!r.ok) throw new Error(`sign ${r.status}`);
        const j = await r.json();
        return j.url as string;
      });

      // Return the fully signed, same-origin relative URL (works with Mapbox)
      return { url: signedUrl };
    } catch {
      return { url }; // fall back silently
    }
  };
}

/** Legend URL helper */
export async function getLegendUrl(baseUrl = "", opts: LegendOpts = {}, hmac = false): Promise<string> {
  const { scheme = "traffic", width = 256, height = 48, bg = "solid", fmt = "png" } = opts;
  if (!hmac) {
    const u = new URL(`/api/legend.${fmt}`, baseUrl || window.location.origin);
    u.searchParams.set("scheme", scheme);
    u.searchParams.set("width", String(width));
    u.searchParams.set("height", String(height));
    u.searchParams.set("bg", bg);
    return u.pathname + "?" + u.searchParams.toString();
  }
  // HMAC: ask signer once
  const signURL = new URL("/api/sign/legend", baseUrl || window.location.origin);
  signURL.searchParams.set("scheme", scheme);
  signURL.searchParams.set("width", String(width));
  signURL.searchParams.set("height", String(height));
  signURL.searchParams.set("bg", bg);
  const r = await fetch(signURL.toString(), { credentials: "include" });
  if (!r.ok) throw new Error("legend sign failed");
  const { url } = await r.json();
  return url;
}

3) Map integrations
A) Mapbox GL JS
import mapboxgl from "mapbox-gl";
import { makeUnsignedTileTemplate, makeMapboxTransformRequest, getLegendUrl } from "./ddTiles";

// If HMAC OFF:
const tileTemplate = makeUnsignedTileTemplate("/api/impact/tiles/{z}/{x}/{y}.png", {
  pollen: 1, grid: 3, scheme: "viridis", ttl: 180
});

// If HMAC ON:
(mapboxgl as any).config.API_URL = ""; // keep same-origin
const transformRequest = makeMapboxTransformRequest("", "/api/sign/tile");

const map = new mapboxgl.Map({
  container: "map",
  style: { version: 8, sources: {}, layers: [] },
  transformRequest // only needed if HMAC is ON
});

map.on("load", async () => {
  map.addSource("impactTiles", {
    type: "raster",
    tiles: [tileTemplate],     // HMAC OFF: direct template
    tileSize: 256
  });

  // If HMAC ON, Mapbox will call transformRequest for each tile and we’ll swap in signed URLs.

  map.addLayer({ id: "impact", type: "raster", source: "impactTiles" });

  // Legend
  const legendUrl = await getLegendUrl("", { scheme: "viridis", width: 256, height: 48, bg: "solid" }, /*hmac*/ true);
  (document.getElementById("legend") as HTMLImageElement).src = legendUrl;
});

B) Leaflet
import L from "leaflet";
import { makeUnsignedTileTemplate, getLegendUrl } from "./ddTiles";

// Unsigned (HMAC OFF)
const template = makeUnsignedTileTemplate("/api/impact/tiles/{z}/{x}/{y}.png", {
  pollen: 1, grid: 3, scheme: "traffic", ttl: 180
});
const map = L.map("map").setView([33.749, -84.388], 10);
L.tileLayer(template, { tileSize: 256, crossOrigin: false }).addTo(map);

// HMAC ON: use a custom tile function that calls /api/sign/tile per xyz
const signedLayer = L.tileLayer("", {
  tileSize: 256,
  crossOrigin: false
});
signedLayer.getTileUrl = async function(coords) {
  const fmt = "png";
  const qs = new URLSearchParams({ pollen: "1", grid: "3", scheme: "viridis", ttl: "180" }).toString();
  const sign = await fetch(`/api/sign/tile?z=${coords.z}&x=${coords.x}&y=${coords.y}&fmt=${fmt}&${qs}`).then(r=>r.json());
  return sign.url; // Leaflet accepts a Promise<string> in modern builds; if not, prefetch in a wrapper.
};
signedLayer.addTo(map);

// Legend
getLegendUrl("", { scheme: "viridis", width: 256, height: 48, bg: "solid" }, true)
  .then(url => { (document.getElementById("legend") as HTMLImageElement).src = url; });


If your Leaflet version doesn’t support Promise in getTileUrl, wrap a small cache that pre-signs during tileloadstart or fall back to unsigned mode with a CDN in front.

Why this setup

No secret in the browser: client never signs—your server does.

Fast: per-tile signer is tiny and cacheable; helpers memoize signatures for current view.

Flexible: same code works whether HMAC is on or off. Toggle later without touching the map layer.
