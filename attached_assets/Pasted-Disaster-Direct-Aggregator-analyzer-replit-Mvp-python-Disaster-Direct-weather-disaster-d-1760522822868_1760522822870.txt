"""
Disaster Direct – weather/disaster data aggregator & analyzer (Replit MVP++)
-------------------------------------------------------------------------


Stack:
- Python 3.11+
- FastAPI (API + simple dashboard)
- httpx (async HTTP client)
- SQLite (caching + parcel store)
- shapely (parcel/event intersections)


Run locally / in Replit:
1) Create a new Python Repl.
2) Create `requirements.txt` with the packages listed at the bottom of this file.
3) Save this file as `app.py`.
4) Run:  uvicorn app:app --host 0.0.0.0 --port 8000 --reload


Environment (optional but recommended for premium data):
- TOMORROWIO_API_KEY=<key>  # adds hyperlocal weather provider


Notes:
- Aggregates free sources (NWS, USGS, FEMA) + optional Tomorrow.io.
- Normalizes into a common schema and computes a risk score.
- Adds a **parcel upload** store and **impact analysis** (street-level if parcels are detailed).
- Includes a simple built-in **Dashboard** at `/` for quick testing.
"""

def km_to_deg_latlon(km: float) -> float:
    # Approx conversion: 1 degree ~ 111km
    return km / 111.0




def now_utc() -> datetime:
    return datetime.now(timezone.utc)

def init_db() -> None:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS cache (
            key TEXT PRIMARY KEY,
            value TEXT NOT NULL,
            expires_at INTEGER NOT NULL
        )
        """
    )
    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS parcels (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            geojson TEXT NOT NULL,
            created_at INTEGER NOT NULL
        )
        """
    )
    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS runs (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            created_at INTEGER NOT NULL,
            lat REAL, lon REAL, radius_km REAL, hours_back INTEGER,
            risk_score REAL,
            events_json TEXT,
            impacted_json TEXT
        )
        """
    )
    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS schedules (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            lat REAL, lon REAL, radius_km REAL, hours_back INTEGER,
            threshold REAL,
            created_at INTEGER NOT NULL
        )
        """
    )
    cur.execute("CREATE TABLE IF NOT EXISTS alerts (id INTEGER PRIMARY KEY, url TEXT, threshold REAL)")
    cur.execute("CREATE TABLE IF NOT EXISTS sms (phone TEXT PRIMARY KEY)")
    con.commit()
    con.close()




def cache_set(key: str, value: str, ttl_seconds: int = 300) -> None:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    expires_at = int((datetime.utcnow() + timedelta(seconds=ttl_seconds)).timestamp())
    cur.execute("REPLACE INTO cache (key, value, expires_at) VALUES (?, ?, ?)", (key, value, expires_at))
    con.commit()
    con.close()




def cache_get(key: str) -> Optional[str]:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("SELECT value, expires_at FROM cache WHERE key = ?", (key,))
    row = cur.fetchone()
    con.close()
    if not row:
        return None
    value, expires_at = row
    if int(datetime.utcnow().timestamp()) > int(expires_at):
        return None
    return value

class Geometry(BaseModel):
    type: str = Field(..., description="Point|Polygon|LineString|MultiPolygon...")
    coordinates: Any




class Event(BaseModel):
    source: str  # e.g., "NWS", "USGS", "FEMA", "Tomorrow.io"
    event_type: str  # e.g., "Severe Weather Warning", "Earthquake", "DisasterDeclaration"
    severity: Optional[str] = None  # e.g., "Severe", "Moderate" or numeric magnitude
    certainty: Optional[str] = None  # e.g., "Observed", "Likely"
    headline: Optional[str] = None
    description: Optional[str] = None
    start_time: Optional[datetime] = None
    end_time: Optional[datetime] = None
    geometry: Optional[Geometry] = None
    properties: Dict[str, Any] = Field(default_factory=dict)




class AggregateResponse(BaseModel):
    query_lat: float
    query_lon: float
    radius_km: float
    events: List[Event]
    risk_score: float
    impacted_parcels: List[Dict[str, Any]] = Field(default_factory=list)
    generated_at: datetime

class Provider:
    name: str


    async def fetch(self, lat: float, lon: float, radius_km: float, since: Optional[datetime]) -> List[Event]:
        raise NotImplementedError




class NWSAlertsProvider(Provider):
    name = "NWS"


    async def fetch(self, lat: float, lon: float, radius_km: float, since: Optional[datetime]) -> List[Event]:
        deg = km_to_deg_latlon(radius_km)
        pts = [(lat, lon), (lat + deg, lon), (lat - deg, lon), (lat, lon + deg), (lat, lon - deg)]
        events: List[Event] = []
        headers = {"User-Agent": "DisasterDirect/1.0 (contact: you@example.com)"}
        async with httpx.AsyncClient(timeout=15.0, headers=headers) as client:
            for plat, plon in pts:
                url = f"https://api.weather.gov/alerts?point={plat:.4f},{plon:.4f}"
                if since:
                    url += f"&start={since.isoformat()}"
                r = await client.get(url)
                if r.status_code != 200:
                    continue
                data = r.json()
                for feat in data.get("features", []):
                    props = feat.get("properties", {})
                    geom = feat.get("geometry")
                    events.append(Event(
                        source=self.name,
                        event_type=props.get("event") or "Alert",
                        severity=props.get("severity"),
                        certainty=props.get("certainty"),
                        headline=props.get("headline"),
                        description=props.get("description"),
                        start_time=(datetime.fromisoformat(props.get("onset")) if props.get("onset") else None),
                        end_time=(datetime.fromisoformat(props.get("ends")) if props.get("ends") else None),
                        geometry=(Geometry(type=geom.get("type"), coordinates=geom.get("coordinates")) if geom else None),
                        properties={
                            "urgency": props.get("urgency"),
                            "areaDesc": props.get("areaDesc"),
                            "eventCode": props.get("eventCode"),
                            "parameters": props.get("parameters"),
                        }
                    ))
        return events




class USGSEarthquakesProvider(Provider):
    name = "USGS"


    async def fetch(self, lat: float, lon: float, radius_km: float, since: Optional[datetime]) -> List[Event]:
        params = {
            "format": "geojson",
            "latitude": lat,
            "longitude": lon,
            "maxradiuskm": radius_km,
            "orderby": "time",
        }
        if since:
            params["starttime"] = since.strftime("%Y-%m-%dT%H:%M:%S")
        url = "https://earthquake.usgs.gov/fdsnws/event/1/query"
        events: List[Event] = []
        async with httpx.AsyncClient(timeout=15.0) as client:
            r = await client.get(url, params=params)
            if r.status_code != 200:
                return events
            data = r.json()
            for feat in data.get("features", []):
                props = feat.get("properties", {})
                geom = feat.get("geometry")
                events.append(Event(
                    source=self.name,
                    event_type="Earthquake",
                    severity=str(props.get("mag")) if props.get("mag") is not None else None,
                    certainty="Observed",
                    headline=f"M{props.get('mag')} earthquake – {props.get('place')}",
                    description=props.get("title"),
                    start_time=datetime.fromtimestamp(props.get("time")/1000, tz=timezone.utc) if props.get("time") else None,
                    end_time=None,
                    geometry=(Geometry(type=geom.get("type"), coordinates=geom.get("coordinates")) if geom else None),
                    properties={
                        "tsunami": props.get("tsunami"),
                        "url": props.get("url"),
                        "detail": props.get("detail"),
                    }
                ))
        return events




class FEMADeclarationsProvider(Provider):
    name = "FEMA"


    async def fetch(self, lat: float, lon: float, radius_km: float, since: Optional[datetime]) -> List[Event]:
        base = "https://www.fema.gov/api/open/v2/DisasterDeclarationsSummaries"
        params = {"$top": 200, "$orderby": "declarationDate desc"}
        if since:
            params["$filter"] = f"declarationDate ge {since.strftime('%Y-%m-%d')}"
        events: List[Event] = []
        async with httpx.AsyncClient(timeout=20.0) as client:
            r = await client.get(base, params=params)
            if r.status_code != 200:
                return events
            data = r.json()
            items = data.get("DisasterDeclarationsSummaries", [])
            for it in items:
                ev = Event(
                    source=self.name,
                    event_type="DisasterDeclaration",
                    severity=it.get("incidentType"),
                    certainty="Official",
                    headline=f"FEMA {it.get('incidentType')} declaration – {it.get('state')}",
                    description=it.get("declarationTitle"),
                    start_time=datetime.fromisoformat(it.get("declarationDate")) if it.get("declarationDate") else None,
                    geometry=None,
                    properties={
                        "disasterNumber": it.get("disasterNumber"),
                        "state": it.get("state"),
                        "county": it.get("declaredCountyArea"),
                        "fyDeclared": it.get("fyDeclared"),
                        "ihProgramDeclared": it.get("ihProgramDeclared"),
                        "iaProgramDeclared": it.get("iaProgramDeclared"),
                        "paProgramDeclared": it.get("paProgramDeclared"),
                    }
                )
                events.append(ev)
        return events




class TomorrowIOProvider(Provider):
    name = "Tomorrow.io"


    async def fetch(self, lat: float, lon: float, radius_km: float, since: Optional[datetime]) -> List[Event]:
        api_key = os.getenv("TOMORROWIO_API_KEY")
        if not api_key:
            return []
        url = "https://api.tomorrow.io/v4/timelines"
        params = {
            "apikey": api_key,
            "location": f"{lat},{lon}",
            "fields": [
                "precipitationIntensity", "hailBinary", "hailSize", "windGust", "windSpeed", "weatherCode"
            ],
            "timesteps": ["hourly"],
            "units": "metric",
            "startTime": (since or now_utc() - timedelta(hours=24)).strftime("%Y-%m-%dT%H:%M:%SZ"),
            "endTime": (now_utc() + timedelta(hours=1)).strftime("%Y-%m-%dT%H:%M:%SZ"),
        }
        events: List[Event] = []
        async with httpx.AsyncClient(timeout=20.0) as client:
            r = await client.post(url, json=params)
            if r.status_code != 200:
                return events
            data = r.json()
            try:
                intervals = data["data"]["timelines"][0]["intervals"]
            except Exception:
                return events
            # Build approximate swath polygons by buffering around the query point per timestep
            for it in intervals:
                values = it.get("values", {})
                start = it.get("startTime")
                hail = values.get("hailBinary")
                hail_size = values.get("hailSize") or 0
                gust = values.get("windGust") or 0
                precip = values.get("precipitationIntensity") or 0
                # Heuristic radius (km): base 5km + wind/2 + hail_size*2, capped at 20km
                radius_km_est = min(20.0, 5.0 + (float(gust) * 0.5) + (float(hail_size) * 2.0))
                if hail or gust >= 10 or precip >= 0.5:  # trigger conditions
                    headline_bits = []
                    if hail:
                        headline_bits.append("Hail")
                    if gust:
                        headline_bits.append(f"Wind gust {gust} m/s")
                    if precip:
                        headline_bits.append(f"Precip {precip}")
                    geom: Optional[Geometry] = None
                    if shape is not None:
                        try:
                            p = Point(lon, lat)
                            poly = p.buffer(radius_km_est / 111.0, resolution=32)
                            geom = Geometry(type=poly.geom_type, coordinates=mapping(poly)["coordinates"])
                        except Exception:
                            geom = Geometry(type="Point", coordinates=[lon, lat])
                    else:
                        geom = Geometry(type="Point", coordinates=[lon, lat])
                    events.append(Event(
                        source=self.name,
                        event_type="Hyperlocal Weather (approx swath)",
                        severity=str(hail_size) if hail_size else None,
                        certainty="Likely",
                        headline=", ".join(headline_bits) or "Weather",
                        description=json.dumps(values),
                        start_time=datetime.fromisoformat(start.replace("Z","+00:00")) if start else None,
                        end_time=None,
                        geometry=geom,
                        properties={"provider": "Tomorrow.io", "radius_km_est": radius_km_est}
                    ))
        return events




# Registry of providers to run in aggregation
PROVIDERS: List[Provider] = [
    NWSAlertsProvider(),
    USGSEarthquakesProvider(),
    FEMADeclarationsProvider(),
    TomorrowIOProvider(),  # only returns data if TOMORROWIO_API_KEY is set
]




# -----------------------------
# Simple risk scoring
# -----------------------------


SEVERITY_WEIGHTS = {
    "Extreme": 5,
    "Severe": 4,
    "Moderate": 3,
    "Minor": 2,
}




def compute_risk(events: List[Event]) -> float:
    score = 0.0
    for ev in events:
        base = {"NWS":1.5, "USGS":1.8, "FEMA":1.2}.get(ev.source, 1.3)
        sev = 0.0
        if ev.severity:
            try:
                sev = float(ev.severity)
            except Exception:
                sev = SEVERITY_WEIGHTS.get(ev.severity, 2.0)
        score += base * (1.0 + sev/5.0)
    return min(100.0, round(score * 5.0, 2))

def parcels_list() -> List[Tuple[int,str,str]]:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("SELECT id, name, geojson FROM parcels ORDER BY id DESC")
    rows = cur.fetchall()
    con.close()
    return rows




def parcels_add(name: str, geojson_text: str) -> int:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("INSERT INTO parcels (name, geojson, created_at) VALUES (?, ?, ?)", (name, geojson_text, int(datetime.utcnow().timestamp())))
    con.commit()
    pid = cur.lastrowid
    con.close()
    return pid




def parcels_delete(pid: int) -> None:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("DELETE FROM parcels WHERE id=?", (pid,))
    con.commit()
    con.close()




def to_shapely(geom: Geometry):
    if shape is None or not geom:
        return None
    return shape({"type": geom.type, "coordinates": geom.coordinates})




def impacted_parcels(events: List[Event]) -> List[Dict[str, Any]]:
    if shape is None:
        return []
    # Build union of event polygons; buffer points a bit.
    event_shapes = []
    for ev in events:
        if not ev.geometry:
            continue
        s = to_shapely(ev.geometry)
        if s is None:
            continue
        # If event is a Point, buffer ~1km for neighborhood impact
        if isinstance(s, Point):
            s = s.buffer(0.009)  # ~1km in degrees
        event_shapes.append(s)
    if not event_shapes:
        return []
    union = unary_union(event_shapes)


    impacted: List[Dict[str, Any]] = []
    for pid, name, gj_text in parcels_list():
        try:
            gj = json.loads(gj_text)
            # supports FeatureCollection or single Feature/Geometry
            features = []
            if gj.get("type") == "FeatureCollection":
                features = gj.get("features", [])
            elif gj.get("type") == "Feature":
                features = [gj]
            else:
                features = [{"type":"Feature", "geometry": gj, "properties": {}}]


            touched = False
            for feat in features:
                geom = feat.get("geometry")
                if not geom:
                    continue
                s = shape(geom)
                if s.intersects(union):
                    touched = True
                    break
            if touched:
                impacted.append({"parcel_id": pid, "name": name})
        except Exception:
            continue
    return impacted

app = FastAPI(title="Disaster Direct API", version="0.2.0")




@app.on_event("startup")
def _startup():
    init_db()




@app.get("/health")
def health():
    return {"ok": True, "time": now_utc().isoformat()}




@app.post("/parcels")
async def upload_parcels(name: str = Query(..., description="Name of the parcel set (e.g., 'Dallas Parcels')"),
                         geojson: Dict[str, Any] = Body(..., description="GeoJSON FeatureCollection/Feature/Geometry")):
    try:
        text = json.dumps(geojson)
    except Exception:
        raise HTTPException(status_code=400, detail="Invalid GeoJSON")
    pid = parcels_add(name, text)
    return {"ok": True, "parcel_id": pid}




@app.get("/parcels")
async def list_parcels():
    return [{"id": pid, "name": name} for pid, name, _ in parcels_list()]




@app.delete("/parcels/{parcel_id}")
async def delete_parcel(parcel_id: int):
    parcels_delete(parcel_id)
    return {"ok": True}




@app.get("/aggregate", response_model=AggregateResponse)
async def aggregate(
    lat: float = Query(..., ge=-90, le=90),
    lon: float = Query(..., ge=-180, le=180),
    radius_km: float = Query(25.0, gt=0, le=500),
    hours_back: int = Query(72, ge=1, le=720),
):
    since = now_utc() - timedelta(hours=hours_back)


    ckey = f"agg:{lat:.4f}:{lon:.4f}:{radius_km}:{hours_back}"
    cached = cache_get(ckey)
    events: List[Event] = []
    if cached:
        raw = json.loads(cached)
        events = [Event(**e) for e in raw]
    else:
        results: List[Event] = []
        for p in PROVIDERS:
            try:
                evs = await p.fetch(lat, lon, radius_km, since)
                results.extend(evs)
            except Exception as e:
                print(f"Provider {p.name} error: {e}")
                continue
        # De-dupe by (source, headline, start_time)
        def ev_key(e: Event):
            return (e.source, e.headline or e.event_type, e.start_time.isoformat() if e.start_time else "")
        seen = set()
        for e in results:
            k = ev_key(e)
            if k in seen:
                continue
            seen.add(k)
            events.append(e)
        cache_set(ckey, json.dumps([e.model_dump() for e in events]), ttl_seconds=120)


    risk = compute_risk(events)


    # Impacted parcels (if shapely available and parcels uploaded)
    touched = impacted_parcels(events)
    return AggregateResponse(
        query_lat=lat,
        query_lon=lon,
        radius_km=radius_km,
        events=events,
        risk_score=risk,
        impacted_parcels=touched,
        generated_at=now_utc(),
    )




@app.get("/risk-score")
async def risk_score(lat: float, lon: float, radius_km: float = 25.0, hours_back: int = 72):
    resp = await aggregate(lat=lat, lon=lon, radius_km=radius_km, hours_back=hours_back)  # type: ignore
    return {"risk_score": resp.risk_score, "impacted_parcels": resp.impacted_parcels, "generated_at": now_utc().isoformat()}
