function openNew(url) {
  window.open(url, "_blank", "noopener,noreferrer");
}

function RadarLayer({ enabled }) {
  const map = useMap();
  const layerRef = useRef(null);
  useEffect(() => {
    if (!enabled) {
      if (layerRef.current) { map.removeLayer(layerRef.current); layerRef.current = null; }
      return;
    }
    const L = window.L;
    if (!L) return;
    const radar = L.tileLayer("https://tilecache.rainviewer.com/v2/radar/now/256/{z}/{x}/{y}/2/1_1.png", {
      opacity: 0.75,
      attribution: "<a href='https://www.rainviewer.com/'>RainViewer</a>"
    });
    radar.addTo(map);
    layerRef.current = radar;
    return () => { if (layerRef.current) { map.removeLayer(layerRef.current); layerRef.current = null; } };
  }, [enabled, map]);
  return null;
}

function NOAAAlertsLayer({ enabled }) {
  const map = useMap();
  const layerRef = useRef(null);
  useEffect(() => {
    if (!enabled) {
      if (layerRef.current) { map.removeLayer(layerRef.current); layerRef.current = null; }
      return;
    }
    const L = window.L;
    if (!L) return;
    const noaa = L.tileLayer.wms("https://idpgis.ncep.noaa.gov/arcgis/services/NWS_Watches_Warnings/MapServer/WMSServer", {
      layers: "1",
      format: "image/png",
      transparent: true,
      attribution: "NOAA/NWS"
    });
    noaa.addTo(map);
    layerRef.current = noaa;
    return () => { if (layerRef.current) { map.removeLayer(layerRef.current); layerRef.current = null; } };
  }, [enabled, map]);
  return null;
}

// ===== Multi-View grid (4 panes) & HLS player =====
function HlsPlayer({ src }) {
  const videoRef = useRef(null);
  useEffect(function () {
    var video = videoRef.current; if (!video || !src) return;
    if (video.canPlayType('application/vnd.apple.mpegurl')) { video.src = src; video.play().catch(function(){}); return; }
    var hls;
    (async function(){
      try {
        var mod = await import('hls.js'); var Hls = mod.default;
        if (Hls && Hls.isSupported()) {
          hls = new Hls(); hls.loadSource(src); hls.attachMedia(video);
          hls.on(Hls.Events.MANIFEST_PARSED, function(){ video.play().catch(function(){}); });
        } else { window.open(src, '_blank'); }
      } catch(e) { console.warn('HLS init failed', e); }
    })();
    return function(){ if (hls) hls.destroy(); };
  }, [src]);
  return (<video ref={videoRef} controls className="w-full h-full object-contain rounded-xl" />);
}

function FlexiblePlayer({ url }) {
  if (!url) return <div className="text-sm text-muted-foreground">No feed set</div>;
  var lower = url.toLowerCase();
  if (lower.indexOf('.m3u8') !== -1) return <HlsPlayer src={url} />;
  return <iframe title="Feed" src={url} className="w-full h-full rounded-xl" allow="autoplay; fullscreen" />;
}

function MultiView() {
  const [mv1, setMv1] = useState(localStorage.getItem('mv1') || '');
  const [mv2, setMv2] = useState(localStorage.getItem('mv2') || '');
  const [mv3, setMv3] = useState(localStorage.getItem('mv3') || '');
  const [mv4, setMv4] = useState(localStorage.getItem('mv4') || '');

  useEffect(function(){ localStorage.setItem('mv1', mv1); }, [mv1]);
  useEffect(function(){ localStorage.setItem('mv2', mv2); }, [mv2]);
  useEffect(function(){ localStorage.setItem('mv3', mv3); }, [mv3]);
  useEffect(function(){ localStorage.setItem('mv4', mv4); }, [mv4]);

  // Listen for cross-tab/provider pin events
  useEffect(function(){
    function onStorage(e){
      if (e.key === 'mv1') setMv1(localStorage.getItem('mv1')||'');
      if (e.key === 'mv2') setMv2(localStorage.getItem('mv2')||'');
      if (e.key === 'mv3') setMv3(localStorage.getItem('mv3')||'');
      if (e.key === 'mv4') setMv4(localStorage.getItem('mv4')||'');
    }
    function onCustom(){
      setMv1(localStorage.getItem('mv1')||'');
      setMv2(localStorage.getItem('mv2')||'');
      setMv3(localStorage.getItem('mv3')||'');
      setMv4(localStorage.getItem('mv4')||'');
    }
    window.addEventListener('storage', onStorage);
    window.addEventListener('mvUpdate', onCustom);
    return function(){ window.removeEventListener('storage', onStorage); window.removeEventListener('mvUpdate', onCustom); };
  }, []);

  return (
    <div className="space-y-4">
      <div className="grid md:grid-cols-4 gap-2">
        <Input value={mv1} onChange={(e)=>setMv1(e.target.value)} placeholder="Pane 1 URL (.m3u8 or https://player)" />
        <Input value={mv2} onChange={(e)=>setMv2(e.target.value)} placeholder="Pane 2 URL" />
        <Input value={mv3} onChange={(e)=>setMv3(e.target.value)} placeholder="Pane 3 URL" />
        <Input value={mv4} onChange={(e)=>setMv4(e.target.value)} placeholder="Pane 4 URL" />
      </div>
      <div className="grid md:grid-cols-2 gap-3">
        <Card className="h-[320px]"><CardContent className="p-2 h-full"><FlexiblePlayer url={mv1} /></CardContent></Card>
        <Card className="h-[320px]"><CardContent className="p-2 h-full"><FlexiblePlayer url={mv2} /></CardContent></Card>
        <Card className="h-[320px]"><CardContent className="p-2 h-full"><FlexiblePlayer url={mv3} /></CardContent></Card>
        <Card className="h-[320px]"><CardContent className="p-2 h-full"><FlexiblePlayer url={mv4} /></CardContent></Card>
      </div>
    </div>
  );
}

// ===== Inbox stream (polls /api/inbox if present) =====
function InboxStream() {
  const [items, setItems] = useState([]);
  const [query, setQuery] = useState('');
  const [onlyTree, setOnlyTree] = useState(false);
  const [onlyLines, setOnlyLines] = useState(false);

  useEffect(function(){
    let alive = true;
    let pollId;

    // Try SSE first
    const ev = new EventSource('/api/stream');
    ev.onmessage = (m) => {
      try { const obj = JSON.parse(m.data); if (alive) setItems((prev)=>[obj, ...prev]); } catch(_){}
    };
    ev.onerror = () => { ev.close(); startPoll(); };

    async function fetchAll(){
      try { const r = await fetch('/api/inbox', { cache: 'no-store' }); if (!r.ok) return; const j = await r.json(); if (alive) setItems(j); } catch(e) {}
    }
    function startPoll(){ fetchAll(); pollId = setInterval(fetchAll, 10000); }

    return function(){ alive = false; ev.close(); if (pollId) clearInterval(pollId); };
  }, []);

  const shown = items.filter((it) => {
    const t = (it.notes||'') + ' ' + (it.address||'');
    if (query && !t.toLowerCase().includes(query.toLowerCase())) return false;
    if (onlyTree && !/tree|limb|branch|oak|pine|on roof|through roof/i.test(t)) return false;
    if (onlyLines && !/line down|power line|utility|pole/i.test(t)) return false;
    return true;
  });

  return (
    <div className="space-y-3">
      <div className="grid md:grid-cols-3 gap-2 items-center">
        <Input value={query} onChange={(e)=>setQuery(e.target.value)} placeholder="Search notes/address (e.g., 'tree on roof')" />
        <div className="flex gap-3 items-center">
          <label className="flex items-center gap-2"><input type="checkbox" checked={onlyTree} onChange={(e)=>setOnlyTree(e.target.checked)} /> Trees on homes</label>
          <label className="flex items-center gap-2"><input type="checkbox" checked={onlyLines} onChange={(e)=>setOnlyLines(e.target.checked)} /> Lines down</label>
        </div>
        <div className="text-right text-sm text-muted-foreground">New items appear instantly via SSE (fallback to 10s polling)</div>
      </div>

      {!shown.length && <div className="text-sm text-muted-foreground">No matching items yet.</div>}

      <div className="grid md:grid-cols-2 gap-4">
        {shown.map((it) => (
          <Card key={it.id || it.mediaUrl}><CardContent className="p-3 space-y-2">
            <div className="font-medium">{(it.provider||'Unknown')} — {new Date(it.timestamp||Date.now()).toLocaleString()}</div>
            {it.thumbnailUrl ? (<img src={it.thumbnailUrl} alt="thumb" className="w-full rounded" />) : null}
            <div className="text-sm">{it.address ? it.address : ((it.lat && it.lon) ? (it.lat+', '+it.lon) : '')}</div>
            {it.mediaUrl ? (<a href={it.mediaUrl} target="_blank" rel="noreferrer"><Button variant="secondary">Open Media</Button></a>) : null}
            {it.notes ? (<div className="text-sm text-muted-foreground">{it.notes}</div>) : null}
          </CardContent></Card>
        ))}
      </div>
    </div>
  );
}
