function openNew(url) {
  window.open(url, "_blank", "noopener,noreferrer");
}

function RadarLayer({ enabled }) {
  const map = useMap();
  const layerRef = useRef(null);
  useEffect(() => {
    if (!enabled) {
      if (layerRef.current) { map.removeLayer(layerRef.current); layerRef.current = null; }
      return;
    }
    const L = window.L;
    if (!L) return;
    const radar = L.tileLayer("https://tilecache.rainviewer.com/v2/radar/now/256/{z}/{x}/{y}/2/1_1.png", {
      opacity: 0.75,
      attribution: "<a href='https://www.rainviewer.com/'>RainViewer</a>"
    });
    radar.addTo(map);
    layerRef.current = radar;
    return () => { if (layerRef.current) { map.removeLayer(layerRef.current); layerRef.current = null; } };
  }, [enabled, map]);
  return null;
}

function NOAAAlertsLayer({ enabled }) {
  const map = useMap();
  const layerRef = useRef(null);
  useEffect(() => {
    if (!enabled) {
      if (layerRef.current) { map.removeLayer(layerRef.current); layerRef.current = null; }
      return;
    }
    const L = window.L;
    if (!L) return;
    const noaa = L.tileLayer.wms("https://idpgis.ncep.noaa.gov/arcgis/services/NWS_Watches_Warnings/MapServer/WMSServer", {
      layers: "1",
      format: "image/png",
      transparent: true,
      attribution: "NOAA/NWS"
    });
    noaa.addTo(map);
    layerRef.current = noaa;
    return () => { if (layerRef.current) { map.removeLayer(layerRef.current); layerRef.current = null; } };
  }, [enabled, map]);
  return null;
}

function QuickMap({ radarOn, alertsOn }) {
  const [lat, setLat] = useState(32.51);
  const [lng, setLng] = useState(-84.87);
  const [z] = useState(6);
  const position = useMemo(() => [lat, lng], [lat, lng]);

  return (
    <Card className="w-full">
      <CardContent className="p-2">
        <div className="h-[420px] rounded-2xl overflow-hidden">
          <MapContainer center={position} zoom={z} className="h-full w-full">
            <TileLayer attribution='&copy; OpenStreetMap' url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png" />
            <RadarLayer enabled={radarOn} />
            <NOAAAlertsLayer enabled={alertsOn} />
          </MapContainer>
        </div>
      </CardContent>
    </Card>
  );
}

function FootageInboxSimple() {
  const [items, setItems] = useState([]);
  const [text, setText] = useState("");

  function addItem() {
    try {
      const obj = JSON.parse(text);
      obj.id = String(Date.now());
      setItems([obj, ...items]);
      setText("");
      alert("New footage received.");
    } catch (e) {
      alert("Invalid JSON");
    }
  }

  return (
    <div className="space-y-3">
      <textarea className="w-full h-28 border rounded-xl p-2" value={text} onChange={(e)=>setText(e.target.value)} placeholder='{"provider":"DroneUp","timestamp":"2025-09-08T23:00:00Z","mediaUrl":"https://.../index.m3u8","lat":32.51,"lon":-84.87,"notes":"Tree on home"}' />
      <div className="flex gap-2">
        <Button onClick={addItem}>Add</Button>
      </div>
      <div className="grid md:grid-cols-2 gap-3">
        {items.map(it => (
          <Card key={it.id}><CardContent className="p-3 space-y-2">
            <div className="font-medium">{it.provider} — {it.timestamp}</div>
            <div className="text-sm">GPS: {it.lat}, {it.lon}</div>
            {it.mediaUrl && <a className="inline-block" href={it.mediaUrl} target="_blank" rel="noreferrer"><Button variant="secondary">Open Media</Button></a>}
            {it.notes && <div className="text-sm text-muted-foreground">{it.notes}</div>}
          </CardContent></Card>
        ))}
      </div>
    </div>
  );
}

function FootageInbox() {
  const [items, setItems] = useState(() => []);
  const [draft, setDraft] = useState("");

  async function reverseGeocode(lat, lon) {
    try {
      const r = await fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lon}`);
      const j = await r.json();
      return j?.display_name || "";
    } catch (_) { return ""; }
  }

  async function addItemFromJSON() {
    try {
      const obj = JSON.parse(draft);
      if (!obj.address && obj.lat && obj.lon) {
        obj.address = await reverseGeocode(obj.lat, obj.lon);
      }
      obj.id = Date.now().toString();
      setItems((prev) => [obj, ...prev]);
      setDraft("");
      alert("New footage received and alert sent to Ops.");
    } catch (e) {
      alert("Invalid JSON. Paste the example and modify it.");
    }
  }

  return (
    <Card>
      <CardContent className="p-4 space-y-3">
        <div className="text-lg font-semibold">Footage Inbox (test harness)</div>
        <div className="text-sm text-muted-foreground">Paste JSON payloads from providers to simulate webhook ingest. We’ll reverse‑geocode to get an address if only GPS is provided.</div>
        <textarea className="w-full h-28 border rounded-xl p-2" value={draft} onChange={(e)=>setDraft(e.target.value)} placeholder='Paste JSON…' />
        <div className="flex gap-2">
          <Button onClick={addItemFromJSON}>Add Footage</Button>
        </div>
        <div className="grid md:grid-cols-2 gap-4">
          {items.map((it) => (
            <Card key={it.id}>
              <CardContent className="p-3 space-y-2">
                <div className="font-medium">{it.provider} — {new Date(it.timestamp||Date.now()).toLocaleString()}</div>
                {it.thumbnailUrl && (<img src={it.thumbnailUrl} alt="thumb" className="w-full rounded" />)}
                <div className="text-sm">{it.address || (it.lat && it.lon ? `${it.lat}, ${it.lon}` : "")}</div>
                {it.mediaUrl && (
                  <div className="flex gap-2">
                    <a href={it.mediaUrl} target="_blank" rel="noreferrer">
                      <Button variant="secondary">Open Media</Button>
                    </a>
                  </div>
                )}
                {it.notes && <div className="text-sm text-muted-foreground">{it.notes}</div>}
              </CardContent>
            </Card>
          ))}
        </div>
      </CardContent>
    </Card>
  );
}
