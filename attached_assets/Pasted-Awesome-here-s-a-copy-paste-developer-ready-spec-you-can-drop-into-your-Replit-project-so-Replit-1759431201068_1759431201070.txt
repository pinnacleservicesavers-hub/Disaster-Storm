Awesome — here’s a copy-paste, developer-ready spec you can drop into your Replit project so “Replit understands what you want.” It wires up a Claims Agent that: answers calls, collects claim info, emails the adjuster a claim packet, places follow-up calls, and uses ElevenLabs for natural-sounding voice (played in calls) + Twilio for phone, SendGrid for email, and Supabase for storage.

📣 “READ ME, REPLIT” — Claims Agent (Voice + Email + Follow-ups)
✅ What to build

An AI-assisted Claims Agent that:

Answers homeowner calls → collects name, address, insurer, policy, damage type, photos link.

Creates a claim record in Supabase.

Gets adjuster contact (we provide an input form for adjuster name/email/phone if unknown).

Emails the claim packet (invoice, photos, notes) to the adjuster.

Calls and emails follow-ups on an automated schedule, escalating smartly.

Uses ElevenLabs voice to sound human on outbound messages and follow-ups.

🧰 Stack & Services

Runtime: Node + Express (TypeScript)

Telephony: Twilio (calls, SMS, status webhooks)

Voice: ElevenLabs (generate MP3 voice lines we can play in Twilio calls)

Email: SendGrid (transactional claim emails)

DB/Auth/Storage: Supabase

Scheduler: lightweight in-app agent (polls for due follow-ups)

🧱 Supabase schema (run in Supabase SQL editor)
-- Homeowner/customer
create table customers (
  id uuid primary key default gen_random_uuid(),
  name text,
  phone text,
  email text,
  address text,
  city text,
  state text,
  zip text,
  created_at timestamptz default now()
);

-- Adjusters
create table adjusters (
  id uuid primary key default gen_random_uuid(),
  name text,
  email text,
  phone text,
  carrier text,
  notes text,
  created_at timestamptz default now()
);

-- Claims
create table claims (
  id uuid primary key default gen_random_uuid(),
  customer_id uuid references customers(id) on delete cascade,
  adjuster_id uuid references adjusters(id),
  insurer text,
  policy_number text,
  claim_number text,
  status text default 'intake',  -- intake|packet_sent|in_review|approved|paid|escalation
  damage_type text,              -- tree on home/building/structure/vehicle, etc.
  severity text,                 -- very_urgent|urgent|standard
  photos_url text,               -- link to photo folder
  invoice_url text,
  packet_email_id text,          -- SendGrid message id
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

-- Contact log (calls/sms/emails)
create table contact_log (
  id uuid primary key default gen_random_uuid(),
  claim_id uuid references claims(id) on delete cascade,
  channel text,   -- call|sms|email
  direction text, -- in|out
  to_addr text,
  from_addr text,
  status text,
  summary text,
  meta jsonb,
  created_at timestamptz default now()
);

-- Follow-up tasks
create table tasks (
  id uuid primary key default gen_random_uuid(),
  claim_id uuid references claims(id) on delete cascade,
  type text,           -- followup_call|followup_email|escalation_call|final_notice
  due_at timestamptz,
  status text default 'pending', -- pending|done|skipped
  payload jsonb,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

create index tasks_due_idx on tasks (due_at) where status='pending';

import sg from '@sendgrid/mail';
sg.setApiKey(process.env.SENDGRID_API_KEY!);

export async function sendClaimPacket(opts: {
  toEmail: string;
  toName?: string;
  subject: string;
  html: string;
  attachments?: { filename: string; type: string; content: string }[];
}) {
  const msg = {
    to: { email: opts.toEmail, name: opts.toName },
    from: { email: process.env.CLAIMS_FROM_EMAIL!, name: process.env.CLAIMS_FROM_NAME! },
    subject: opts.subject,
    html: opts.html,
    attachments: opts.attachments
  };
  const res = await sg.send(msg as any);
  return res[0]?.headers?.['x-message-id'] || res[0]?.headers?.['x-message-id'.toLowerCase()];
}


server/lib/eleven.ts

import fetch from 'node-fetch';

const BASE = 'https://api.elevenlabs.io/v1';

export async function synthesizeLine(text: string): Promise<string> {
  // Returns a public URL to an MP3 (we’ll store to Supabase storage or temp host in Replit).
  // For simplicity, request the audio buffer and write to /tmp, then serve via /media route.
  const voiceId = process.env.ELEVENLABS_VOICE_ID!;
  const resp = await fetch(`${BASE}/text-to-speech/${voiceId}`, {
    method: 'POST',
    headers: {
      'xi-api-key': process.env.ELEVENLABS_API_KEY!,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      text,
      model_id: "eleven_multilingual_v2",
      voice_settings: { stability: 0.5, similarity_boost: 0.8 }
    })
  });

  const buf = Buffer.from(await resp.arrayBuffer());
  const file = `/tmp/line-${Date.now()}.mp3`;
  await Bun.write(file, buf); // If Bun not available, use fs. If error, switch to fs:
  return file;
}


If Bun isn’t available on your Replit, replace the last two lines with Node fs:

// import { writeFileSync } from 'fs';

import { Router } from 'express';
import { supabase } from '../lib/db';
import { TWILIO_FROM } from '../lib/twilio';

const router = Router();

/**
 * Twilio Voice Webhook (POST)
 * Set Voice URL in Twilio number to: {PUBLIC_BASE_URL}/intake/voice
 */
router.post('/voice', async (req, res) => {
  // Simple IVR using TwiML Gather (speech)
  const hello = `
    <Response>
      <Gather input="speech" action="/intake/step1" method="POST" timeout="6">
        <Say>Hi, this is Strategic Land Management Claims Assistant.
        Please say your full name after the beep.</Say>
      </Gather>
      <Say>Sorry, I didn't catch that.</Say>
      <Redirect>/intake/voice</Redirect>
    </Response>`;
  res.type('text/xml').send(hello);
});

router.post('/step1', async (req, res) => {
  const name = req.body.SpeechResult || 'Unknown';
  const twiml = `
    <Response>
      <Gather input="speech" action="/intake/step2?name=${encodeURIComponent(name)}" method="POST" timeout="6">
        <Say>Thanks, ${name}. Please say your phone number slowly.</Say>
      </Gather>
      <Say>Sorry, I didn't catch that.</Say>
      <Redirect>/intake/voice</Redirect>
    </Response>`;
  res.type('text/xml').send(twiml);
});

router.post('/step2', async (req, res) => {
  const { name } = req.query as any;
  const phone = req.body.SpeechResult || req.body.From || '';
  const twiml = `
    <Response>
      <Gather input="speech" action="/intake/step3?name=${encodeURIComponent(name)}&phone=${encodeURIComponent(phone)}" method="POST" timeout="8">
        <Say>Please say your full street address, city, and state.</Say>
      </Gather>
      <Say>Sorry, I didn't catch that.</Say>
      <Redirect>/intake/voice</Redirect>
    </Response>`;
  res.type('text/xml').send(twiml);
});

router.post('/step3', async (req, res) => {
  const { name, phone } = req.query as any;
  const address = req.body.SpeechResult || '';
  const twiml = `
    <Response>
      <Gather input="speech" action="/intake/step4?name=${encodeURIComponent(name)}&phone=${encodeURIComponent(phone)}&address=${encodeURIComponent(address)}" method="POST" timeout="8">
        <Say>What insurer? For example, State Farm, USAA, or Allstate.</Say>
      </Gather>
      <Say>Sorry, I didn't catch that.</Say>
      <Redirect>/intake/voice</Redirect>
    </Response>`;
  res.type('text/xml').send(twiml);
});

router.post('/step4', async (req, res) => {
  const { name, phone, address } = req.query as any;
  const insurer = req.body.SpeechResult || '';
  const twiml = `
    <Response>
      <Gather input="speech" action="/intake/finish?name=${encodeURIComponent(name)}&phone=${encodeURIComponent(phone)}&address=${encodeURIComponent(address)}&insurer=${encodeURIComponent(insurer)}" method="POST" timeout="8">
        <Say>Briefly describe the damage. For example, tree on home or tree on shed.</Say>
      </Gather>
      <Say>Sorry, I didn't catch that.</Say>
      <Redirect>/intake/voice</Redirect>
    </Response>`;
  res.type('text/xml').send(twiml);
});

router.post('/finish', async (req, res) => {
  const { name, phone, address, insurer } = req.query as any;
  const damage_type = (req.body.SpeechResult || '').toLowerCase();
  const severity = /(tree on home|house)/.test(damage_type) ? 'very_urgent'
                 : /(tree on|structure|building)/.test(damage_type) ? 'urgent'
                 : 'standard';

  // create customer + claim
  const { data: cust } = await supabase
    .from('customers')
    .insert([{ name, phone, address }])
    .select()
    .single();

  const { data: claim } = await supabase
    .from('claims')
    .insert([{ customer_id: cust.id, insurer, damage_type, severity, status: 'intake' }])
    .select()
    .single();

  // create first follow-up task: request adjuster contact if missing
  await supabase.from('tasks').insert([{
    claim_id: claim.id,
    type: 'followup_email',
    due_at: new Date(Date.now() + 60 * 60 * 1000).toISOString(), -- /* 1h */
    payload: { reason: 'request_adjuster_contact' }
  }]);

  const done = `
    <Response>
      <Say>Thank you. Your claim intake is complete. Our team will email your insurer and follow up. Goodbye.</Say>
      <Hangup/>
    </Response>`;
  res.type('text/xml').send(done);
});

export default router;

✉️ Send claim packet to adjuster

server/routes/claims.ts

import { Router } from 'express';
import { supabase } from '../lib/db';
import { sendClaimPacket } from '../lib/email';

const router = Router();

/**
 * POST /claims/:id/send-packet
 * body: { adjuster: { name, email, phone }, claim_number?, policy_number?, invoice_url?, photos_url? }
 */
router.post('/:id/send-packet', async (req, res) => {
  const claimId = req.params.id;
  const { data: claim } = await supabase.from('claims').select(`
    id, insurer, claim_number, policy_number,
    customers ( name, phone, email, address )
  `).eq('id', claimId).single();

  if (!claim) return res.status(404).json({ error: 'Claim not found' });

  // upsert adjuster
  const adjPayload = req.body.adjuster || {};
  const { data: adj } = await supabase.from('adjusters')
    .upsert({ name: adjPayload.name, email: adjPayload.email, phone: adjPayload.phone, carrier: claim.insurer })
    .select().single();

  await supabase.from('claims').update({
    adjuster_id: adj.id,
    claim_number: req.body.claim_number || claim.claim_number,
    policy_number: req.body.policy_number || claim.policy_number,
    invoice_url: req.body.invoice_url,
    photos_url: req.body.photos_url,
    status: 'packet_sent'
  }).eq('id', claimId);

  const html = `
    <p>Dear ${adj.name || 'Adjuster'},</p>
    <p>Please find the claim packet for the insured below:</p>
    <ul>
      <li><b>Name:</b> ${claim.customers.name}</li>
      <li><b>Phone:</b> ${claim.customers.phone}</li>
      <li><b>Address:</b> ${claim.customers.address}</li>
      <li><b>Insurer:</b> ${claim.insurer}</li>
      <li><b>Policy #:</b> ${req.body.policy_number || claim.policy_number || '(pending)'}</li>
      <li><b>Claim #:</b> ${req.body.claim_number || claim.claim_number || '(pending)'}</li>
      <li><b>Invoice:</b> <a href="${req.body.invoice_url}">View Invoice</a></li>
      <li><b>Photos:</b> <a href="${req.body.photos_url}">View Photos</a></li>
    </ul>
    <p>Please confirm receipt and advise on payment or next steps.</p>
    <p>Thank you,<br/>Strategic Land Management Claims</p>
  `;

  const msgId = await sendClaimPacket({
    toEmail: adj.email,
    toName: adj.name,
    subject: `Claim Packet – ${claim.customers.name} – ${claim.insurer}`,
    html
  });

  await supabase.from('claims').update({ packet_email_id: msgId }).eq('id', claimId);

  // schedule follow-ups: 48h email, 72h call
  const now = Date.now();
  await supabase.from('tasks').insert([
    { claim_id: claimId, type: 'followup_email', due_at: new Date(now + 48*3600*1000).toISOString(), payload: { template: 'gentle_reminder' } },
    { claim_id: claimId, type: 'followup_call',  due_at: new Date(now + 72*3600*1000).toISOString(), payload: { script: 'status_check' } }
  ]);

  res.json({ ok: true, message_id: msgId });
});

export default router;

🤖 Agent loop (runs every minute; executes due tasks)

server/routes/agent.ts

import { Router } from 'express';
import { supabase } from '../lib/db';
import { twilio, TWILIO_FROM } from '../lib/twilio';
import { sendClaimPacket } from '../lib/email';
// import { synthesizeLine } from '../lib/eleven';  // optional: generate spoken lines

const router = Router();

router.get('/run', async (_req, res) => {
  const now = new Date().toISOString();
  const { data: tasks } = await supabase.from('tasks')
    .select('id, type, claim_id, payload')
    .lte('due_at', now)
    .eq('status', 'pending')
    .limit(10);

  for (const t of tasks || []) {
    const { data: claim } = await supabase.from('claims').select(`
      id, insurer, claim_number, policy_number, status,
      customers ( name, phone, email, address ),
      adjusters ( name, email, phone )
    `).eq('id', t.claim_id).single();

    try {
      if (t.type === 'followup_email') {
        const html = `
          <p>Hi ${claim.adjusters?.name || 'Adjuster'},</p>
          <p>Following up on claim for ${claim.customers.name} (${claim.insurer}). Please advise on payment status or next steps.</p>
          <p>Thanks,<br/>SLM Claims</p>`;
        await sendClaimPacket({
          toEmail: claim.adjusters?.email,
          toName: claim.adjusters?.name,
          subject: `Follow-up: ${claim.customers.name} – ${claim.insurer}`,
          html
        });
      }

      if (t.type === 'followup_call' && claim.adjusters?.phone) {
        // Simple Twilio outbound call playing a polite status line.
        const statusLine = encodeURIComponent(
          `Hello, this is Strategic Land Management following up on ${claim.customers.name}'s claim with ${claim.insurer}. ` +
          `Please return this call or reply to our email with payment status. Thank you.`
        );
        await twilio.calls.create({
          to: claim.adjusters.phone,
          from: TWILIO_FROM,
          url: `${process.env.PUBLIC_BASE_URL}/webhooks/twilio/playline?text=${statusLine}`
        });
      }

      await supabase.from('tasks').update({ status: 'done', updated_at: new Date().toISOString() }).eq('id', t.id);
    } catch (e) {
      // keep pending or log error
      await supabase.from('tasks').update({ status: 'pending', updated_at: new Date().toISOString() }).eq('id', t.id);
    }
  }

  res.json({ ran: (tasks || []).length });
});

export default router;


server/routes/webhooks.ts

import { Router } from 'express';
import { createReadStream } from 'fs';
import { synthesizeLine } from '../lib/eleven';

const router = Router();

/**
 * Twilio will fetch this URL to play a spoken line to the adjuster.
 * We synthesize with ElevenLabs to /tmp and then serve it.
 */
router.post('/twilio/status', (_req, res) => res.sendStatus(200));

router.get('/twilio/playline', async (req, res) => {
  const text = decodeURIComponent((req.query.text as string) || 'Hello from S L M');
  const mp3Path = await synthesizeLine(text);
  const twiml = `
    <Response>
      <Play>${process.env.PUBLIC_BASE_URL}/media?file=${encodeURIComponent(mp3Path)}</Play>
      <Hangup/>
    </Response>`;
  res.type('text/xml').send(twiml);
});

// serve synthesized mp3s from /tmp
router.get('/media', (req, res) => {
  const file = req.query.file as string;
  if (!file?.startsWith('/tmp/')) return res.sendStatus(403);
  res.type('audio/mpeg');
  createReadStream(file).pipe(res);
});

export default router;

import express from 'express';

      await fetch(`${process.env.PUBLIC_BASE_URL}/agent/run`);
    } catch {}
  }, 60_000);
});

⚖️ Compliance notes (important)

Call recording / consent laws: If you record or transcribe calls, play a consent line first and store a yes/no flag. (Two-party consent states require both parties’ consent.)

Adjuster systems: We’re contacting via email and phone; we are not logging into carrier systems.

PHI/PII: Store only what you need (name, address, policy/claim) and lock down Supabase RLS if you expose a front-end.
