import { db } from "./index";
import { leadCampaigns, campaignRuns, campaignLeads, followUpTasks } from "../../shared/schema";
import { eq, and, sql } from "drizzle-orm";

export async function createCampaign(userId: string, campaign: any) {
  const [created] = await db.insert(leadCampaigns).values({
    userId,
    ...campaign
  }).returning();

  return created;
}

export async function getCampaigns(userId: string) {
  return db.select().from(leadCampaigns).where(eq(leadCampaigns.userId, userId));
}

export async function setCampaignEnabled(userId: string, campaignId: string, enabled: boolean) {
  const [updated] = await db.update(leadCampaigns)
    .set({ enabled })
    .where(and(eq(leadCampaigns.id, campaignId), eq(leadCampaigns.userId, userId)))
    .returning();
  return updated;
}

export async function createRun(userId: string, campaignId: string) {
  const [run] = await db.insert(campaignRuns).values({
    userId,
    campaignId,
    status: "running"
  }).returning();
  return run;
}

export async function completeRun(runId: string, updates: any) {
  const [updated] = await db.update(campaignRuns)
    .set({ ...updates, status: "completed", completedAt: new Date() })
    .where(eq(campaignRuns.id, runId))
    .returning();
  return updated;
}

export async function saveCampaignLead(runId: string, campaignId: string, userId: string, lead: any) {
  const [saved] = await db.insert(campaignLeads).values({
    campaignRunId: runId,
    campaignId,
    userId,
    leadExternalId: lead.external_id || lead.place_id || lead.company_name,
    leadSnapshot: lead,
    score: lead.score || 0,
    scoreLabel: lead.score_label || "cold"
  }).returning();

  return saved;
}

export async function getTasksDueToday(userId: string) {
  return db.select()
    .from(followUpTasks)
    .where(and(
      eq(followUpTasks.userId, userId),
      eq(followUpTasks.status, "pending"),
      sql`DATE(${followUpTasks.dueAt}) = CURRENT_DATE`
    ))
    .orderBy(followUpTasks.dueAt);
}
