1) Add Last-Modified support

Put these helpers near your cache helpers (you already have crypto, jsonETag, sendCachedJSON). Replace sendCachedJSON with this version that also handles If-Modified-Since:

// ---- HTTP cache helpers (ETag + Last-Modified + 304) ----
function jsonETag(obj) {
  const str = JSON.stringify(obj);
  const hash = crypto.createHash("sha1").update(str).digest("hex");
  return `W/"${hash}"`;
}

/**
 * Send JSON with Cache-Control, ETag, and Last-Modified.
 * @param {IncomingMessage} req
 * @param {ServerResponse} res
 * @param {*} body - JSON-serializable body
 * @param {number} clientMaxAgeSec - Cache-Control max-age (seconds)
 * @param {number} lastModifiedMs - epoch ms to use for Last-Modified (usually your cache timestamp)
 */
function sendCachedJSON(req, res, body, clientMaxAgeSec = 60, lastModifiedMs = Date.now()) {
  const etag = jsonETag(body);
  const inm = req.headers["if-none-match"];
  const ims = req.headers["if-modified-since"]; // RFC 7232

  const lastModStr = new Date(lastModifiedMs).toUTCString();
  res.setHeader("Cache-Control", `public, max-age=${clientMaxAgeSec}`);
  res.setHeader("ETag", etag);
  res.setHeader("Last-Modified", lastModStr);
  res.setHeader("Content-Type", "application/json; charset=utf-8");

  // Revalidation: ETag takes precedence if present
  if (inm && inm === etag) {
    res.statusCode = 304;
    return res.end();
  }
  if (ims && Date.parse(ims) >= lastModifiedMs) {
    res.statusCode = 304;
    return res.end();
  }

  res.status(200).end(JSON.stringify(body));
}

2) /health route with cache stats

Add basic counters and a /health endpoint.

A) Counters (near your cache code)
// --- stats ---
const STATS = {
  cacheHits: 0,
  cacheMisses: 0,
  evictions: 0,
  sweeps: 0,
  startedAt: Date.now()
};

// Patch eviction points to increment STATS.evictions
function setCache(key, data, ttlMs) {
  while (CACHE.size >= CACHE_LIMIT) {
    const oldestKey = CACHE.keys().next().value;
    CACHE.delete(oldestKey);
    STATS.evictions++;
  }
  CACHE.set(key, { at: now(), ttl: ttlMs, data });
}

function getFreshEntry(key) {
  const v = CACHE.get(key);
  if (!v) return null;
  if (now() - v.at > v.ttl) {
    CACHE.delete(key);
    STATS.evictions++; // expired
    return null;
  }
  // LRU bump
  CACHE.delete(key);
  CACHE.set(key, v);
  return v;
}

// sweep expired
setInterval(() => {
  const t = now();
  let removed = 0;
  for (const [k, v] of CACHE.entries()) {
    if (t - v.at > v.ttl) { CACHE.delete(k); removed++; }
  }
  if (removed) STATS.evictions += removed;
  STATS.sweeps++;
}, SWEEP_EVERY_MS).unref?.();


Use getFreshEntry instead of getCache so you can access both data and at.

app.get("/api/geo/search", async (req, res) => {
  try {
    const q = (req.query.q || "").trim();
    if (!q) return sendCachedJSON(req, res, [], 300, Date.now());

    const cacheKey = `geo:${q.toLowerCase()}`;
    const v = getFreshEntry(cacheKey);
    if (v) {
      STATS.cacheHits++;
      return sendCachedJSON(req, res, v.data, 300, v.at);
    }
    STATS.cacheMisses++;

    const url = `https://nominatim.openstreetmap.org/search?format=json&limit=5&q=${encodeURIComponent(q)}`;
    const r = await fetch(url, { headers: { "User-Agent": UA } });
    if (!r.ok) throw new Error("Geocoding search failed");
    const rows = await r.json();
    const results = rows.map(x => ({ label: x.display_name, lat: parseFloat(x.lat), lon: parseFloat(x.lon) }));

    setCache(cacheKey, results, 5 * 60 * 1000); // 5m
    return sendCachedJSON(req, res, results, 300, Date.now());
  } catch (e) {
    res.status(500).json({ error: e.message || "search failed" });
  }
});

4) Optional: swap to lru-cache package (drop-in)

If you prefer a tested LRU, add it to package.json dependencies:

"dependencies": {
  "cors": "^2.8.5",
  "express": "^4.19.2",
  "lru-cache": "^11.0.1"
}


Then replace your manual CACHE with:

const LRU = require("lru-cache");
const cache = new LRU({
  max: 500,              // entries
  ttl: 1000 * 60 * 5,    // default TTL 5m (can override per set)
  updateAgeOnGet: true
});

function setCache(key, data, ttlMs) { cache.set(key, { data, at: Date.now() }, { ttl: ttlMs }); }
function getFreshEntry(key) {
  const v = cache.get(key);
  return v ? { data: v.data, at: v.at } : null;
}


Everything else (routes + sendCachedJSON) stays the same.

server.js (add):

const { Redis } = require("@upstash/redis");
const redis = new Redis({
  url: process.env.UPSTASH_REDIS_REST_URL,
  token: process.env.UPSTASH_REDIS_REST_TOKEN
});

// Redis-backed get/set with TTL (seconds)
async function rSet(key, value, ttlSec) {
  await redis.set(key, JSON.stringify({ at: Date.now(), data: value }), { ex: ttlSec });
}
async function rGet(key) {
  const raw = await redis.get(key);
  if (!raw) return null;
  try { return typeof raw === "string" ? JSON.parse(raw) : raw; } catch { return null; }
}
