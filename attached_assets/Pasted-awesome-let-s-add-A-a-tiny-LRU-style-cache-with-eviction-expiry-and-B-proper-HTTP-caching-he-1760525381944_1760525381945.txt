A) Replace your current cache with an LRU + sweeper

Add/replace the cache section near the top of server.js:

// --- LRU-ish in-memory cache with TTL + eviction ---
const crypto = require("crypto");

const CACHE = new Map(); // key -> { at:number, ttl:number, data:any }
const CACHE_LIMIT = 500; // max keys stored
const SWEEP_EVERY_MS = 60 * 1000; // sweep once per minute

function now() { return Date.now(); }

function setCache(key, data, ttlMs) {
  // Evict if over limit (oldest first)
  while (CACHE.size >= CACHE_LIMIT) {
    const oldestKey = CACHE.keys().next().value;
    CACHE.delete(oldestKey);
  }
  CACHE.set(key, { at: now(), ttl: ttlMs, data });
}

function getCache(key) {
  const v = CACHE.get(key);
  if (!v) return null;
  if (now() - v.at > v.ttl) {
    CACHE.delete(key);
    return null;
  }
  // LRU bump: re-insert to move to newest position
  CACHE.delete(key);
  CACHE.set(key, v);
  return v.data;
}

// Periodic sweep for expired entries
setInterval(() => {
  const t = now();
  for (const [k, v] of CACHE.entries()) {
    if (t - v.at > v.ttl) CACHE.delete(k);
  }
}, SWEEP_EVERY_MS).unref?.();


This keeps about ~500 hot entries, drops oldest when full, and sweeps expired items every minute.

B) Add HTTP caching headers (+ optional ETag/304)

Drop these helpers near the cache code:

// ---- HTTP cache helpers ----
function jsonETag(obj) {
  // Weak ETag based on stable string of JSON (sufficient for this use)
  const str = JSON.stringify(obj);
  const hash = crypto.createHash("sha1").update(str).digest("hex");
  return `W/"${hash}"`;
}

function sendCachedJSON(req, res, body, clientMaxAgeSec = 60) {
  const etag = jsonETag(body);
  const inm = req.headers["if-none-match"];

  res.setHeader("Cache-Control", `public, max-age=${clientMaxAgeSec}`);
  res.setHeader("ETag", etag);
  res.setHeader("Content-Type", "application/json; charset=utf-8");

  if (inm && inm === etag) {
    res.status(304).end(); // client can use its cached copy
    return;
  }
  res.status(200).send(JSON.stringify(body));
}

app.get("/api/geo/search", async (req, res) => {
  try {
    const q = (req.query.q || "").trim();
    if (!q) return sendCachedJSON(req, res, [], 300);

    const cacheKey = `geo:${q.toLowerCase()}`;
    const hit = getCache(cacheKey);
    if (hit) return sendCachedJSON(req, res, hit, 300); // 5 minutes client cache

    const url = `https://nominatim.openstreetmap.org/search?format=json&limit=5&q=${encodeURIComponent(q)}`;
    const r = await fetch(url, { headers: { "User-Agent": UA } });
    if (!r.ok) throw new Error("Geocoding search failed");
    const rows = await r.json();
    const results = rows.map(x => ({ label: x.display_name, lat: parseFloat(x.lat), lon: parseFloat(x.lon) }));

    setCache(cacheKey, results, 5 * 60 * 1000); // 5 min server cache

    res.status(500).json({ error: e.message || "search failed" });
