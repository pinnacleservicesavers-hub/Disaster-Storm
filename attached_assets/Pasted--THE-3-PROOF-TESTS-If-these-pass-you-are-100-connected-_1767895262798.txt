✅ TEST #3 — Hit a DB Test URL (Best proof)

This is the most reliable proof.

Add this API route to your server:

If you’re using Express (rest-express) add this file:

✅ server/routes/db-test.ts
import { Pool } from "pg";

export function registerDbTest(app: any) {
  app.get("/api/db-test", async (_req: any, res: any) => {
    if (!process.env.DATABASE_URL) {
      return res.json({
        connected: false,
        error: "DATABASE_URL missing"
      });
    }

    const pool = new Pool({ connectionString: process.env.DATABASE_URL });

    try {
      const result = await pool.query("SELECT NOW() as now");
      return res.json({
        connected: true,
        now: result.rows[0].now
      });
    } catch (err) {
      return res.status(500).json({
        connected: false,
        error: String(err)
      });
    }
  });
}


Then register it in your server/index.ts:

import { registerDbTest } from "./routes/db-test";
registerDbTest(app);


✅ Now visit:

/api/db-test

✅ If you see:
{ "connected": true, "now": "..." }


🔥 You are 100% connected.

❌ If you see:
{ "connected": false, "error": "DATABASE_URL missing" }


❌ Not connected.
