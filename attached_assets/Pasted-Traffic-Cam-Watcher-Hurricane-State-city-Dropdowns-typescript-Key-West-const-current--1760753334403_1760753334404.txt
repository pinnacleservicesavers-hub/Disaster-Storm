  const onStateChange = (newState: string | null) => {
    const cities = newState ? HURRICANE_DATA[newState] : [];
    const nextCity = cities?.length ? cities[0] : null;
    setValue({ state: newState, city: nextCity });
  };


  const onCityChange = (newCity: string | null) => {
    setValue({ state: current.state, city: newCity });
  };


  const cityOptions = useMemo(() => {
    return current.state ? (HURRICANE_DATA[current.state] ?? []) : [];
  }, [current.state]);


  return (
    <div className={"w-full max-w-2xl "+ (className ?? "") }>
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        <div className="flex flex-col gap-2">

          <select
            className="w-full rounded-2xl border px-4 py-3 shadow-sm focus:outline-none focus:ring-2"
            value={current.state ?? ""}
            onChange={(e) => onStateChange(e.target.value || null)}
          >
            <option value="" disabled>
              Select a state…
            </option>
            {stateOptions.map((s) => (
              <option key={s} value={s}>

              </option>
            ))}

            value={current.city ?? ""}

            disabled={!current.state}
          >

              ))
            ) : (
