from fastapi.responses import PlainTextResponse


@app.get("/export/events.csv")
async def export_events_csv(lat: float, lon: float, radius_km: float = 25.0, hours_back: int = 72):
    agg = await aggregate(lat=lat, lon=lon, radius_km=radius_km, hours_back=hours_back)  # type: ignore
    lines = ["source,event_type,severity,headline,start_time,description"]
    def esc(s):
        if s is None: return ""
        return str(s).replace('"','""')
    for e in agg.events:
        lines.append(
            f'"{esc(e.source)}","{esc(e.event_type)}","{esc(e.severity)}","{esc(e.headline)}","{esc(e.start_time.isoformat() if e.start_time else "")}","{esc(e.description)}"'
        )
    csv = "
".join(lines)
    return PlainTextResponse(csv, media_type="text/csv", headers={"Content-Disposition": "attachment; filename=events.csv"})


@app.get("/export/impacted.csv")
async def export_impacted_csv(lat: float, lon: float, radius_km: float = 25.0, hours_back: int = 72):
    agg = await aggregate(lat=lat, lon=lon, radius_km=radius_km, hours_back=hours_back)  # type: ignore
    lines = ["parcel_id,name"]
    for p in agg.impacted_parcels:
        name = str(p.get("name",""))
        lines.append(f'{p.get("parcel_id")},"{name.replace("\"","\"\"")}"')
    csv = "
".join(lines)
    return PlainTextResponse(csv, media_type="text/csv", headers={"Content-Disposition": "attachment; filename=impacted_parcels.csv"})
