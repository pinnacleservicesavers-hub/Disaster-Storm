// index.js - Replit-ready minimal proxy + cache for property + enrichment APIs
require('dotenv').config();
const express = require('express');
const fetch = require('node-fetch');
const NodeCache = require('node-cache');
const helmet = require('helmet');
const morgan = require('morgan');

const app = express();
app.use(express.json());
app.use(helmet());
app.use(morgan('combined'));

const cache = new NodeCache({ stdTTL: 60 * 60, checkperiod: 120 }); // 1 hour cache

// Config: set these in Replit environment variables (See instructions below)
const ESTATED_KEY = process.env.ESTATED_API_KEY || '';   // or ATTOM_API_KEY
const ATTOM_KEY = process.env.ATTOM_API_KEY || '';
const WHITEPAGES_KEY = process.env.WHITEPAGES_API_KEY || '';
const PROVIDER = process.env.PROVIDER || 'estated'; // 'estated' or 'attom'

// Utility: cache wrapper
async function cached(key, ttlSeconds, fn) {
  const v = cache.get(key);
  if (v) return v;
  const result = await fn();
  cache.set(key, result, ttlSeconds);
  return result;
}

// Endpoint: property lookup by address
// Example: GET /api/property?address=5385 Westwood Drive, Columbus GA
app.get('/api/property', async (req, res) => {
  try {
    const q = (req.query.address || '').trim();
    if (!q) return res.status(400).json({ error: 'address query required' });

    const cacheKey = `prop:${q}`;
    const data = await cached(cacheKey, 60 * 60, async () => {
      if (PROVIDER === 'attom' && ATTOM_KEY) {
        // ATTOM example: see docs for exact endpoint & query parameters
        const url = `https://api.gateway.attomdata.com/propertyapi/v1.0.0/property/detail?address=${encodeURIComponent(q)}`;
        const r = await fetch(url, { headers: { 'apikey': ATTOM_KEY }});
        if (!r.ok) throw new Error('ATTOM error ' + r.status);
        return r.json();
      } else if (ESTATED_KEY) {
        // Estated simple endpoint (v4): https://estated.com/developers/docs/v4
        const url = `https://api.estated.com/property/v4?token=${ESTATED_KEY}&address=${encodeURIComponent(q)}`;
        const r = await fetch(url);
        if (!r.ok) throw new Error('Estated error ' + r.status);
        return r.json();
      } else {
        throw new Error('No provider configured');
      }
    });

    // Minimal normalization - map to common fields your frontend expects
    const normalized = {
      raw: data,
      ownerName: (data.owner && data.owner.name) || data.ownerName || null,
      ownerMailing: (data.owner && data.owner.mailing_address) || data.ownerMailing || null,
      parcelId: data.parcel_number || data.parcel || data.property && data.property.parcel_number || null,
      lastSaleDate: data.last_sale && data.last_sale.date || data.lastSaleDate || null,
      address: q
    };

    res.json({ success: true, provider: PROVIDER, data: normalized });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: err.message });
  }
});

// Endpoint: enrichment (phone / email) - requires Whitepages or similar
// POST /api/enrich  { "name":"John Doe","address":"5385 Westwood Drive Columbus GA" }
app.post('/api/enrich', async (req, res) => {
  try {
    const { name, address } = req.body;
    if (!name && !address) return res.status(400).json({ error: 'name or address required' });

    const cacheKey = `enrich:${name||''}:${address||''}`;
    const data = await cached(cacheKey, 60 * 60 * 24, async () => {
      if (!WHITEPAGES_KEY) throw new Error('Whitepages API key missing');
      // Example Whitepages Pro phone match endpoint pattern (you must consult doc)
      // This is a placeholder; follow Whitepages Pro docs for exact URL & params.
      const url = `https://proapi.whitepages.com/3.0/identity_check?api_key=${WHITEPAGES_KEY}&name=${encodeURIComponent(name||'')}&address=${encodeURIComponent(address||'')}`;
      const r = await fetch(url);
      if (!r.ok) throw new Error('Whitepages error ' + r.status);
      return r.json();
    });

    res.json({ success: true, data });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: err.message });
  }
});

// Health
app.get('/', (req, res) => res.send('contractor-data-proxy running'));

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server listening on ${PORT}`));

4) Security, caching, cost & production best practices

Never expose API keys to client-side code. Keep provider calls server-side in Replit. Use the Replit Secrets UI.

Cache aggressively. Property/owner data rarely changes minute-to-minute. Use in-memory cache for small scale and Redis for production. This reduces API calls and cost.

Rate limits & error handling. Respect provider rate limits. Add retries + exponential backoff and fail-safe responses to the contractor UI (e.g., “Try again in a few minutes”).

Logging & audit trail. Log who requested what and why (user id, timestamp, query) — save to a secure database and purge PII per your retention policy.

PII handling & compliance: If you plan to call/market/email owners, get explicit permissible purpose and follow TCPA & CAN-SPAM. If you plan to do background checks, tenant or employment decisions, use FCRA-compliant providers only. (Enterprise vendors like LexisNexis offer DaaS for regulated uses.)
