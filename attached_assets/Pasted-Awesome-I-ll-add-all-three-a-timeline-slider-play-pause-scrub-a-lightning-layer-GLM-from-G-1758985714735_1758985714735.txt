server/events.js

import { nanoid } from "nanoid";

const subscribers = new Set();   // SSE clients
const queue = [];                // in-memory ring buffer

export function publish(evt) {
  const event = { id: nanoid(), ts: Date.now(), ...evt };
  queue.push(event);
  // keep last 500
  if (queue.length > 500) queue.shift();
  const payload = `id: ${event.id}\nevent: ${event.type || "storm"}\ndata: ${JSON.stringify(event)}\n\n`;
  for (const res of subscribers) {
    try { res.write(payload); } catch {}
  }
}

export function sseHandler(req, res) {
  res.writeHead(200, {
    "Content-Type": "text/event-stream",
    "Cache-Control": "no-cache",
    Connection: "keep-alive",
    "Access-Control-Allow-Origin": "*"
  });
  res.write("\n");
  subscribers.add(res);
  // replay last 50 on connect
  for (const e of queue.slice(-50)) {
    res.write(`id: ${e.id}\nevent: ${e.type || "storm"}\ndata: ${JSON.stringify(e)}\n\n`);
  }
  req.on("close", () => subscribers.delete(res));
}

export function postEventHandler(req, res) {
  let body = "";
  req.on("data", (c) => body += c);
  req.on("end", () => {
    try {
      const evt = JSON.parse(body || "{}");
      if (!evt.type) evt.type = "storm";
      publish(evt);
      res.json?.({ ok: true }) || res.end(JSON.stringify({ ok: true }));
    } catch (e) {
      res.status(400).send("Bad JSON");
    }
  });
}

import { useEffect, useRef } from "react";
import L from "leaflet";

/**
 * TimeTileLayer: Leaflet tile layer with time parameter.
 * Props:
 *  - map            : Leaflet map instance
 *  - urlTemplate    : contains {time}, {z}/{x}/{y}
 *  - timestamps     : ISO[] ascending
 *  - playing        : bool (auto-advance)
 *  - index          : number (controlled current frame)
 *  - onNeedNext     : () => void   (called to request next index when playing)
 *  - opacity, zIndex, maxZoom
 *  - onFrame        : (iso, i) => void  (callback fired when frame applied)
 */
export default function TimeTileLayer({
  map, urlTemplate, timestamps,
  playing=false, index=0, onNeedNext,
  frameMs=800, opacity=0.85, zIndex=500, maxZoom=12,
  onFrame
}) {
  const layerRef = useRef(null);
  const timerRef = useRef(null);
  const lastIdxRef = useRef(-1);

  const buildUrl = (iso) => {
    const encodedTime = encodeURIComponent(iso);
    const direct = urlTemplate.replace("{time}", encodedTime);
    return `/proxy?url=${encodeURIComponent(direct)}`;
  };

  useEffect(() => {
    if (!map || !timestamps?.length) return;

    // draw whenever index changes
    if (index !== lastIdxRef.current) {
      const iso = timestamps[index];
      const url = buildUrl(iso);
      if (layerRef.current) map.removeLayer(layerRef.current);
      layerRef.current = L.tileLayer(url, { opacity, zIndex, maxZoom, crossOrigin: true });
      layerRef.current.addTo(map);
      lastIdxRef.current = index;
      onFrame && onFrame(iso, index);
    }

    // autoplay
    if (playing) {
      timerRef.current && clearTimeout(timerRef.current);
      timerRef.current = setTimeout(() => onNeedNext && onNeedNext(), frameMs);
    } else if (timerRef.current) {
      clearTimeout(timerRef.current);
      timerRef.current = null;
    }

    return () => {};
  }, [map, timestamps, index, playing, frameMs, opacity, zIndex, maxZoom]);

  // cleanup on unmount
  useEffect(() => () => {
    if (timerRef.current) clearTimeout(timerRef.current);
    if (layerRef.current && map) map.removeLayer(layerRef.current);
  }, [map]);

  return null;
}

2b) Generic GIBS static layer helper (for radar or any XYZ without time)

src/components/TileLayerURL.jsx

import { useEffect, useRef } from "react";
import L from "leaflet";

export default function TileLayerURL({ map, url, opacity=0.8, zIndex=600, maxZoom=12 }) {
  const layerRef = useRef(null);
  useEffect(() => {
    if (!map) return;
    const lyr = L.tileLayer(url, { opacity, zIndex, maxZoom, crossOrigin: true });
    layerRef.current = lyr;
    lyr.addTo(map);
    return () => { if (layerRef.current) map.removeLayer(layerRef.current); };
  }, [map, url, opacity, zIndex, maxZoom]);
  return null;
}

/** Helpers **/
function buildTimes({ hours=3, stepMin=10 }) {
  const arr = [];
  const end = Date.now();
  const steps = Math.ceil((hours*60)/stepMin);
  for (let i = steps-1; i >= 0; i--) {
    arr.push(new Date(end - i*stepMin*60*1000).toISOString());
  }
  return arr;
}

async function publishEvent(evt) {
  try {
    await fetch("/events", { method: "POST", body: JSON.stringify(evt) });
  } catch {}
}

  );
}

3) Storm Predictions module: consume the event stream (SSE)

Create src/modules/StormPredictions.jsx (starter)

import { useEffect, useState } from "react";

export default function StormPredictions() {
  const [events, setEvents] = useState([]);

  useEffect(() => {
    const src = new EventSource("/events/stream");
    const handler = (e) => {
      try { setEvents((prev) => [...prev.slice(-199), JSON.parse(e.data)]); } catch {}
    };
    src.addEventListener("storm", handler);
    src.addEventListener("frame_advanced", handler);
    src.addEventListener("lightning_frame", handler);
    src.addEventListener("heartbeat", handler);
    return () => src.close();
  }, []);

  // Here you can hook AI logic: windowed analysis, thresholds, etc.
  // Example heuristic: every lightning_frame => if frequency spikes, raise "very_urgent" priority.

  return (
    <div style={{ padding: 16 }}>
      <h2>Storm Predictions – Live Events</h2>
      <p>Feed from Weather Center (frames, lightning) to drive AI scoring & lead prioritization.</p>
      <div style={{ maxHeight: "70vh", overflow: "auto", border: "1px solid #ddd", borderRadius: 8, padding: 8 }}>
        {events.slice().reverse().map((e, i) => (
          <pre key={i} style={{ margin: 0, padding: 8, borderBottom: "1px dashed #eee" }}>
            {JSON.stringify(e, null, 2)}
          </pre>
        ))}
      </div>
    </div>
  );
}
