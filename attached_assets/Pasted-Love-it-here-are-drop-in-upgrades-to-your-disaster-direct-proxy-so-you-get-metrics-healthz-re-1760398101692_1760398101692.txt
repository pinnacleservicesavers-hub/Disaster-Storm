What you’re adding (in plain English)

/metrics (Prometheus): live counts, error rates, latency, cache hits → real dashboards & alerts.

/healthz: “I’m alive” (uptime, version, git SHA) → k8s/container liveness checks.

/readyz: “I can serve traffic” (Ambee check + circuit breaker state) → safe rollouts.

/config: one JSON for UI (palette, thresholds, tile defaults, flags) → no redeploy for small changes.

HMAC tile signing: short-lived signatures on tile/legend URLs → blocks hotlinking/scraping.

import crypto from "crypto";

function hmacSign({ secret, algo = "sha256" }, msg) {
  return crypto.createHmac(algo, secret).update(msg).digest("hex");
}
function unixNow() { return Math.floor(Date.now() / 1000); }

B) Circuit breaker (near other helpers, inside createDisasterDirectProxy)
const cb = {
  failures: 0,
  openUntil: 0,
  threshold: options?.circuit?.threshold ?? Number(process.env.CIRCUIT_THRESHOLD ?? 8),
  cooldownMs: options?.circuit?.cooldownMs ?? Number(process.env.CIRCUIT_COOLDOWN_MS ?? 60_000),
  state() { return (Date.now() < this.openUntil) ? "open" : "closed"; },
  markSuccess() { this.failures = 0; },
  markFailure() {
    this.failures++;
    if (this.failures >= this.threshold) {
      this.openUntil = Date.now() + this.cooldownMs;
      this.failures = 0; // reset counter after opening
    }
  }
};


Wrap every Ambee fetch (ambeeGET) with breaker + retries:

async function ambeeGET(path, q = {}) {
  const qp = new URLSearchParams(q);
  const url = `${AMBEE}/${path}?${qp.toString()}`;
  const hit = getCache(url); if (hit) return hit;

  // Circuit breaker: if open, try to serve stale cache; otherwise throw
  if (cb.state() === "open") {
    const stale = getCache(url); // (same as hit here; simple breaker still returns miss)
    if (stale) return stale;
    const err = new Error("Circuit open");
    err.status = 503;
    throw err;
  }

  // Retries with backoff
  const maxAttempts = 3;
  let lastErr;
  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    try {
      const r = await fetchFn(url, { headers: { "x-api-key": ambeeApiKey, "Content-Type": "application/json" } });
      const data = await r.json().catch(() => ({}));

      if (r.status === 429 || r.status >= 500) {
        lastErr = new Error(`Upstream ${r.status}`);
        lastErr.status = r.status; lastErr.data = data;
        const retryAfter = Number(r.headers.get("retry-after")) || 0;
        const backoff = retryAfter ? retryAfter * 1000 : (200 * attempt + Math.random() * 300);
        await new Promise(res => setTimeout(res, backoff));
        continue;
      }
      if (!r.ok) { const e = new Error("Upstream error"); e.status = r.status; e.data = data; throw e; }

      setCache(url, data);
      cb.markSuccess();
      return data;
    } catch (e) {
      lastErr = e;
    }
  }
  cb.markFailure();
  throw lastErr;
}

F) HMAC-signed tiles & legend

1) Verify on incoming requests (inside the tiles and legend routes, before generating the image):

function verifyHmacIfEnabled(kind, req) {
  const hmacCfg = options?.hmac;
  if (!hmacCfg || !hmacCfg.secret) return; // disabled
  const enabledFor = hmacCfg.enabledFor || ["tiles","legend"];
  if (!enabledFor.includes(kind)) return;

  const sigParam = hmacCfg.param || "_sig";
  const expParam = hmacCfg.ttlParam || "_exp";
  const algo = hmacCfg.algo || "sha256";
  const skew = Number(hmacCfg.clockSkewSec || 60);

  const url = new URL(req.protocol + "://" + req.get("host") + req.originalUrl);
  const sig = url.searchParams.get(sigParam);
  const exp = Number(url.searchParams.get(expParam) || 0);
  if (!sig || !exp) return req.res.status(401).json({ error: "signature required" });

  // Expiration
  const now = unixNow();
  if (now > exp + skew) return req.res.status(401).json({ error: "signature expired" });

  // Canonical string: method + path + sorted query (excluding signature)
  const qp = [];
  for (const [k, v] of url.searchParams.entries()) {
    if (k === sigParam) continue;
    qp.push([k, v]);
  }
  qp.sort((a,b) => a[0] < b[0] ? -1 : a[0] > b[0] ? 1 : (a[1] < b[1] ? -1 : 1));
  const canonical = `${req.method}\n${url.pathname}\n${qp.map(([k,v]) => `${k}=${v}`).join("&")}`;

  const expect = hmacSign({ secret: hmacCfg.secret, algo }, canonical);
  if (!crypto.timingSafeEqual(Buffer.from(expect, "hex"), Buffer.from(sig, "hex"))) {
    return req.res.status(401).json({ error: "bad signature" });
  }
}


Call it at the top of each route:

// tiles route handler
verifyHmacIfEnabled("tiles", req);
if (res.headersSent) return;

// legend route handler

2) Export a tiny signer (so your server can mint URLs for your frontend):

At bottom of index.js, export:

export function signTileUrl(baseUrl, { secret, algo = "sha256", param = "_sig", ttlParam = "_exp", expiresInSec = 300 }) {
  const url = new URL(baseUrl, "http://x"); // base to parse even relative paths
  const exp = unixNow() + expiresInSec;
  url.searchParams.set(ttlParam, String(exp));
  const qp = [];
  for (const [k,v] of url.searchParams.entries()) qp.push([k,v]);
  qp.sort((a,b) => a[0] < b[0] ? -1 : a[0] > b[0] ? 1 : (a[1] < b[1] ? -1 : 1));
  const canonical = `GET\n${url.pathname}\n${qp.map(([k,v]) => `${k}=${v}`).join("&")}`;
  const sig = hmacSign({ secret, algo }, canonical);
  url.searchParams.set(param, sig);
  return url.pathname + "?" + url.searchParams.toString(); // relative path for same-origin calls
}


You can use the same function for legend URLs.

4) Why this helps (quick recap)

/metrics → real visibility (alert on 429/5xx, latency, breaker).

/healthz /readyz → stable rollouts & auto-healing.

/config → change palettes/thresholds without redeploys.

HMAC → only your clients (that get signed URLs from your server) can fetch tiles/legend; stops scraping & hotlinking.

Circuit breaker & retries → survive provider blips gracefully.
