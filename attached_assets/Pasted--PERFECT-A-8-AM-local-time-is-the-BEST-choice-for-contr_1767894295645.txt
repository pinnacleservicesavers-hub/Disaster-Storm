/lead-engine-node/scheduler/campaignScheduler.js
import cron from "node-cron";
import fetch from "node-fetch";
import { DateTime } from "luxon";

const TZ = "America/Chicago";

// Helper: logs with local time
function logLocal(message) {
  const now = DateTime.now().setZone(TZ).toFormat("yyyy-LL-dd HH:mm:ss ZZZZ");
  console.log(`[${now}] ${message}`);
}

export function startCampaignScheduler() {
  /**
   * ✅ Daily campaigns at 8:05 AM America/Chicago
   * Cron: "5 8 * * *"
   */
  cron.schedule("5 8 * * *", async () => {
    logLocal("🟢 Daily Campaign Scheduler triggered (8:05 AM CT).");
    await runScheduledCampaigns("daily");
  }, { timezone: TZ });

  /**
   * ✅ Weekly campaigns Monday at 8:10 AM America/Chicago
   * Cron: "10 8 * * 1" (Monday)
   */
  cron.schedule("10 8 * * 1", async () => {
    logLocal("🔵 Weekly Campaign Scheduler triggered (Monday 8:10 AM CT).");
    await runScheduledCampaigns("weekly");
  }, { timezone: TZ });

  /**
   * ✅ Bonus WOW: Storm Trigger Check (every 2 hours)
   * Runs storm cleanup campaign when storm risk is high.
   */
  cron.schedule("0 */2 * * *", async () => {
    logLocal("⚡ Storm Trigger Check running (every 2 hours).");
    await runStormTriggerCampaigns();
  }, { timezone: TZ });

  logLocal("✅ Campaign Scheduler started (America/Chicago).");
}

async function runScheduledCampaigns(scheduleType) {
  try {
    const res = await fetch(`${process.env.NEXT_APP_URL}/api/leadvault/campaigns/scheduled/run`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "X-VAULT-KEY": process.env.VAULT_SERVICE_KEY
      },
      body: JSON.stringify({ scheduleType })
    });

    const data = await res.json();
    logLocal(`✅ ${scheduleType.toUpperCase()} campaigns completed: ${JSON.stringify(data.results || [])}`);
  } catch (e) {
    logLocal(`❌ Campaign Scheduler failed: ${String(e)}`);
  }
}

async function runStormTriggerCampaigns() {
  try {
    const res = await fetch(`${process.env.NEXT_APP_URL}/api/leadvault/campaigns/scheduled/storm-trigger`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "X-VAULT-KEY": process.env.VAULT_SERVICE_KEY
      }
    });

    const data = await res.json();
    logLocal(`⚡ Storm Trigger result: ${JSON.stringify(data)}`);
  } catch (e) {
    logLocal(`❌ Storm Trigger Scheduler failed: ${String(e)}`);
  }
}

/app-next/app/leadvault/campaigns/page.jsx
"use client";
import { useEffect, useState } from "react";

const DEFAULT_TARGETS = [
  "property management",
  "hoa",
  "apartment complex",
  "commercial property",
  "construction company",
  "general contractor",
  "restoration company",
  "insurance agency"
];

export default function CampaignsPage() {
  const [campaigns, setCampaigns] = useState([]);
  const [creating, setCreating] = useState(false);
  const [newCampaign, setNewCampaign] = useState({
    name: "Property Manager Contracts",
    target_location: "Cataula, GA",
    radius: 25,
    lead_targets: ["property management", "hoa", "apartment complex"],
    schedule_type: "weekly",
    schedule_hour: 8,
    min_score: 25,
    max_leads_per_run: 30,
    generate_outreach_for: "hot"
  });

  async function loadCampaigns() {
    const res = await fetch("/api/leadvault/campaigns/list");
    const data = await res.json();
    setCampaigns(data.campaigns || []);
  }

  useEffect(() => {
    loadCampaigns();
  }, []);

  async function createCampaign() {
    setCreating(true);
    const res = await fetch("/api/leadvault/campaigns/create", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(newCampaign)
    });
    await res.json();
    setCreating(false);
    await loadCampaigns();
  }

  async function runNow(campaign) {
    const res = await fetch("/api/leadvault/campaigns/run", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ campaign })
    });
    const data = await res.json();
    alert(`Campaign Run Complete: Found ${data.results?.runLog?.leads_found || 0} leads`);
  }

  function toggleTarget(target) {
    setNewCampaign(prev => {
      const exists = prev.lead_targets.includes(target);
      return {
        ...prev,
        lead_targets: exists
          ? prev.lead_targets.filter(t => t !== target)
          : [...prev.lead_targets, target]
      };
    });
  }

  return (
    <div className="p-6 space-y-6">
      <div>
        <h1 className="text-3xl font-bold">Campaign Builder</h1>
        <p className="text-gray-500">
          Set campaigns to run automatically at <b>8 AM</b> and fill your pipeline every week.
        </p>
      </div>

      <div className="rounded-2xl border p-4 shadow-sm space-y-3">
        <h2 className="text-lg font-semibold">Create New Campaign</h2>

        <div className="grid md:grid-cols-2 gap-3">
          <input
            className="border rounded-xl p-2"
            value={newCampaign.name}
            onChange={e => setNewCampaign({ ...newCampaign, name: e.target.value })}
            placeholder="Campaign Name"
          />
          <input
            className="border rounded-xl p-2"
            value={newCampaign.target_location}
            onChange={e => setNewCampaign({ ...newCampaign, target_location: e.target.value })}
            placeholder="Target Location"
          />
          <input
            className="border rounded-xl p-2"
            type="number"
            value={newCampaign.radius}
            onChange={e => setNewCampaign({ ...newCampaign, radius: Number(e.target.value) })}
            placeholder="Radius"
          />
          <select
            className="border rounded-xl p-2"
            value={newCampaign.schedule_type}
            onChange={e => setNewCampaign({ ...newCampaign, schedule_type: e.target.value })}
          >
            <option value="daily">Daily</option>
            <option value="weekly">Weekly</option>
          </select>
        </div>

        <div className="pt-2">
          <p className="text-sm font-semibold">Lead Targets</p>
          <div className="flex flex-wrap gap-2 mt-2">
            {DEFAULT_TARGETS.map((t) => (
              <button
                key={t}
                onClick={() => toggleTarget(t)}
                className={`text-xs px-3 py-1 rounded-full border ${
                  newCampaign.lead_targets.includes(t) ? "bg-black text-white" : "bg-white"
                }`}
              >
                {t}
              </button>
            ))}
          </div>
        </div>

        <button
          onClick={createCampaign}
          className="rounded-xl bg-black text-white px-4 py-2"
          disabled={creating}
        >
          {creating ? "Creating..." : "Create Campaign"}
        </button>
      </div>

      <div className="rounded-2xl border p-4 shadow-sm space-y-4">
        <h2 className="text-lg font-semibold">Your Campaigns</h2>

        <div className="grid gap-3">
          {campaigns.map((c) => (
            <div key={c.id} className="rounded-xl bg-gray-50 p-3 flex flex-col md:flex-row md:items-center md:justify-between gap-3">
              <div>
                <p className="font-semibold">{c.name}</p>
                <p className="text-sm text-gray-600">
                  {c.schedule_type.toUpperCase()} • {c.target_location} • {c.radius} miles
                </p>
                <p className="text-xs text-gray-500 mt-1">
                  Targets: {(c.lead_targets || []).slice(0, 5).join(", ")}
                </p>
              </div>

              <div className="flex gap-3">
                <button className="text-sm underline" onClick={() => runNow(c)}>
                  Run Now
                </button>
                <button className="text-sm underline">
                  Results
                </button>
              </div>
            </div>
          ))}
        </div>

        {!campaigns.length && (
          <p className="text-sm text-gray-500">No campaigns yet. Create one above.</p>
        )}
      </div>
    </div>
  );
}
