6) Plug-in upgrade #1 — Local caller-ID mapping

Replace the placeholder in services/twilio.js with a simple mapping file.
Create area-codes.json (copy area-codes.example.json and add your numbers):

{
  "813": { "voiceFrom": "+18135550100" },
  "904": { "voiceFrom": "+19045550100" },
  "305": { "voiceFrom": "+13055550100" }
}


Then update services/twilio.js:

import fs from 'fs';

let AREA_MAP = {};
try {
  AREA_MAP = JSON.parse(fs.readFileSync('./area-codes.json','utf8'));
} catch (e) {
  try { AREA_MAP = JSON.parse(fs.readFileSync('./area-codes.example.json','utf8')); } catch {}
}

export function pickLocalNumberFor(areaCode) {
  const entry = AREA_MAP[areaCode];
  if (entry?.voiceFrom) return { fromNumber: entry.voiceFrom, messagingServiceSid: process.env.TWILIO_MESSAGING_SERVICE_SID || null };
  // fallback to messaging service only (Geo-Match for SMS)
  return { fromNumber: null, messagingServiceSid: process.env.TWILIO_MESSAGING_SERVICE_SID || null };
}

7) Plug-in upgrade #2 — Tier 1 / Tier 2 smart routing

Drop-in logic for routeLead() in services/lead.js that:

Finds contractors by service category

Filters by license/insurance, proximity (you can store ZIP/lat-lng), and availability

Ranks by performance score + response-time

Issues round 1 offers (Tier 1). If no accept, issue round 2 (Tier 2)

Example (pseudo-implementation you can paste and adapt):

export async function routeLead(leadId) {
  // Get services needing assignment
  const services = await query(`
    select ls.id, ls.category, l.address, l.priority
    from lead_services ls
    join leads l on l.id = ls.lead_id
    where ls.lead_id = $1 and ls.assigned_contractor_id is null and ls.needed = true
  `, [leadId]);

  for (const svc of services) {
    // Find candidate contractors for this category
    const candidates = await query(`
      select c.*
      from contractors c
      where $1 = any(c.trades)
        and c.active = true
        and c.license_verified = true
        and c.insurance_verified = true
      order by c.performance_score desc, c.response_time_ms_avg asc
      limit 20
    `, [svc.category]);

    // Split into Tier 1 (top 5) and Tier 2 (next 15)
    const tier1 = candidates.slice(0,5);
    const tier2 = candidates.slice(5);

    // Offer to Tier 1 now
    for (const c of tier1) {
      await query(`
        insert into assignments (lead_service_id, contractor_id, state, round)
        values ($1, $2, 'offered', 1)
      `, [svc.id, c.id]);
      // TODO: push notification/SMS to contractor app or number
    }

    // You can schedule a job to escalate to Tier 2 in ~10–15 minutes
    // (e.g., via cron/queue) if none accepts.
  }
  return true;
}
