export function treeServiceOutreachPrompt({ contractor, lead, context }) {
  return `
You are a high-converting B2B sales strategist for a TREE SERVICE contractor.
Your job is to generate a complete outreach pack that gets responses and booked estimates.

CONTRACTOR PROFILE:
- Company/Name: ${contractor.name}
- Service Type: Tree Service (removal, trimming, storm cleanup, hazard trees, lot clearing)
- Tone: confident, professional, short, no fluff, no hype
- Goal: schedule a site visit / quote this week

LEAD:
${JSON.stringify(lead, null, 2)}

CONTEXT:
- Location targeted: ${context.location}
- Radius: ${context.radius} miles
- Why this lead is good: ${context.why}

REQUIREMENTS:
- Output MUST be valid JSON ONLY (no markdown).
- SMS pitch max 300 characters.
- Email subject must be strong and specific.
- Phone script must include a qualifying question and a close.
- Follow-up sequence must include 4 messages (day 2, day 5, day 10, day 20 re-engage).
- Offer stack must be: quick_win, standard, premium.
- Include objection handlers: price, already_have_vendor, not_now.
- Include personalization_notes (mention something from their category or signals).
- The offer should be tailored to the lead’s category:
  - Property Manager/HOA/Apartment → recurring maintenance contract + hazard inspections
  - Construction/GC → clearing & debris hauling + schedule coordination
  - Commercial lot → monthly trimming + cleanup compliance
  - Restoration/Insurance → storm response partner pitch

Return JSON with this schema exactly:

✅ STEP 4 — Next.js API Route (Responses API — Recommended)

This is the real production route that your UI will call.

app-next/app/api/leadvault/outreach/route.js
import OpenAI from "openai";
import { getUser } from "@/lib/auth";
import { contractorOnly } from "@/lib/contractorOnly";
import { treeServiceOutreachPrompt } from "@/lib/prompts/treeServiceOutreach";

const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export async function POST(req) {
  const user = await getUser();
  try {
    contractorOnly(user);
  } catch (e) {
    return Response.json({ error: e.message }, { status: 403 });
  }

  const { lead, context } = await req.json();

  // 🔒 IMPORTANT: Basic validation to prevent prompt injection via lead fields
  const safeLead = {
    company_name: lead.company_name || "",
    category: lead.category || "",
    address: lead.address || "",
    phone: lead.phone || "",
    website: lead.website || "",
    signal_tags: lead.signal_tags || [],
    score_label: lead.score_label || "",
    competition_level: lead.competition_level || "",
  };

  const prompt = treeServiceOutreachPrompt({
    contractor: { name: user.name },
    lead: safeLead,
    context: {
      location: context?.location || "Unknown",
      radius: context?.radius || 0,
      why: context?.why || "High-fit tree service lead",
    },
  });

  try {
    // ✅ Responses API (recommended for new projects)
    const resp = await client.responses.create({
      model: "gpt-4o-mini",
      input: prompt,
      max_output_tokens: 900
    });

    // The Responses API returns text in output_text
    const text = resp.output_text?.trim() || "{}";
    const outreachPack = JSON.parse(text);

    return Response.json({ outreachPack });
  } catch (err) {
    // Fallback: return readable error
    return Response.json(
      { error: "Failed to generate outreach pack.", detail: String(err) },
      { status: 500 }
    );
  }
}

import { useState } from "react";

export default function LeadCard({ lead }) {
  const [loading, setLoading] = useState(false);
  const [pack, setPack] = useState(null);

  async function generatePack() {
    setLoading(true);
    setPack(null);

    const res = await fetch("/api/leadvault/outreach", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        lead,
        context: {
          location: "Cataula, GA",
          radius: 25,
          why: (lead.signal_tags || []).slice(0, 3).join(", ") || "High-fit tree service lead"
        }
      }),
    });

    const data = await res.json();
    setPack(data.outreachPack || null);
    setLoading(false);
  }

  return (
    <div className="rounded-2xl border p-4 shadow-sm space-y-2">
      {/* existing card header stuff here */}

      <div className="flex flex-wrap gap-2 pt-2">
        {lead.phone && <a className="text-sm underline" href={`tel:${lead.phone}`}>Call</a>}
        {lead.website && <a className="text-sm underline" href={lead.website} target="_blank">Website</a>}

        <button className="text-sm underline" onClick={generatePack}>
          {loading ? "Generating..." : "Generate Outreach Pack"}
        </button>
      </div>

      {pack && (
        <div className="mt-3 rounded-xl bg-gray-50 p-3 space-y-2">
          <div>
            <p className="text-xs font-semibold">SMS</p>
            <p className="text-sm text-gray-700">{pack.sms_pitch}</p>
          </div>
          <div>
            <p className="text-xs font-semibold">Email Subject</p>
            <p className="text-sm text-gray-700">{pack.email_subject}</p>
          </div>
          <div>
            <p className="text-xs font-semibold">Call To Action</p>
            <p className="text-sm text-gray-700">{pack.call_to_action}</p>
          </div>
        </div>
      )}
    </div>
  );
}

✅ OPTIONAL: Node Microservice Outreach Endpoint (Better Security)

If you want OpenAI only in Node microservice:

Add to Node .env
OPENAI_API_KEY=your_openai_key_here

Add a new Node route:

lead-engine-node/routes/outreach.js

import express from "express";
import { authKey } from "../middleware/authKey.js";
import OpenAI from "openai";

const router = express.Router();
const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

router.post("/generate", authKey, async (req, res) => {
  const { contractor, lead, context } = req.body;

  const prompt = `
Return JSON ONLY. Create a tree-service outreach pack for this lead.
Contractor: ${JSON.stringify(contractor)}
Lead: ${JSON.stringify(lead)}
Context: ${JSON.stringify(context)}
JSON schema: {
  "sms_pitch":"","email_subject":"","email_body":"","phone_script":"",
  "followups":["","","",""],
  "offer_stack":{"quick_win":"","standard":"","premium":""},
  "objection_handlers":{"price":"","already_have_vendor":"","not_now":""},
  "call_to_action":"","best_time_to_contact":"","personalization_notes":""
}`;

  try {
    const resp = await client.responses.create({
      model: "gpt-4o-mini",
      input: prompt,
      max_output_tokens: 900
    });

    const outreachPack = JSON.parse(resp.output_text.trim());
    res.json({ outreachPack });
  } catch (e) {
    res.status(500).json({ error: "Outreach generation failed", detail: String(e) });
  }
});

export default router;
