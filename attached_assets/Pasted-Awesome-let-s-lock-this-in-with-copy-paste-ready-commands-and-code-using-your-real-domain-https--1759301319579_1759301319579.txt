4) Add the Google auth strategy

Create server/auth/google.ts:

// server/auth/google.ts
import passport from "passport";
import { Strategy as GoogleStrategy } from "passport-google-oauth20";

passport.serializeUser((user: any, done) => done(null, user));
passport.deserializeUser((obj: any, done) => done(null, obj));

const CALLBACK_URL = `${process.env.BASE_URL}/auth/callback`;

passport.use(new GoogleStrategy(
  {
    clientID: process.env.GOOGLE_CLIENT_ID!,
    clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
    callbackURL: CALLBACK_URL, // absolute URL avoids http/https mismatch
  },
  (_accessToken, _refreshToken, profile, done) => {
    const user = {
      id: profile.id,
      name: profile.displayName,
      email: profile.emails?.[0]?.value,
      photo: profile.photos?.[0]?.value,
    };
    return done(null, user);
  }
));

export default passport;

5) Wire it into your server

Open server/index.ts and make sure these blocks exist (adjust imports to your file paths). Place them after you create app = express() and before your routes:

import session from "express-session";
import passport from "./auth/google.js"; // path must match the file you created

// Trust Replit proxy so req.protocol is 'https'
app.set("trust proxy", 1);

const isProd = /^https:\/\//.test(process.env.BASE_URL ?? "");

// Sessions (cookie works in both localhost and Replit)
app.use(session({
  secret: process.env.SESSION_SECRET!,
  resave: false,
  saveUninitialized: false,
  cookie: {
    secure: isProd,         // true on Replit HTTPS
    sameSite: "lax"
  }
}));

app.use(passport.initialize());
app.use(passport.session());

// --- Auth routes ---
app.get("/auth/google",
  passport.authenticate("google", { scope: ["profile", "email"] })
);

app.get("/auth/callback",
  passport.authenticate("google", { failureRedirect: "/login?error=oauth" }),
  (_req, res) => res.redirect("/")
);

// --- Health + WhoAmI (handy for Replit route verification) ---
app.get("/health", (_req, res) =>
  res.status(200).json({ ok: true, ts: new Date().toISOString() })
);

app.get("/api/me", (req: any, res) =>
  res.json({ user: req.user ?? null })
);


If you previously added any middleware that forces HTTPS redirects, comment it out for now so /health and auth won’t loop while we verify.
