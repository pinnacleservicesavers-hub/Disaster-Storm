B) Add these helpers to your <script> (below your other helpers):

function showToast(msg) {
  const t = document.getElementById("toast");
  t.textContent = msg;
  t.style.display = "block";
  setTimeout(() => { t.style.display = "none"; }, 1600);
}

function buildIdText(payload) {
  if (!payload) return "";
  const z = payload.zones || {};
  const grid = z.grid || {};
  const office = z.office || "";
  const fz = (z.forecastZone || "").split("/").pop() || "";
  const cz = (z.county || "").split("/").pop() || "";
  const fwz = (z.fireWeatherZone || "").split("/").pop() || "";
  const gridStr = (grid.id && grid.x != null && grid.y != null) ? `${grid.id}/${grid.x},${grid.y}` : "";
  const loc = payload.locationDisplay || payload.location || "";

  return [
    loc ? `Location: ${loc}` : null,
    office ? `Office: ${office}` : null,
    fz ? `Forecast Zone: ${fz}` : null,
    cz ? `County Zone: ${cz}` : null,
    fwz ? `Fire Weather Zone: ${fwz}` : null,
    gridStr ? `Grid: ${gridStr}` : null
  ].filter(Boolean).join("\n");
}

// Attach once after DOM loaded
document.addEventListener("click", async (e) => {
  if (e.target.id !== "copyIdsBtn") return;
  const text = buildIdText(lastPayload);
  if (!text) { showToast("No IDs to copy yet"); return; }
  try {
    await navigator.clipboard.writeText(text);
    showToast("Copied zone & office IDs");
  } catch {
    // Fallback: prompt (rarely needed)
    window.prompt("Copy IDs:", text);
  }
});
