B) Session cookies (work in both dev and Replit)

import session from "express-session";

const isProd = /^https:\/\//.test(process.env.BASE_URL ?? "");

app.use(session({
  secret: process.env.SESSION_SECRET!,
  resave: false,
  saveUninitialized: false,
  cookie: {
    secure: isProd,       // Replit uses https → true
    sameSite: isProd ? "lax" : "lax"
  }
}));

4) Make the Google callback absolute (prevents http/https confusion)

In server/auth/google.ts (or wherever you configure Passport):

import passport from "passport";
import { Strategy as GoogleStrategy } from "passport-google-oauth20";

const CALLBACK_URL = `${process.env.BASE_URL}/auth/callback`;

passport.use(new GoogleStrategy(
  {
    clientID: process.env.GOOGLE_CLIENT_ID!,
    clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
    callbackURL: CALLBACK_URL,  // absolute URL
  },
  (_at, _rt, profile, done) => {
    const user = {
      id: profile.id,
      name: profile.displayName,
      email: profile.emails?.[0]?.value,
      photo: profile.photos?.[0]?.value,
    };
    return done(null, user);
  }
));

// routes
app.get("/auth/google", passport.authenticate("google", { scope: ["profile", "email"] }));
app.get("/auth/callback",
  passport.authenticate("google", { failureRedirect: "/login?error=oauth" }),
  (_req, res) => res.redirect("/")
);
