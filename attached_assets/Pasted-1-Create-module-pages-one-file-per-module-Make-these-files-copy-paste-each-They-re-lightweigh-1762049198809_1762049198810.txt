2) Lazy-load the pages in your router

Edit client/src/App.tsx to import pages with React.lazy and wrap routes in a <Suspense> fallback. This keeps initial load tiny and plays nicely with your animations.

import React, { Suspense, lazy } from "react";
import { Routes, Route, Navigate } from "react-router-dom";
import ModuleGallery from "./StormOpsProHub"; // path to your gallery

// Lazy imports
const WeatherCenter      = lazy(() => import("./modules/WeatherCenter"));
const StormPredictions   = lazy(() => import("./modules/StormPredictions"));
const TrafficCamWatcher  = lazy(() => import("./modules/TrafficCamWatcher"));
const DroneOperations    = lazy(() => import("./modules/DroneOperations"));
const AIDamageDetection  = lazy(() => import("./modules/AIDamageDetection"));
const XrayReality        = lazy(() => import("./modules/XrayReality"));

// Optional: shared loader
function Loader() {
  return (
    <div className="w-full py-16 text-center text-white/70">
      <div className="animate-pulse inline-flex items-center gap-3 text-lg">
        <span className="w-3 h-3 rounded-full bg-[color:var(--neon-blue,#00c2ff)]"></span>
        Loading…
      </div>
    </div>
  );
}

// Route map used by the gallery buttons
const galleryRoutes = {
  "weather-center":        { launch: "/ops/weather",  preview: "/ops/weather/preview",  docs: "/docs/weather" },
  "storm-predictions":     { launch: "/ops/predict",  preview: "/ops/predict/preview",  docs: "/docs/predict" },
  "traffic-cam-watcher":   { launch: "/ops/traffic",  preview: "/ops/traffic/preview",  docs: "/docs/traffic" },
  "drone-operations":      { launch: "/ops/drones",   preview: "/ops/drones/preview",   docs: "/docs/drones" },
  "ai-damage-detection":   { launch: "/ops/ai",       preview: "/ops/ai/preview",       docs: "/docs/ai" },
  "xray-reality":          { launch: "/ops/xray",     preview: "/ops/xray/preview",     docs: "/docs/xray" },
};

export default function App() {
  return (
    <>
      {/* Neon cinematic gallery */}
      <ModuleGallery routes={galleryRoutes} />

      {/* Lazy routes */}
      <Suspense fallback={<Loader />}>
        <Routes>
          <Route path="/" element={<Navigate to="/ops/weather" replace />} />

          <Route path="/ops/weather" element={<WeatherCenter />} />
          <Route path="/ops/predict" element={<StormPredictions />} />
          <Route path="/ops/traffic" element={<TrafficCamWatcher />} />
          <Route path="/ops/drones" element={<DroneOperations />} />
          <Route path="/ops/ai" element={<AIDamageDetection />} />
          <Route path="/ops/xray" element={<XrayReality />} />

          {/* Preview placeholders (optional) */}
          <Route path="/ops/weather/preview" element={<WeatherCenter />} />
          <Route path="/ops/predict/preview" element={<StormPredictions />} />
          <Route path="/ops/traffic/preview" element={<TrafficCamWatcher />} />
          <Route path="/ops/drones/preview" element={<DroneOperations />} />
          <Route path="/ops/ai/preview" element={<AIDamageDetection />} />
          <Route path="/ops/xray/preview" element={<XrayReality />} />

          {/* Docs placeholders (optional) */}
          <Route path="/docs/weather" element={<WeatherCenter />} />
          <Route path="/docs/predict" element={<StormPredictions />} />
          <Route path="/docs/traffic" element={<TrafficCamWatcher />} />
          <Route path="/docs/drones" element={<DroneOperations />} />
          <Route path="/docs/ai" element={<AIDamageDetection />} />
          <Route path="/docs/xray" element={<XrayReality />} />

          <Route path="*" element={<Navigate to="/ops/weather" replace />} />
        </Routes>
      </Suspense>
    </>
  );
}
