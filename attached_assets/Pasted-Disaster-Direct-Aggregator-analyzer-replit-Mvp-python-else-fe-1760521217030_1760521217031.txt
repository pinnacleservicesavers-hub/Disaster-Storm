def alerts_get() -> Optional[Tuple[str,float]]:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("CREATE TABLE IF NOT EXISTS alerts (id INTEGER PRIMARY KEY, url TEXT, threshold REAL)")
    cur.execute("SELECT url, threshold FROM alerts WHERE id=1")
    row = cur.fetchone()
    con.close()
    if row:
        return row[0], float(row[1])
    return None




def alerts_set(url: str, threshold: float) -> None:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("CREATE TABLE IF NOT EXISTS alerts (id INTEGER PRIMARY KEY, url TEXT, threshold REAL)")
    cur.execute("REPLACE INTO alerts (id, url, threshold) VALUES (1, ?, ?)", (url, threshold))
    con.commit()
    con.close()




@app.post("/alerts")
async def save_alerts(cfg: Dict[str, Any] = Body(...)):
    url = cfg.get("url")
    threshold = float(cfg.get("threshold", 0))
    if not url:
        raise HTTPException(status_code=400, detail="url is required")
    alerts_set(url, threshold)
    return {"ok": True, "url": url, "threshold": threshold}




@app.get("/alerts")
async def get_alerts():
    cfg = alerts_get()
    if not cfg:
        return {"configured": False}
    return {"configured": True, "url": cfg[0], "threshold": cfg[1]}




async def maybe_fire_webhook(summary: Dict[str, Any]):
    cfg = alerts_get()
    if not cfg:
        return
    url, threshold = cfg
    try:
        if float(summary.get("risk_score", 0)) >= float(threshold):
            async with httpx.AsyncClient(timeout=10.0) as client:
                await client.post(url, json=summary)
    except Exception as e:
        print("webhook error:", e)
