2) Create a test route that proxies to xAI
(Adjust the URL/model/body to match the xAI endpoint you’re using.)

import express from "express";
import fetch from "node-fetch";

const app = express();
app.use(express.json({ limit: "10mb" }));

app.post("/api/grok", async (req, res) => {
  try {
    const { prompt } = req.body;

    const r = await fetch("https://api.x.ai/v1/chat/completions", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${process.env.XAI_API_KEY}`,
      },
      body: JSON.stringify({
        model: "grok-beta",           // or the exact model name you enabled
        messages: [{ role: "user", content: prompt || "Say hello" }],
        temperature: 0.3
      })
    });

    if (!r.ok) {
      const text = await r.text();
      return res.status(r.status).json({ error: text });
    }
    const data = await r.json();
    res.json(data);
  } catch (err) {
    res.status(500).json({ error: err?.message || "Unknown error" });
  }
});

export default app;

async function askGrok(prompt) {

    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ prompt })
  });
  const data = await r.json();

  return data?.choices?.[0]?.message?.content ?? JSON.stringify(data);

5) Common gotchas
Restart after adding secrets so process.env.XAI_API_KEY is loaded.

Never commit the key to source control; keep it only in Replit Secrets.

If you run locally, create a .env with XAI_API_KEY=... and npm i dotenv, then:

import 'dotenv/config';
Make sure your model name and endpoint URL match what your xAI Console shows for your access.

If you see 401/403, double-check: key copied correctly, permissions = All, and the account has billing/creditsenabled.
