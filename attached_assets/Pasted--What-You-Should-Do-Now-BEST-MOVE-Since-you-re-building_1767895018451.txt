Here’s the Express version (since your log shows rest-express):

import { Pool } from "pg";

export function registerDbTestRoute(app: any) {
  app.get("/api/db-test", async (req, res) => {
    if (!process.env.DATABASE_URL) {
      return res.json({ connected: false, error: "DATABASE_URL missing" });
    }

    const pool = new Pool({ connectionString: process.env.DATABASE_URL });

    try {
      const result = await pool.query("SELECT NOW()");
      res.json({ connected: true, now: result.rows[0].now });
    } catch (e) {
      res.status(500).json({ connected: false, error: String(e) });
    }
  });
}
