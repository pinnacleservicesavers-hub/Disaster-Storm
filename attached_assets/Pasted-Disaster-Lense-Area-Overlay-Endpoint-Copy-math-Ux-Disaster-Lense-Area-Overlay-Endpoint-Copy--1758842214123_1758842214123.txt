This patch delivers:

1) Server — Area Overlay Route

Create apps/server/src/routes/area.ts:

import { Router } from 'express';
import sharp from 'sharp';


export const router = Router();


/*
POST /api/area/overlay
Body: {
  imageBase64: string,           // data URL of JPEG/PNG
  points: {x:number,y:number}[], // polygon in image pixel coords
  scalePxPerInch: number,        // calibration from QR
  color?: string,                // CSS color, e.g. 'gold' or '#ffd700'
  label?: string                 // optional custom label text
}
Returns: PNG with polygon overlay. Headers:
  X-Area-SqFt: numeric string (computed)
*/
router.post('/overlay', async (req, res) => {
  try {
    const { imageBase64, points = [], scalePxPerInch, color = '#ffd700', label } = req.body || {};
    if (!imageBase64) return res.status(400).json({ ok:false, error:'imageBase64 required' });
    if (!Array.isArray(points) || points.length < 3) return res.status(400).json({ ok:false, error:'points[] (>=3) required' });
    if (!scalePxPerInch) return res.status(400).json({ ok:false, error:'scalePxPerInch required' });


    const b64 = String(imageBase64).split(',').pop();
    const buf = Buffer.from(b64!, 'base64');


    // Compute area via shoelace in pixel^2 → in^2 → ft^2
    let sum = 0; for (let i=0; i<points.length; i++){ const a=points[i], b=points[(i+1)%points.length]; sum += (a.x*b.y - b.x*a.y); }
    const areaPx2 = Math.abs(sum)/2;
    const inchesPerPx = 1/Number(scalePxPerInch);
    const areaIn2 = areaPx2 * (inchesPerPx*inchesPerPx);
    const areaFt2 = areaIn2 / 144;


    // Build SVG overlay for polygon + label
    const pts = points.map(p=>`${p.x},${p.y}`).join(' ');
    const cx = points.reduce((s,p)=>s+p.x,0)/points.length; const cy = points.reduce((s,p)=>s+p.y,0)/points.length;
    const text = label || `${areaFt2.toFixed(2)} sq ft`;
    const svg = Buffer.from(`<?xml version="1.0"?>
      <svg xmlns="http://www.w3.org/2000/svg" width="100%" height="100%" viewBox="0 0 4000 4000" preserveAspectRatio="none">
        <polygon points="${pts}" fill="${color}33" stroke="${color}" stroke-width="4" />
        <rect x="${cx-80}" y="${cy-16}" width="160" height="26" rx="6" ry="6" fill="black" fill-opacity="0.6" />
        <text x="${cx}" y="${cy+3}" fill="white" font-size="18" font-family="Arial" text-anchor="middle">${text}</text>
      </svg>`);


    const out = await sharp(buf).composite([{ input: svg, gravity: 'northwest' }]).png().toBuffer();
    res.setHeader('Content-Type','image/png');
    res.setHeader('X-Area-SqFt', areaFt2.toFixed(4));
    res.send(out);
  } catch (e:any) {
    res.status(500).json({ ok:false, error: String(e.message||e) });
  }
});

2) Client — Copy Math UX

Add a tiny measurement store and Copy Math button logic. Below is a minimal, drop‑in pattern you can merge into your current Annotator.

2.1 Measurement store (append near top of Annotator component)
// Track last computed measurements for clipboard
const [lastMath, setLastMath] = useState<{ kind:'diameter'|'length'|'area', text:string }|null>(null);


async function copyMath(){
  if (!lastMath) return alert('Nothing to copy yet');
  try { await navigator.clipboard.writeText(lastMath.text); alert('Math copied to clipboard'); }
  catch { alert(lastMath.text); }
}
2.2 When you finish Diameter (after /api/tape/overlay response):
setLastMath({
  kind: 'diameter',
  text: `Diameter = pixel_distance / (px/in)\n= ${px.toFixed(2)} px / ${scalePxPerInch.toFixed(2)} px/in\n= ${(inches).toFixed(2)} in`
});

If you didn’t keep px, compute it from the two points before the API call: const px = Math.hypot(pt2.x-pt1.x, pt2.y-pt1.y).

2.3 When you finish Length:
setLastMath({
  kind: 'length',
  text: `Length = pixel_distance / (px/in)\n= ${px.toFixed(2)} px / ${scalePxPerInch.toFixed(2)} px/in\n= ${(inches).toFixed(2)} in  ≈  ${(feet).toFixed(2)} ft`
});
2.4 When you finish Area (client‑side polygon) OR when using the new server endpoint:

Client‑side version:

setLastMath({
  kind: 'area',
  text: `Area_px² = |∑(x_i*y_{i+1} − x_{i+1}*y_i)|/2 = ${areaPx2.toFixed(2)} px²\nArea_in² = Area_px² * (in/px)² = ${(areaIn2).toFixed(2)} in²\nArea_ft² = Area_in² / 144 = ${(areaFt2).toFixed(2)} ft²`
});

Server version (after calling /api/area/overlay):

const ft2 = Number(res.headers.get('X-Area-SqFt')||'0');
setLastMath({ kind: 'area', text: `Area_ft² (server) = ${ft2.toFixed(2)} ft² (QR-calibrated)` });
2.5 Add a Copy Math button in your Annotator UI actions
<button onClick={copyMath} className="rounded-xl border py-2">Copy Math</button>

Place it next to your existing actions (e.g., near “Suggest Labels” or under the Diameter/Length/Area buttons).

3) Optional — Use the Server Area Overlay instead of client overlay

Replace your completePolygon() body with:

async function completePolygon(){
  if (!scalePxPerInch) return alert('Calibrate first');
  if (polyPts.length < 3) return alert('Add at least 3 points');
  const base64 = await ensureBase64(selectedImage);
  const res = await fetch('/api/area/overlay', {
    method:'POST', headers:{'Content-Type':'application/json'},
    body: JSON.stringify({ imageBase64: base64, points: polyPts, scalePxPerInch, color:'#ffd700' })
  });
  if (!res.ok) return alert('Area overlay failed');
  const ft2 = Number(res.headers.get('X-Area-SqFt')||'0');
  const blob = await res.blob();
  const url = URL.createObjectURL(blob);
  setAreaPreview(url);
  const arrBuf = await blob.arrayBuffer();
  const b64 = `data:image/png;base64,${btoa(String.fromCharCode(...new Uint8Array(arrBuf)))}`;
  window.dispatchEvent(new CustomEvent('DL_ADD_OVERLAY_TO_REPORT', { detail: { imageBase64: b64, caption: `Area: ${ft2.toFixed(2)} sq ft (AI-estimated)` } }));
  setLastMath({ kind:'area', text: `Area_ft² (server) = ${ft2.toFixed(2)} ft² (QR-calibrated)` });
  setPolyPts([]);
}
