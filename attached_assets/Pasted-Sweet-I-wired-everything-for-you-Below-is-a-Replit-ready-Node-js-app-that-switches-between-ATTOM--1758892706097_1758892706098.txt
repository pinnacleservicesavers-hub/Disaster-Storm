// index.js - Replit-ready property + enrichment proxy (ATTOM <> Estated)
require('dotenv').config();
const express = require('express');
const fetch = require('node-fetch');
const NodeCache = require('node-cache');
const helmet = require('helmet');
const morgan = require('morgan');

const app = express();
app.use(express.json());
app.use(helmet());
app.use(morgan('combined'));

const cache = new NodeCache({ stdTTL: 60 * 60, checkperiod: 120 }); // 1 hour default cache

// Environment
const PROVIDER = (process.env.PROVIDER || 'estated').toLowerCase(); // 'attom' or 'estated'
const ATTOM_API_KEY = process.env.ATTOM_API_KEY || '';
const ESTATED_API_TOKEN = process.env.ESTATED_API_TOKEN || '';
const WHITEPAGES_API_KEY = process.env.WHITEPAGES_API_KEY || '';
const PORT = process.env.PORT || 3000;

// Cache helper
async function cached(key, ttlSeconds, fn) {
  const v = cache.get(key);
  if (v) return v;
  const value = await fn();
  cache.set(key, value, ttlSeconds);
  return value;
}

// Normalizer utility: try to extract common fields from provider responses
function normalizePropertyResponse(providerName, raw) {
  const out = { raw, ownerName: null, ownerMailing: null, parcelId: null, lastSaleDate: null, normalizedAddress: null };

  try {
    if (!raw) return out;

    if (providerName === 'attom') {
      // ATTOM responses often include property and assessor/ownership objects
      const prop = raw.property || raw.property && raw.property[0] || raw;
      // Try multiple common locations
      out.normalizedAddress = prop.address && (prop.address.address1 || prop.address.streetNumber || prop.address.line) || prop.address1 || prop.fullAddress || null;
      // owner info could be at raw.property.owner or raw.property.owner.name etc - defensive
      const owners = (raw.owners && raw.owners[0]) || (raw.property && raw.property[0] && raw.property[0].owners && raw.property[0].owners[0]);
      if (owners) {
        out.ownerName = owners.name || owners.ownerName || out.ownerName;
        out.ownerMailing = owners.mailingAddress || owners.mailing_address || out.ownerMailing;
      } else {
        // fallback fields
        out.ownerName = raw.owner && raw.owner.name || raw.ownerName || out.ownerName;
      }
      out.parcelId = raw.parcel_number || raw.property && raw.property[0] && raw.property[0].parcel_number || raw.property && raw.property.parcel_number || out.parcelId;
      out.lastSaleDate = raw.lastSale && raw.lastSale.date || raw.last_sale && raw.last_sale.date || raw.property && raw.property[0] && raw.property[0].lastSale && raw.property[0].lastSale.date || out.lastSaleDate;
    }

    if (providerName === 'estated') {
      // Estated v4 response shape: { data: { property: { ... } } } OR older: { property: { ... } }
      const property = raw.data && raw.data.property || raw.property || raw;
      out.normalizedAddress = property.address && (property.address.formatted || `${property.address.street} ${property.address.city}`) || property.address;
      if (property.owner) {
        out.ownerName = property.owner.name || property.owner;
        out.ownerMailing = property.owner.mailing_address || property.owner.mailing || out.ownerMailing;
      } else {
        // sometimes Estated lists owners array
        const ownerObj = property.owners && property.owners[0];
        if (ownerObj) {
          out.ownerName = ownerObj.name || out.ownerName;
          out.ownerMailing = ownerObj.mailing_address || out.ownerMailing;
        }
      }
      out.parcelId = property.parcel_number || property.parcel || out.parcelId;
      out.lastSaleDate = property.last_sale && property.last_sale.date || property.lastSaleDate || out.lastSaleDate;
    }

    return out;
  } catch (e) {
    console.warn('normalize error', e);
    return out;
  }
}

// Endpoint: GET /api/property?address=... OR provide street_address,city,state,zip for Estated
app.get('/api/property', async (req, res) => {
  try {
    const address = (req.query.address || '').trim();
    const street = (req.query.street_address || '').trim();
    const city = (req.query.city || '').trim();
    const state = (req.query.state || '').trim();
    const zip = (req.query.zip || '').trim();

    if (!address && !(street && city && state)) {
      return res.status(400).json({ error: 'Provide address OR street_address, city, state (zip optional)' });
    }

    const cacheKey = `property:${PROVIDER}:${address || `${street}|${city}|${state}|${zip}`}`;
    const raw = await cached(cacheKey, 60 * 60, async () => {
      if (PROVIDER === 'attom') {
        if (!ATTOM_API_KEY) throw new Error('ATTOM_API_KEY not set');
        // Use ATTOM BasicProfile endpoint (simple & broad)
        // Example endpoint from ATTOM docs:
        // https://api.gateway.attomdata.com/propertyapi/v1.0.0/property/basicprofile?address=4529%20Winona%20Court%2C%20Denver%2C%20CO
        const q = address || `${street}, ${city}, ${state} ${zip}`;
        const url = `https://api.gateway.attomdata.com/propertyapi/v1.0.0/property/basicprofile?address=${encodeURIComponent(q)}`;
        const r = await fetch(url, { headers: { 'accept': 'application/json', 'apikey': ATTOM_API_KEY }});
        if (!r.ok) {
          const txt = await r.text().catch(()=>String(r.status));
          throw new Error(`ATTOM request failed ${r.status} ${txt}`);
        }
        return r.json();
      } else { // default => estated
        if (!ESTATED_API_TOKEN) throw new Error('ESTATED_API_TOKEN not set');
        // Estated v4 expects token & street_address+city+state
        // Example: https://apis.estated.com/v4/property?token=TOKEN&street_address=151 Battle Green Dr&city=Rochester&state=NY
        if (address) {
          // attempt to split address into components (best-effort). If split fails, pass whole address to Estated 'query' param if supported.
          const parts = address.split(',');
          const streetPart = parts[0] || '';
          const cityPart = (parts[1] || '').trim().split(' ')[0] || '';
          const statePart = (parts[1] || '').trim().split(' ')[1] || '';
          const url = `https://apis.estated.com/v4/property?token=${ESTATED_API_TOKEN}&street_address=${encodeURIComponent(streetPart)}&city=${encodeURIComponent(cityPart)}&state=${encodeURIComponent(statePart)}`;
          const r = await fetch(url);
          if (!r.ok) {
            const txt = await r.text().catch(()=>String(r.status));
            throw new Error(`Estated request failed ${r.status} ${txt}`);
          }
          return r.json();
        } else {
          const url = `https://apis.estated.com/v4/property?token=${ESTATED_API_TOKEN}&street_address=${encodeURIComponent(street)}&city=${encodeURIComponent(city)}&state=${encodeURIComponent(state)}${zip ? `&zip_code=${encodeURIComponent(zip)}`: ''}`;
          const r = await fetch(url);
          if (!r.ok) {
            const txt = await r.text().catch(()=>String(r.status));
            throw new Error(`Estated request failed ${r.status} ${txt}`);
          }
          return r.json();
        }
      }
    });

    const normalized = normalizePropertyResponse(PROVIDER, raw);
    return res.json({ success: true, provider: PROVIDER, data: normalized });
  } catch (err) {
    console.error('property error', err);
    res.status(500).json({ error: err.message });
  }
});

// Endpoint: POST /api/enrich  { name, address } -> identity enrichment (Whitepages placeholder)
app.post('/api/enrich', async (req, res) => {
  try {
    const { name, address } = req.body;
    if (!name && !address) return res.status(400).json({ error: 'name or address required' });

    const cacheKey = `enrich:${name||''}:${address||''}`;
    const data = await cached(cacheKey, 60 * 60 * 24, async () => {
      if (!WHITEPAGES_API_KEY) throw new Error('WHITEPAGES_API_KEY not set (or you chose not to use enrichment)');
      // NOTE: replace this placeholder URL with the exact Whitepages Pro endpoint you have access to.
      // Example Whitepages Pro endpoints vary by account; check your Whitepages Pro docs after signup.
      const url = `https://proapi.whitepages.com/3.0/identity_check.json?api_key=${WHITEPAGES_API_KEY}&name=${encodeURIComponent(name||'')}&address=${encodeURIComponent(address||'')}`;
      const r = await fetch(url);
      if (!r.ok) {
        const t = await r.text().catch(()=>String(r.status));
        throw new Error(`Whitepages error ${r.status} ${t}`);
      }
      return r.json();
    });

    return res.json({ success: true, data });
  } catch (err) {
    console.error('enrich error', err);
    res.status(500).json({ error: err.message });
  }
});

app.get('/', (req, res) => res.send(`contractor-data-proxy running. Provider=${PROVIDER}`));

app.listen(PORT, () => console.log(`Server listening on port ${PORT} (Provider=${PROVIDER})`));

Post-deploy checklist & production tips

Use Replit Secrets — never commit keys.

Cache aggressively (we use 1 hour by default). Increase TTL for parcel/owner data (24 hr) if you want fewer calls.

Rate limiting & exponential backoff — add client-side throttle & server-side queuing if you scale to many contractors.

Audit log & consent capture — store who performed the lookup, timestamp, and purpose. Keep minimal PII retention policies.

Legal — don’t scrape TruthFinder or other consumer sites. Use licensed APIs. If you plan to call/text/email owners, ensure TCPA & CAN-SPAM compliance and check state privacy laws. If you plan to use data for regulated decisions (tenant/employer/credit), use an FCRA-compliant data provider (LexisNexis / specialized screening vendors).
