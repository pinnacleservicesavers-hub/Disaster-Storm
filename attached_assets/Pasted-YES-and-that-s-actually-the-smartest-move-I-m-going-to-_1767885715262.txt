✅ services/leadScorer.js
export function leadScorer(lead, tradeType) {
  let score = 0;
  const tags = [];

  if (!lead.website) { score += 15; tags.push("no_website"); }
  if (lead.rating && lead.rating < 4.0) { score += 10; tags.push("low_rating"); }
  if (lead.reviewCount && lead.reviewCount > 50) { score += 7; tags.push("high_volume"); }

  // Trade-specific boosts
  if (tradeType === "pressure_washing" && lead.category.includes("Restaurant")) score += 8;
  if (tradeType === "roofing" && lead.category.includes("Property Management")) score += 10;

  let label = "cold";
  if (score >= 35) label = "hot";
  else if (score >= 20) label = "warm";

  return { score, score_label: label, signal_tags: tags };
}

export async function outreachAgent({ lead, tradeType }) {

Create:
1) SMS pitch (max 300 chars)
2) Email pitch
3) Phone script
4) 4-message followup sequence
5) 3-tier offer stack (quick win, standard, premium)
Trade: ${tradeType}
Lead: ${JSON.stringify(lead)}
  `;

  const completion = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{ role: "user", content: prompt }],
  });

  return completion.choices[0].message.content;
}

def search_leads(payload: dict, user=Depends(contractor_only)):
    location = payload["location"]
    radius = payload["radius"]
    trade_type = payload["tradeType"]

    raw_leads = lead_hunter(location, radius, trade_type)

    results = []
    for lead in raw_leads:
        score_data = lead_scorer(lead, trade_type)
        comp = competition_detector(lead, location, radius)
        results.append({**lead, **score_data, **comp})

    return {"leads": results}
