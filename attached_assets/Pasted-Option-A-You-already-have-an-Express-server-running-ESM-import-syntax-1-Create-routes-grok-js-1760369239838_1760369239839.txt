Option A — You already have an Express server running (ESM: import syntax)

1) Create routes/grok.js

// routes/grok.js
import fetch from "node-fetch";

export const grokRouter = (app) => {
  app.post("/api/grok", async (req, res) => {
    try {
      const { prompt } = req.body;

      const r = await fetch("https://api.x.ai/v1/chat/completions", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${process.env.XAI_API_KEY}`,
        },
        body: JSON.stringify({
          model: "grok-beta", // use exact model name from your xAI Console
          messages: [{ role: "user", content: prompt || "Say hello" }],
          temperature: 0.3,
        }),
      });

      if (!r.ok) {
        const text = await r.text();
        return res.status(r.status).json({ error: text });
      }

      const data = await r.json();
      res.json(data);
    } catch (err) {
      res.status(500).json({ error: err?.message || "Unknown error" });
    }
  });
};

Option C — Next.js API Route (if you don’t run a separate Express server)

Create pages/api/grok.ts (or .js)

// pages/api/grok.ts
import type { NextApiRequest, NextApiResponse } from "next";

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  if (req.method !== "POST") return res.status(405).json({ error: "Method not allowed" });

  try {
    const { prompt } = req.body;

    const r = await fetch("https://api.x.ai/v1/chat/completions", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${process.env.XAI_API_KEY as string}`,
      },
      body: JSON.stringify({
        model: "grok-beta",
        messages: [{ role: "user", content: prompt || "Say hello" }],
        temperature: 0.3,
      }),
    });

    if (!r.ok) {
      const text = await r.text();
      return res.status(r.status).json({ error: text });
    }

    const data = await r.json();
    res.status(200).json(data);
  } catch (err: any) {
    res.status(500).json({ error: err?.message || "Unknown error" });
  }
}

Front-end helper (works with any option above)
export async function askGrok(prompt) {
  const r = await fetch("/api/grok", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ prompt }),
  });
  const data = await r.json();
  // adapt to xAI’s exact response shape
  return data?.choices?.[0]?.message?.content ?? JSON.stringify(data);
}
