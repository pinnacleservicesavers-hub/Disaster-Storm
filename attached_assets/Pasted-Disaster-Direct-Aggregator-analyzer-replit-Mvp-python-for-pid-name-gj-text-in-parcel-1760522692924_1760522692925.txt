def run_save(name: str, lat: float, lon: float, radius_km: float, hours_back: int, risk_score: float, events: List[Event], impacted: List[Dict[str,Any]]) -> int:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("INSERT INTO runs (name, created_at, lat, lon, radius_km, hours_back, risk_score, events_json, impacted_json) VALUES (?,?,?,?,?,?,?,?,?)",
                (name, int(datetime.utcnow().timestamp()), lat, lon, radius_km, hours_back, risk_score, json.dumps([e.model_dump() for e in events]), json.dumps(impacted)))
    con.commit()
    rid = cur.lastrowid
    con.close()
    return rid




def run_list() -> List[Dict[str,Any]]:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("SELECT id, name, created_at, risk_score FROM runs ORDER BY id DESC LIMIT 100")
    rows = cur.fetchall()
    con.close()
    return [{"id": r[0], "name": r[1], "created_at": r[2], "risk_score": r[3]} for r in rows]




def run_get(run_id: int) -> Optional[Dict[str,Any]]:
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("SELECT id, name, created_at, lat, lon, radius_km, hours_back, risk_score, events_json, impacted_json FROM runs WHERE id=?", (run_id,))
    r = cur.fetchone()
    con.close()
    if not r:
        return None
    return {"id": r[0], "name": r[1], "created_at": r[2], "lat": r[3], "lon": r[4], "radius_km": r[5], "hours_back": r[6], "risk_score": r[7], "events": json.loads(r[8] or '[]'), "impacted": json.loads(r[9] or '[]')}
