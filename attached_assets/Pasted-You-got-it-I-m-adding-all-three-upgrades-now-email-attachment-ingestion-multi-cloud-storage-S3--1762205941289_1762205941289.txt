1) Inbound email attachments → stored, linked to claim

Replace your SendGrid inbound route with this version that handles attachments of any count and saves them to your configured object storage. It also appends the stored file links to the claim’s notes.

app/routers/email_inbound_sendgrid.py
from fastapi import APIRouter, Depends, Form, Request
from sqlalchemy.orm import Session
from typing import Optional
import re

from app.core.db import SessionLocal
from app.models.entities import Claim, Carrier, Asset

router = APIRouter()

def get_db():
    db = SessionLocal()
    try: yield db
    finally: db.close()

def _parse_policy(text: str) -> Optional[str]:
    m = re.search(r"policy[:\s]+([A-Za-z0-9\-]+)", text or "", re.IGNORECASE)
    return m.group(1) if m else None

def _parse_asset_ref(text: str) -> Optional[str]:
    m = re.search(r"asset[_\s-]*id[:\s]+([A-Za-z0-9\-]+)", text or "", re.IGNORECASE)
    return m.group(1) if m else None

@router.post("/email/inbound/sendgrid")
async def inbound_sendgrid(request: Request, db: Session = Depends(get_db)):
    """
    Accepts SendGrid Inbound Parse multipart/form-data.
    Common fields: from, subject, text, html, attachments (int), attachment1..N
    """
    form = await request.form()
    from_email: str = str(form.get("from") or form.get("from_email") or "")
    subject: str = str(form.get("subject") or "")
    text: str = str(form.get("text") or "")
    html: str = str(form.get("html") or "")
    attachments = int(form.get("attachments") or 0)

    # Carrier from domain
    carrier_name = from_email.split("@")[-1].split(".")[0].title() if "@" in from_email else "Unknown"
    carrier = db.query(Carrier).filter(Carrier.name == carrier_name).first()
    if not carrier:
        carrier = Carrier(name=carrier_name)
        db.add(carrier); db.commit(); db.refresh(carrier)

    policy = _parse_policy(text) or subject.strip() or "(unknown)"
    asset_ref = _parse_asset_ref(text)
    asset = db.query(Asset).filter(Asset.ref_id == asset_ref).first() if asset_ref else None

    cl = Claim(
        carrier_id=carrier.id,
        policy_number=policy,
        asset_id=(asset.id if asset else None),
        status="open",
        notes=f"Inbound (SendGrid) from {from_email}: {subject}"
    )
    db.add(cl); db.commit(); db.refresh(cl)

    # Save attachments to object storage if configured
    stored_links = []
    if attachments:
        try:
            from app.utils.storage import store_bytes_and_get_url
            idx = 1
            for key, val in form.multi_items():
                # SendGrid names files as attachment1, attachment2...
                if key.startswith("attachment"):
                    upf = val  # UploadFile
                    if hasattr(upf, "read"):
                        data = await upf.read()
                        url = store_bytes_and_get_url(
                            data,
                            key=f"claims/{cl.id}/attachments/{upf.filename or f'attachment_{idx}'}",
                            content_type=upf.content_type or "application/octet-stream"
                        )
                        stored_links.append(url)
                        idx += 1
        except Exception as e:
            # Continue without failing claim creation
            stored_links.append(f"(storage error: {str(e)[:80]})")

    if stored_links:
        note_append = "\nAttachments:\n" + "\n".join(f"- {u}" for u in stored_links)
        cl.notes = (cl.notes or "") + note_append
        db.add(cl); db.commit()

    return {"ok": True, "created_claim_id": cl.id, "attachments": stored_links}


This route reads the whole form (so you don’t need to predeclare attachment1..N). If storage isn’t configured yet, the claim still gets created; errors are noted in attachments.

2.3 Unified storage helpers
app/utils/storage.py
import os
from datetime import datetime, timedelta
from typing import Optional

from app.core.config import settings

# ---------- S3 ----------
def _s3_client():
    import boto3
    session = boto3.session.Session(
        aws_access_key_id=settings.aws_access_key_id,
        aws_secret_access_key=settings.aws_secret_access_key,
        region_name=settings.s3_region
    )
    return session.client("s3")

def _s3_put_and_url(data: bytes, key: str, content_type: str) -> str:
    s3 = _s3_client()
    s3.put_object(Bucket=settings.s3_bucket, Key=key, Body=data, ContentType=content_type)
    return s3.generate_presigned_url(
        ClientMethod="get_object",
        Params={"Bucket": settings.s3_bucket, "Key": key},
        ExpiresIn=7 * 24 * 3600
    )

def _s3_list(prefix: str) -> list[dict]:
    s3 = _s3_client()
    resp = s3.list_objects_v2(Bucket=settings.s3_bucket, Prefix=prefix)
    items = []
    for c in resp.get("Contents", []):
        items.append({"key": c["Key"], "size": c["Size"], "last_modified": c["LastModified"].isoformat()})
    return items

# ---------- GCS ----------
def _gcs_client():
    from google.cloud import storage
    return storage.Client()

def _gcs_put_and_url(data: bytes, key: str, content_type: str) -> str:
    client = _gcs_client()
    bucket = client.bucket(settings.gcs_bucket)
    blob = bucket.blob(key)
    blob.upload_from_string(data, content_type=content_type)
    # Signed URL 7 days
    url = blob.generate_signed_url(expiration=timedelta(days=7), method="GET")
    return url

def _gcs_list(prefix: str) -> list[dict]:
    client = _gcs_client()
    bucket = client.bucket(settings.gcs_bucket)
    blobs = client.list_blobs(bucket, prefix=prefix)
    items = []
    for b in blobs:
        items.append({"key": b.name, "size": b.size, "last_modified": b.updated.isoformat() if b.updated else None})
    return items

# ---------- Azure ----------
def _az_client():
    from azure.storage.blob import BlobServiceClient
    conn = settings.azure_connection_string or os.getenv("AZURE_STORAGE_CONNECTION_STRING")
    return BlobServiceClient.from_connection_string(conn)

def _az_put_and_url(data: bytes, key: str, content_type: str) -> str:
    from azure.storage.blob import ContentSettings, generate_blob_sas, AccountSasPermissions
    svc = _az_client()
    container = svc.get_container_client(settings.azure_container)
    container.upload_blob(name=key, data=data, overwrite=True, content_settings=ContentSettings(content_type=content_type))
    # SAS URL (7 days)
    from datetime import datetime, timedelta
    account_name = svc.account_name
    # For connection-string SAS, build via user delegation token is more complex; simplest is public container
    # or use account key-based SAS if available in env variables.
    # As a fallback, return the public URL path; secure your container appropriately.
    return f"{svc.url}/{settings.azure_container}/{key}"

def _az_list(prefix: str) -> list[dict]:
    svc = _az_client()
    container = svc.get_container_client(settings.azure_container)
    items = []
    for b in container.list_blobs(name_starts_with=prefix):
        items.append({"key": b.name, "size": b.size, "last_modified": b.last_modified.isoformat() if b.last_modified else None})
    return items

# ---------- Unified API ----------
def store_bytes_and_get_url(data: bytes, key: str, content_type: str = "application/octet-stream") -> str:
    provider = (settings.storage_provider or "").lower()
    if provider == "s3":
        return _s3_put_and_url(data, key, content_type)
    if provider == "gcs":
        return _gcs_put_and_url(data, key, content_type)
    if provider == "azure":
        return _az_put_and_url(data, key, content_type)
    raise RuntimeError("Storage provider not configured")

def store_pdf_and_get_url(pdf_bytes: bytes, filename: str) -> str:
    key = f"fnol/{datetime.utcnow().strftime('%Y/%m/%d')}/{filename}"
    return store_bytes_and_get_url(pdf_bytes, key, content_type="application/pdf")

def list_prefix(prefix: str) -> list[dict]:
    provider = (settings.storage_provider or "").lower()
    if provider == "s3":
        return _s3_list(prefix)
    if provider == "gcs":
        return _gcs_list(prefix)
    if provider == "azure":
        return _az_list(prefix)
    return []


S3/GCS return signed URLs; the Azure example returns a blob URL (configure your container access as needed, or extend with SAS if you have account key).

3) Admin page to list/download archived FNOL PDFs
3.1 API: list stored FNOL files
app/routers/reports_admin.py
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from app.core.db import SessionLocal
from app.utils.storage import list_prefix, store_pdf_and_get_url

router = APIRouter()

def get_db():
    db = SessionLocal()
    try: yield db
    finally: db.close()

@router.get("/reports/fnol/list")
def list_fnol():
    items = list_prefix("fnol/")
    # Shape the list uniformly; clients can resolve URLs via a temporary signed URL call if desired.
    return {"count": len(items), "items": items}

@router.get("/reports/fnol/url")
def get_temp_url(key: str):
    # Quick helper to mint a fresh signed URL (S3/GCS) or return blob URL (Azure)
    # This reads nothing; it just generates a URL for the provided key.
    return {"url": store_pdf_and_get_url(b"", key.split("/")[-1])}  # reuse signer with same filename path


If you don’t like the store_pdf_and_get_url trick for URL signing, add a separate signer in storage.py that takes a key and returns a signed URL without uploading.
