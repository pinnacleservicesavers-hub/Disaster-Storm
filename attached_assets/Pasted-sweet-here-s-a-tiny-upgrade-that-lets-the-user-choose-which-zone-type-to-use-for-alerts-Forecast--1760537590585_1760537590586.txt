2) Add helpers (JS)

Drop these near your other helpers (right after buildIdText is perfect):

function extractZoneIdFromUrl(url) {
  if (!url) return null;
  const parts = String(url).split("/");
  return parts[parts.length - 1] || null;
}

// Return a zone ID based on requested type
function getZoneIdByType(payload, type /* 'forecast' | 'county' | 'fire' */) {
  const z = payload?.zones || {};
  if (type === "forecast") return extractZoneIdFromUrl(z.forecastZone);
  if (type === "county")   return extractZoneIdFromUrl(z.county);
  if (type === "fire")     return extractZoneIdFromUrl(z.fireWeatherZone);
  return null;
}

3) Upgrade the zone-alert fetcher (JS)

Replace your existing fetchZoneAlertsForCurrent() with this version that reads the dropdown and falls back smartly:

async function fetchZoneAlertsForCurrent() {
  const zoneWrap = document.getElementById("zone-alerts");
  const sel = document.getElementById("zoneTypeSel");
  const type = sel ? sel.value : "forecast";

  if (!lastPayload) {
    zoneWrap.innerHTML = `<div class="card">No location loaded yet. Pick a city or use GPS first.</div>`;
    return;
  }

  // Try the chosen type; if missing, fall back to forecast -> county -> fire
  const order = (() => {
    if (type === "forecast") return ["forecast", "county", "fire"];
    if (type === "county")   return ["county", "forecast", "fire"];
    return ["fire", "forecast", "county"];
  })();

  let zoneId = null, usedType = null;
  for (const t of order) {
    const id = getZoneIdByType(lastPayload, t);
    if (id) { zoneId = id; usedType = t; break; }
  }

  if (!zoneId) {
    zoneWrap.innerHTML = `<div class="card">Couldn’t determine a zone ID for this location.</div>`;
    return;
  }

  zoneWrap.innerHTML = `<div class="card skeleton" style="height:80px"></div>`;

  try {
    const r = await fetch(`/api/alerts?zone=${encodeURIComponent(zoneId)}`);
    if (!r.ok) throw new Error(`HTTP ${r.status}`);
    const data = await r.json();
    renderZoneAlertsList(data.items || []);
    showToast(`Loaded ${usedType} alerts for ${zoneId}`);
  } catch (e) {
    zoneWrap.innerHTML = `
      <div class="error">
        <div class="card">
          <div class="headline">Zone alerts failed</div>
          <div class="small">${e.message}</div>
          <div class="error-actions">
            <button class="retry-btn" data-target="${lastMode}">Retry</button>
          </div>
        </div>
      </div>`;
  }
}
