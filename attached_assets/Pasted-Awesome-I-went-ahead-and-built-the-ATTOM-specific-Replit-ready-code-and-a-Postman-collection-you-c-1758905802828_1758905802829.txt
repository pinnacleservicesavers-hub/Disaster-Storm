1) ATTOM-ready index.js (copy → Replit index.js)
// index.js - ATTOM-first property proxy for Replit
require('dotenv').config();
const express = require('express');
const fetch = require('node-fetch');
const NodeCache = require('node-cache');
const helmet = require('helmet');
const morgan = require('morgan');

const app = express();
app.use(express.json());
app.use(helmet());
app.use(morgan('combined'));

const cache = new NodeCache({ stdTTL: 60 * 60, checkperiod: 120 }); // 1 hour default

const PROVIDER = (process.env.PROVIDER || 'attom').toLowerCase();
const ATTOM_KEY = process.env.ATTOM_API_KEY || '';
const WHITEPAGES_KEY = process.env.WHITEPAGES_API_KEY || '';
const PORT = process.env.PORT || 3000;

// helper: cache wrapper
async function cached(key, ttlSeconds, fn) {
  const v = cache.get(key);
  if (v) return v;
  const val = await fn();
  cache.set(key, val, ttlSeconds);
  return val;
}

// defensive helper to safely read nested fields
function get(o, path, fallback = null) {
  try {
    return path.split('.').reduce((acc, p) => (acc && acc[p] !== undefined ? acc[p] : undefined), o) ?? fallback;
  } catch (e) { return fallback; }
}

// ATTOM-specific request + normalization
async function fetchAttomByAddress(address) {
  if (!ATTOM_KEY) throw new Error('ATTOM_API_KEY not set in environment');
  const url = `https://api.gateway.attomdata.com/propertyapi/v1.0.0/property/detail?address=${encodeURIComponent(address)}`;
  const res = await fetch(url, { headers: { 'accept': 'application/json', 'apikey': ATTOM_KEY }});
  if (!res.ok) {
    const txt = await res.text().catch(()=>String(res.status));
    throw new Error(`ATTOM error ${res.status}: ${txt}`);
  }
  const json = await res.json();
  return json;
}

// Normalize ATTOM response into a stable object your frontend can use
function normalizeAttom(raw) {
  // ATTOM often returns property: [ { ... } ] plus owners in owners or property[0].owners
  const out = {
    raw,
    provider: 'attom',
    address: null,
    ownerNames: [],
    ownerMailing: null,
    parcelId: null,
    lastSale: null,
    assessedValue: null,
    taxYear: null,
    taxAmount: null,
    latitude: null,
    longitude: null
  };

  // Try common shapes
  const prop = get(raw, 'property.0') || get(raw, 'property') || raw;
  // Address attempts
  out.address = get(prop, 'address.address1') || get(prop, 'address.line') || get(prop, 'address.fullAddress') || null;
  // Parcel
  out.parcelId = get(prop, 'identifiers.parcelNumber') || get(prop, 'parcel_number') || get(prop, 'identifiers.assessorParcelNumber') || get(raw, 'parcel_number') || null;
  // Owners: ATTOM may have owners array under raw.owners or prop.owners
  const ownersArr = get(raw, 'owners') || get(prop, 'owners') || [];
  if (Array.isArray(ownersArr) && ownersArr.length) {
    ownersArr.forEach(o => {
      const n = o.name || o.ownerName || `${o.firstName||''} ${o.lastName||''}`.trim();
      if (n) out.ownerNames.push(n);
      if (!out.ownerMailing) out.ownerMailing = o.mailingAddress || o.mailing_address || null;
    });
  } else {
    // fallback single owner fields
    const fallbackOwner = get(prop, 'owner') || get(raw, 'owner') || null;
    if (fallbackOwner) {
      const n = fallbackOwner.name || fallbackOwner.ownerName || null;
      if (n) out.ownerNames.push(n);
      out.ownerMailing = fallbackOwner.mailingAddress || fallbackOwner.mailing_address || out.ownerMailing;
    }
  }

  // Last sale
  out.lastSale = get(prop, 'lastSale') || get(prop, 'sales.0') || get(raw, 'last_sale') || null;

  // Assessed / tax info (defensive)
  out.assessedValue = get(prop, 'assessment.assessedValue') || get(prop, 'assessment.assessed_value') || get(prop, 'tax.assessedValue') || null;
  out.taxYear = get(prop, 'tax.taxYear') || get(prop, 'assessment.taxYear') || null;
  out.taxAmount = get(prop, 'tax.amount') || get(prop, 'tax.taxAmount') || null;

  // Lat/Lon
  out.latitude = get(prop, 'location.latitude') || get(prop, 'address.latitude') || null;
  out.longitude = get(prop, 'location.longitude') || get(prop, 'address.longitude') || null;

  // If still empty, try top-level lookups
  if (!out.ownerNames.length) {
    const maybeName = get(raw, 'ownerName') || get(raw, 'owner') || null;
    if (maybeName) out.ownerNames.push(maybeName);
  }

  return out;
}

// GET /api/property?address=...
app.get('/api/property', async (req, res) => {
  try {
    const address = (req.query.address || '').trim();
    if (!address) return res.status(400).json({ error: 'address query param required (e.g. ?address=5385 Westwood Drive Columbus GA)' });

    const cacheKey = `attom:prop:${address}`;
    const raw = await cached(cacheKey, 60 * 60 * 24, async () => {
      // try ATTOM first
      return fetchAttomByAddress(address);
    });

    const normalized = normalizeAttom(raw);
    return res.json({ success: true, provider: 'attom', data: normalized });
  } catch (err) {
    console.error('property error', err);
    res.status(500).json({ error: err.message });
  }
});

    const result = await cached(cacheKey, 60 * 60 * 24, async () => {
      if (!WHITEPAGES_KEY) return { note: 'WHITEPAGES_API_KEY not set; enrichment disabled' };

      const r = await fetch(url);

  } catch (err) {
    console.error('enrich error', err);
    res.status(500).json({ error: err.message });
  }
});

app.get('/', (req, res) => res.send(`ATTOM proxy running (provider=attom).`));
app.listen(PORT, () => console.log(`Server listening on ${PORT} (PROVIDER=attom)`));
