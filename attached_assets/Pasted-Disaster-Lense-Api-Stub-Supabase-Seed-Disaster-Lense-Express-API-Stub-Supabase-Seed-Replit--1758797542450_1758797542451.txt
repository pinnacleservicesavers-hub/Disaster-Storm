9) apps/server/src/lib/pdf.ts (minimal PDF using pdf-lib)
import { PDFDocument, StandardFonts, rgb } from 'pdf-lib';


export async function buildSimplePdf(opts: { title: string; projectTitle: string; items: { caption: string }[]; }){
  const pdf = await PDFDocument.create();
  const font = await pdf.embedFont(StandardFonts.Helvetica);
  const page = pdf.addPage([612, 792]);
  const { width, height } = page.getSize();
  const margin = 40;


  page.drawText(opts.title, { x: margin, y: height - margin - 20, size: 18, font });
  page.drawText(`Project: ${opts.projectTitle}`, { x: margin, y: height - margin - 44, size: 12, font, color: rgb(0.2,0.2,0.2) });


  let y = height - margin - 80;
  opts.items.forEach((it, idx) => {
    page.drawText(`${idx+1}. ${it.caption}`, { x: margin, y, size: 11, font });
    y -= 18;
  });


  page.drawText(`Generated: ${new Date().toISOString()}`, { x: margin, y: margin, size: 9, font, color: rgb(0.4,0.4,0.4) });
  return await pdf.save();
}
