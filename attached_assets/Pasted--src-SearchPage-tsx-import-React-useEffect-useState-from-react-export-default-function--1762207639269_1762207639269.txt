// src/SearchPage.tsx
import React, { useEffect, useState } from 'react';


export default function SearchPage(){
const [q,setQ] = useState('');
const [trade,setTrade] = useState('');
const [area,setArea] = useState('');
const [state,setState] = useState('');
const [minScore,setMinScore] = useState(60);
const [hasLicense,setHasLicense] = useState(false);
const [hasInsurance,setHasInsurance] = useState(false);
const [rows,setRows] = useState<any[]>([]);


async function run(){
const params = new URLSearchParams({ q, trade, area, state, minScore: String(minScore), hasLicense: String(hasLicense), hasInsurance: String(hasInsurance) });
const res = await fetch(`/api/search?${params.toString()}`);
setRows(await res.json());
}
useEffect(()=>{ run(); }, []);


return (
<div className="max-w-5xl mx-auto p-6">
<div className="grid md:grid-cols-6 gap-2 mb-4">
<input className="border rounded-lg px-3 py-2" placeholder="Search" value={q} onChange={e=>setQ(e.target.value)} />
<input className="border rounded-lg px-3 py-2" placeholder="Trade" value={trade} onChange={e=>setTrade(e.target.value)} />
<input className="border rounded-lg px-3 py-2" placeholder="Area (city)" value={area} onChange={e=>setArea(e.target.value)} />
<input className="border rounded-lg px-3 py-2" placeholder="State (e.g. TX)" value={state} onChange={e=>setState(e.target.value)} />
<input type="number" className="border rounded-lg px-3 py-2" placeholder="Min score" value={minScore} onChange={e=>setMinScore(Number(e.target.value))} />
<button onClick={run} className="px-4 py-2 rounded-xl bg-blue-600 text-white">Search</button>
</div>
<div className="flex gap-4 mb-4">
<label className="text-sm"><input type="checkbox" checked={hasLicense} onChange={e=>setHasLicense(e.target.checked)} /> License verified</label>
<label className="text-sm"><input type="checkbox" checked={hasInsurance} onChange={e=>setHasInsurance(e.target.checked)} /> Insurance verified</label>
</div>


<div className="grid md:grid-cols-2 gap-4">
{rows.map(r => (
<a key={r.slug} href={`/c/${r.slug}`} className="border rounded-2xl p-4 hover:shadow">
<div className="text-xl font-semibold">{r.company_name || r.companyName}</div>
<div className="text-sm text-gray-600">{r.address}</div>
<div className="text-sm mt-1">Rank: {r.rank} · Score: {r.score}</div>
</a>
))}
</div>
</div>
);
}
